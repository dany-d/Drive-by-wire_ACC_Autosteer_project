132/*FUNCTION**********************************************************************



133 *



134 * Function Name : FTM_DRV_Init



135 * Description   : Initializes the FTM driver and get the clock frequency value



136 * which select one of three possible clock sources for the FTM counter.



137 *



138 * Implements    : FTM_DRV_Init_Activity



139 *END**************************************************************************/



140status_t FTM_DRV_Init(uint32_t instance,



141                      const ftm_user_config_t * info,



142                      ftm_state_t * state)



143{



144  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);



145  DEV_ASSERT(info != NULL);



146  DEV_ASSERT(state != NULL);



147  FTM_Type * ftmBase = g_ftmBase[instance];



148  status_t status = STATUS_SUCCESS;



149  uint8_t index = 0U;



150



151  /* Check if this instance is already initialized */



152  if (ftmStatePtr[instance] != NULL) {



153    status = STATUS_ERROR;



154  } else {



155    /* Configure state structure. */



156    state->ftmClockSource = info->ftmClockSource;



157    state->ftmMode = FTM_MODE_NOT_INITIALIZED;



158    state->ftmPeriod = 0U;



159    ftmStatePtr[instance] = state;



160    for (index = 0U; index < FEATURE_FTM_CHANNEL_COUNT; index++) {



161      state->measurementResults[index] = 0U;



162      state->channelsCallbacksParams[index] = NULL;



163      state->channelsCallbacks[index] = NULL;



164      state->enableNotification[index] = false;



165    }



166



167    /* The reset operation doesn't care about write protection. FTM_DRV_Reset will



168     * disable this protection.*/



169    FTM_DRV_Reset(ftmBase);



170    FTM_DRV_InitModule(ftmBase, info->ftmPrescaler);



171



172    /* Get clock name used to configure the FlexTimer module */



173    state->ftmSourceClockFrequency = FTM_DRV_GetFrequency(instance);



174



175    /* Check the functional clock is selected for FTM */



176    DEV_ASSERT(state->ftmSourceClockFrequency > 0U);



177  }



178



179  if (STATUS_SUCCESS == status) {



180    /* Check if the mode operation in PWM mode */



181    if ((FTM_MODE_EDGE_ALIGNED_PWM == info->ftmMode) ||



182        (FTM_MODE_CEN_ALIGNED_PWM == info->ftmMode) || (FTM_MODE_OUTPUT_COMPARE ==



183         info->ftmMode)) {



184      /* Configure sync for between registers and buffers */



185      status = FTM_DRV_SetSync(instance, &(info->syncMethod));



186    }



187



188    /* Enable the generation of initialization trigger on chip module */



189    FTM_DRV_SetInitTriggerCmd(ftmBase, info->enableInitializationTrigger);



190    FTM_DRV_SetBdmMode(ftmBase, info->BDMMode);



191



192    /* Check if enable interrupt in counter mode */



193    if (info->isTofIsrEnabled) {



194      FTM_DRV_SetTimerOverflowInt(ftmBase, true);



195      INT_SYS_EnableIRQ(g_ftmOverflowIrqId[instance]);



196    } else {



197      FTM_DRV_SetTimerOverflowInt(ftmBase, false);



198      INT_SYS_DisableIRQ(g_ftmOverflowIrqId[instance]);



199    }



200  }



201



202  return status;



203}

402/*FUNCTION**********************************************************************



403 *



404 * Function Name : FTM_DRV_SetSync



405 * Description   : This function configure the synchronization for PWM register



406 * (CnV, MOD, CINT, HCR, OUTMASK).If this function is used whit wrong parameters



407 * it's possible to generate wrong waveform. Registers synchronization need to



408 * be configured for PWM and output compare mode.



409 *



410 * Implements : FTM_DRV_SetSync_Activity



411 *END**************************************************************************/



412status_t FTM_DRV_SetSync(uint32_t instance,



413  const ftm_pwm_sync_t * param)



414{



415  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);



416  DEV_ASSERT(param != NULL);



417  FTM_Type * ftmBase = g_ftmBase[instance];



418  status_t retStatus = STATUS_SUCCESS;



419  bool hardwareSync = param->hardwareSync0 || param->hardwareSync1 ||



420    param->hardwareSync2;



421



422  /* Software and hardware triggers are not allowed in the same time */



423  if ((param->softwareSync && hardwareSync) || (true != (param->softwareSync ||



424        hardwareSync))) {



425    retStatus = STATUS_ERROR;



426  } else if (param->softwareSync) {



427    /* Configure sync for OUTMASK register */



428    FTM_DRV_SetOutmaskSoftwareSyncModeCmd(ftmBase, true);



429



430    /* Configure sync for INVCTRL register */



431    FTM_DRV_SetInvctrlSoftwareSyncModeCmd(ftmBase, true);



432



433    /* Configure sync for SWOCTRL register */



434    FTM_DRV_SetSwoctrlSoftwareSyncModeCmd(ftmBase, true);



435



436    /* Configure sync for MOD, HCR, CNTIN, and CnV registers */



437    FTM_DRV_SetModCntinCvSoftwareSyncModeCmd(ftmBase, true);



438



439    /* Configure synchronization method (waiting next loading point or now) */



440    FTM_DRV_SetCounterSoftwareSyncModeCmd(ftmBase, param->syncPoint);



441  } else {



442    /* Configure sync for OUTMASK register */



443    FTM_DRV_SetOutmaskHardwareSyncModeCmd(ftmBase, true);



444



445    /* Configure sync for INVCTRL register */



446    FTM_DRV_SetInvctrlHardwareSyncModeCmd(ftmBase, true);



447



448    /* Configure sync for SWOCTRL register */



449    FTM_DRV_SetSwoctrlHardwareSyncModeCmd(ftmBase, true);



450



451    /* Configure sync for MOD, HCR, CNTIN, and CnV registers */



452    FTM_DRV_SetModCntinCvHardwareSyncModeCmd(ftmBase, true);



453



454    /* Configure synchronization method (waiting next loading point or now) */



455    FTM_DRV_SetCounterHardwareSyncModeCmd(ftmBase, (bool)param->syncPoint);



456  }



457



458  if (STATUS_SUCCESS == retStatus) {



459    /* Enhanced PWM sync is used */



460    FTM_DRV_SetPwmSyncModeCmd(ftmBase, true);



461



462    /* Configure trigger source for sync */



463    FTM_DRV_SetHardwareSyncTriggerSrc(ftmBase, 0U, param->hardwareSync0);



464    FTM_DRV_SetHardwareSyncTriggerSrc(ftmBase, 1U, param->hardwareSync1);



465    FTM_DRV_SetHardwareSyncTriggerSrc(ftmBase, 2U, param->hardwareSync2);



466



467    /* Configure loading points */



468    FTM_DRV_SetMaxLoadingCmd(ftmBase, param->maxLoadingPoint);



469    FTM_DRV_SetMinLoadingCmd(ftmBase, param->minLoadingPoint);



470



471    /* Configure sync for OUTMASK register */



472    FTM_DRV_SetOutmaskPwmSyncModeCmd(ftmBase, (bool)param->maskRegSync);



473



474    /* Configure sync for INVCTRL register */



475    FTM_DRV_SetInvctrlPwmSyncModeCmd(ftmBase, param->inverterSync);



476



477    /* Configure sync for SWOCTRL register */



478    FTM_DRV_SetSwoctrlPwmSyncModeCmd(ftmBase, param->outRegSync);



479



480    /* Configure sync for MOD, HCR, CNTIN, and CnV registers */



481    FTM_DRV_SetCntinPwmSyncModeCmd(ftmBase, param->initCounterSync);



482



483    /* Configure if FTM clears TRIGj (j=0,1,2) when the hardware trigger j is detected. */



484    FTM_DRV_SetHwTriggerSyncModeCmd(ftmBase, param->autoClearTrigger);



485  }



486



487  return retStatus;



488}

490/*FUNCTION**********************************************************************



491 *



492 * Function Name : FTM_DRV_EnableInterrupts



493 * Description   : This function will enable the generation a list of interrupts.



494 * It includes the FTM overflow interrupts, the reload point interrupt, the fault



495 * interrupt and the channel (n) interrupt.



496 *



497 * Implements : FTM_DRV_EnableInterrupts_Activity



498 *END**************************************************************************/



499status_t FTM_DRV_EnableInterrupts(uint32_t instance,



500  uint32_t interruptMask)



501{



502  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);



503  FTM_Type * ftmBase = g_ftmBase[instance];



504  uint32_t chnlInts = (interruptMask & 0x000000FFU);



505  uint8_t channel = 0U;



506



507  /* Enable the timer overflow interrupt */



508  if ((interruptMask & (uint32_t)FTM_TIME_OVER_FLOW_INT_ENABLE) != 0x0U) {



509    FTM_DRV_SetTimerOverflowInt(ftmBase, true);



510    INT_SYS_EnableIRQ(g_ftmOverflowIrqId[instance]);



511  }



512



513  /* Enable the fault interrupt */



514  if ((interruptMask & (uint32_t)FTM_FAULT_INT_ENABLE) != 0x0U) {



515    FTM_DRV_SetFaultInt(ftmBase, true);



516    INT_SYS_EnableIRQ(g_ftmFaultIrqId[instance]);



517  }



518



519  /* Enable the reload interrupt */



520  if ((interruptMask & (uint32_t)FTM_RELOAD_INT_ENABLE) != 0x0U) {



521    FTM_DRV_SetReIntEnabledCmd(ftmBase, true);



522    INT_SYS_EnableIRQ(g_ftmReloadIrqId[instance]);



523  }



524



525  /* Enable the channel interrupts */



526  while (chnlInts != 0U) {



527    if ((chnlInts & 0x1U) != 0x0U) {



528      FTM_DRV_EnableChnInt(ftmBase, channel);



529      INT_SYS_EnableIRQ(g_ftmIrqId[instance][channel]);



530    }



531



532    channel++;



533    chnlInts = chnlInts >> 1U;



534  }



535



536  return STATUS_SUCCESS;



537}

539/*FUNCTION**********************************************************************



540 *



541 * Function Name : FTM_DRV_DisableInterrupts



542 * Description   : This function is used to disable some interrupts.



543 *



544 * Implements : FTM_DRV_DisableInterrupts_Activity



545 *END**************************************************************************/



546void FTM_DRV_DisableInterrupts(uint32_t instance,



547  uint32_t interruptMask)



548{



549  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);



550  FTM_Type * ftmBase = g_ftmBase[instance];



551  uint32_t chnlInts = (interruptMask & 0x000000FFU);



552  uint8_t channel = 0U;



553



554  /* Disable the timer overflow interrupt */



555  if ((interruptMask & (uint32_t)FTM_TIME_OVER_FLOW_INT_ENABLE) != 0x0U) {



556    FTM_DRV_SetTimerOverflowInt(ftmBase, false);



557    INT_SYS_DisableIRQ(g_ftmOverflowIrqId[instance]);



558  }



559



560  /* Disable the fault interrupt */



561  if ((interruptMask & (uint32_t)FTM_FAULT_INT_ENABLE) != 0x0U) {



562    FTM_DRV_SetFaultInt(ftmBase, false);



563    INT_SYS_DisableIRQ(g_ftmFaultIrqId[instance]);



564  }



565



566  /* Disable the reload interrupt */



567  if ((interruptMask & (uint32_t)FTM_RELOAD_INT_ENABLE) != 0x0U) {



568    FTM_DRV_SetReIntEnabledCmd(ftmBase, false);



569    INT_SYS_DisableIRQ(g_ftmReloadIrqId[instance]);



570  }



571



572  /* Disable the channel interrupts */



573  while (chnlInts != 0U) {



574    if ((chnlInts & 0x1U) != 0x0U) {



575      FTM_DRV_DisableChnInt(ftmBase, channel);



576      INT_SYS_DisableIRQ(g_ftmIrqId[instance][channel]);



577    }



578



579    channel++;



580    chnlInts = chnlInts >> 1U;



581  }



582}

584/*FUNCTION**********************************************************************



585 *



586 * Function Name : FTM_DRV_GetEnabledInterrupts



587 * Description   : This function will get the enabled FTM interrupts.



588 *



589 * Implements : FTM_DRV_GetEnabledInterrupts_Activity



590 *END**************************************************************************/



591uint32_t FTM_DRV_GetEnabledInterrupts(uint32_t instance)



592{



593  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);



594  FTM_Type const * ftmBase = g_ftmBase[instance];



595  uint32_t enabledInterrupts = 0U;



596  uint8_t channel = FEATURE_FTM_CHANNEL_COUNT;



597



598  /* Check if timer overflow interrupt is enabled */



599  if (FTM_DRV_IsOverflowIntEnabled(ftmBase) == true) {



600    enabledInterrupts |= (uint32_t)FTM_TIME_OVER_FLOW_INT_ENABLE;



601  }



602



603  /* Check if fault interrupt is enabled */



604  if (FTM_DRV_IsFaultIntEnabled(ftmBase) == true) {



605    enabledInterrupts |= (uint32_t)FTM_FAULT_INT_ENABLE;



606  }



607



608  /* Check if the reload interrupt is enabled */



609  if (FTM_DRV_IsReloadIntEnabled(ftmBase) == true) {



610    enabledInterrupts |= (uint32_t)FTM_RELOAD_INT_ENABLE;



611  }



612



613  /* Check if the channel interrupts are enabled */



614  while (channel > 0U) {



615    channel--;



616    if (FTM_DRV_IsChnIntEnabled(ftmBase, channel) == true) {



617      enabledInterrupts |= (1UL << (uint32_t)channel);



618    }



619  }



620



621  return enabledInterrupts;



622}

636  uint32_t statusFlags = 0U;



637



638  /* Check the timer flag */



639  if (FTM_DRV_HasTimerOverflowed(ftmBase) == true) {



640    statusFlags |= (uint32_t)FTM_TIME_OVER_FLOW_FLAG;



641  }



642



643  /* Check fault flag */



644  if (FTM_DRV_GetDetectedFaultInput(ftmBase) == true) {



645    statusFlags |= (uint32_t)FTM_FAULT_FLAG;



646  }



647



648  /* Check reload flag */



649  if (FTM_DRV_GetReloadFlag(ftmBase) == true) {



650    statusFlags |= (uint32_t)FTM_RELOAD_FLAG;



651  }



652



653  /* Check channel trigger flag */



654  if (FTM_DRV_IsChnTriggerGenerated(ftmBase) == true) {



655    statusFlags |= (uint32_t)FTM_CHANNEL_TRIGGER_FLAG;



656  }



657



658  /* Lower 8 bits contain the channel status flags */



659  for (channel = 0U; channel < FEATURE_FTM_CHANNEL_COUNT; channel++) {



660    if (FTM_DRV_HasChnEventOccurred(ftmBase, channel) == true) {



661      statusFlags |= (1UL << (uint32_t)channel);



662    }



663  }



664



665  return statusFlags;



666}

668/*FUNCTION**********************************************************************



669 *



670 * Function Name : FTM_DRV_ClearStatusFlags



671 * Description   : This function is used to clear the FTM status flags.



672 *



673 * Implements : FTM_DRV_ClearStatusFlags_Activity



674 *END**************************************************************************/



675void FTM_DRV_ClearStatusFlags(uint32_t instance,



676  uint32_t flagMask)



677{



678  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);



679  FTM_Type * ftmBase = g_ftmBase[instance];



680  uint32_t chnlMask = (flagMask & 0x000000FFU);



681  uint8_t channel = 0U;



682



683  /* Clear the timer overflow flag by writing a 0 to the bit while it is set */



684  if ((flagMask & (uint32_t)FTM_TIME_OVER_FLOW_FLAG) != 0x0U) {



685    FTM_DRV_ClearTimerOverflow(ftmBase);



686  }



687



688  /* Clear fault flag by writing a 0 to the bit while it is set */



689  if ((flagMask & (uint32_t)FTM_FAULT_FLAG) != 0x0U) {



690    FTM_DRV_ClearFaultsIsr(ftmBase);



691  }



692



693  /* Check reload flag by writing a 0 to the bit while it is set */



694  if ((flagMask & (uint32_t)FTM_RELOAD_FLAG) != 0x0U) {



695    FTM_DRV_ClearReloadFlag(ftmBase);



696  }



697



698  /* Clear channel trigger flag */



699  if ((flagMask & (uint32_t)FTM_CHANNEL_TRIGGER_FLAG) != 0x0U) {



700    FTM_DRV_ClearChnTriggerFlag(ftmBase);



701  }



702



703  /* Clear the channel status flags by writing a 0 to the bit */



704  for (channel = 0U; channel < FEATURE_FTM_CHANNEL_COUNT; channel++) {



705    if ((chnlMask & 0x00000001U) != 0x0U) {



706      FTM_DRV_ClearChnEventStatus(ftmBase, channel);



707    }



708



709    chnlMask = chnlMask >> 1U;



710  }



711}

713/*FUNCTION**********************************************************************



714 *



715 * Function Name : FTM_DRV_GetFrequency



716 * Description   : Retrieves the frequency of the clock source feeding the FTM counter.



717 * Function will return a 0 if no clock source is selected and the FTM counter is disabled.



718 * The returned value is clock sources for the FTM counter.



719 *



720 * Implements    : FTM_DRV_GetFrequency_Activity



721 *END**************************************************************************/



722uint32_t FTM_DRV_GetFrequency(uint32_t instance)



723{



724  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);



725  FTM_Type const * ftmBase = g_ftmBase[instance];



726  status_t returnCode = STATUS_SUCCESS;



727  clock_names_t ftmClkName;



728  uint8_t clkPs;



729  uint32_t frequency = 0U;



730  const ftm_state_t * state = ftmStatePtr[instance];



731  clkPs = (uint8_t)(1U << FTM_DRV_GetClockPs(ftmBase));



732  switch (state->ftmClockSource)



733  {



734   case FTM_CLOCK_SOURCE_EXTERNALCLK:



735    returnCode = CLOCK_SYS_GetFreq(g_ftmExtClockSel[instance][1], &frequency);



736    if (0U == frequency) {



737      ftmClkName = g_ftmExtClockSel[instance][0];



738    } else {



739      ftmClkName = g_ftmExtClockSel[instance][1];



740    }



741



742    /* Get the clock frequency value */



743    returnCode = CLOCK_SYS_GetFreq(ftmClkName, &frequency);



744    break;



745



746   case FTM_CLOCK_SOURCE_FIXEDCLK:



747    /* Get the clock frequency value */



748    returnCode = CLOCK_SYS_GetFreq(SIM_RTCCLK_CLK, &frequency);



749    break;



750



751   case FTM_CLOCK_SOURCE_SYSTEMCLK:



752    /* Get the clock frequency value */



753    returnCode = CLOCK_SYS_GetFreq(CORE_CLK, &frequency);



754    break;



755



756   default:



757    /* Nothing to do */



758    break;



759  }



760



761  /* Checks the functional clock of FTM module */



762  (void)returnCode;



763  DEV_ASSERT(returnCode == STATUS_SUCCESS);



764  return (uint32_t)(frequency / clkPs);



765}

769 * Function Name : FTM_DRV_ConvertFreqToPeriodTicks



770 * Description   : This function converts the input parameters representing



771 * frequency in Hz to a period value in ticks needed by the hardware timer.



772 *



773 * Implements    : FTM_DRV_ConvertFreqToPeriodTicks_Activity



774 *END**************************************************************************/



775uint16_t FTM_DRV_ConvertFreqToPeriodTicks(uint32_t instance,

782  uFTMhz = state->ftmSourceClockFrequency;



783  return (uint16_t)(uFTMhz / freqencyHz);



784}
