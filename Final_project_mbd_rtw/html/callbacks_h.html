37/*!



38 * @brief Define the enum of the events which can trigger I2C slave callback



39 *



40 * This enum should include the events for all platforms



41 */



42typedef enum {



43  I2C_SLAVE_EVENT_RX_FULL = 0x00U,



44  I2C_SLAVE_EVENT_TX_EMPTY = 0x01U,



45  I2C_SLAVE_EVENT_TX_REQ = 0x02U,



46  I2C_SLAVE_EVENT_RX_REQ = 0x03U,



47  I2C_SLAVE_EVENT_STOP = 0X04U,



48} i2c_slave_event_t;



49



50/*!



51 * @brief Define the enum of the events which can trigger I2C master callback



52 *



53 * This enum should include the events for all platforms



54 *



55 */



56typedef enum {



57  I2C_MASTER_EVENT_END_TRANSFER = 0x4U,



58} i2c_master_event_t;



59

102/*! @brief Defines a structure used to pass information to the ADC PAL callback



103 *



104 * Implements : adc_callback_info_t_Class



105 */



106typedef struct {



107  uint32_t groupIndex;                 /*!< Index of the group executing the callback. */



108  uint16_t resultBufferTail;           /*!< Offset of the most recent conversion result in the result buffer. */



109} adc_callback_info_t;



110



111/*! @brief Defines the callback used to be called by ADC PAL after the last conversion result in a group



112 * has been copied to the result buffer.



113 */



114typedef void (* const adc_callback_t)(const adc_callback_info_t * const



115  callbackInfo, void * userData);

117/* I2S */



118/* Define the enum of the events which can trigger i2s callback */



119/* Events for all peripherals which support i2s



120 *



121 * Implements : i2s_event_t_Class



122 */



123typedef enum {



124  I2S_EVENT_RX_FULL = 0x00U,           /*!< Rx buffer is full */



125  I2S_EVENT_TX_EMPTY = 0x01U,          /*!< Tx buffer is empty */



126  I2S_EVENT_END_TRANSFER = 0x02U,      /*!< The current transfer is ending. Only FLEXIO instance uses this event. The difference between this and event TX_EMPTY is:



127                                          TX_EMPTY is generated when all data has been pushed to hardware fifo, users should not call DeInit here or some last data will be lost;



128                                          END_TRANSFER is generated when all data has been pushed to line, the transmission will be stopped before users can start transmit again, user can call DeInit here.



129                                          For receiving case, this event is the same as RX_FULL.



130                                        */



131  I2S_EVENT_ERROR = 0x03U,             /*!< An error occurred during transfer */



132} i2s_event_t;



133



134/* Callback for all peripherals which support i2s



135 *



136 * Implements : i2s_callback_t_Class



137 */



138typedef void (*i2s_callback_t)(i2s_event_t event, void *userData);
