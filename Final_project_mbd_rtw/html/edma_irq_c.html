142/*! @brief DMA IRQ handler with the same name in the startup code*/



143void DMA0_15_IRQHandler(void)



144{



145  /* Read the status flags register */



146  const DMA_Type * edmaRegBase = EDMA_DRV_GetDmaRegBaseAddr(0U);



147  uint32_t mask = 0xFFFF;



148  uint32_t flags = edmaRegBase->INT;



149  uint8_t virtualChannel = 0U;



150  flags &= mask;



151



152  /* Check all the flags from 0 to 15 and call the handler for the appropriate channel */



153  while (flags > 0U) {



154    if ((flags & 1U) > 0U) {



155      EDMA_DRV_IRQHandler(virtualChannel);



156    }



157



158    virtualChannel++;



159    flags >>= 1U;



160  }



161}
