50/*! @brief Type for a mutex. */



51typedef SemaphoreHandle_t mutex_t;



52



53/*! @brief Type for a semaphore. */



54typedef SemaphoreHandle_t semaphore_t;



55



56#else



57



58/* Bare-metal implementation */



59/*! @brief Type for a mutex. */



60typedef uint8_t mutex_t;



61



62/*! @brief Type for a semaphore. */



63typedef volatile uint8_t semaphore_t;

71/*******************************************************************************



72 * API



73 ******************************************************************************/



74#if defined (__cplusplus)



75



76extern "C" {



77



78#endif



79



80  /*!



81   * @brief Delays execution for a number of milliseconds.



82   *



83   * @param[in] delay Time delay in milliseconds.



84   */



85  void OSIF_TimeDelay(const uint32_t delay);



86



87  /*!



88   * @brief Returns the number of miliseconds elapsed since starting the internal timer



89   * or starting the scheduler.



90   *



91   * @return the number of miliseconds elapsed



92   */



93  uint32_t OSIF_GetMilliseconds(void);



94



95  /*!



96   * @brief Waits for a mutex and locks it.



97   *



98   * @param[in] pMutex reference to the mutex object



99   * @param[in] timeout time-out value in milliseconds



100   * @return  One of the possible status codes:



101   * - STATUS_SUCCESS: mutex lock operation success



102   * - STATUS_ERROR: mutex already owned by current thread



103   * - STATUS_TIMEOUT: mutex lock operation timed out



104   *



105   */



106  status_t OSIF_MutexLock(const mutex_t * const pMutex,



107    const uint32_t timeout);



108



109  /*!



110   * @brief Unlocks a previously locked mutex.



111   *



112   * @param[in] pMutex reference to the mutex object



113   * @return  One of the possible status codes:



114   * - STATUS_SUCCESS: mutex unlock operation success



115   * - STATUS_ERROR: mutex unlock failed



116   */



117  status_t OSIF_MutexUnlock(const mutex_t * const pMutex);



118



119  /*!



120   * @brief Create an unlocked mutex.



121   *



122   * @param[in] pMutex reference to the mutex object



123   * @return  One of the possible status codes:



124   * - STATUS_SUCCESS: mutex created



125   * - STATUS_ERROR: mutex could not be created



126   */



127  status_t OSIF_MutexCreate(mutex_t * const pMutex);



128



129  /*!



130   * @brief Destroys a previously created mutex.



131   *



132   * @param[in] pMutex reference to the mutex object



133   * @return  One of the possible status codes:



134   * - STATUS_SUCCESS: mutex destroyed



135   */



136  status_t OSIF_MutexDestroy(const mutex_t * const pMutex);



137



138  /*!



139   * @brief Decrement a semaphore with timeout.



140   *



141   * @param[in] pSem reference to the semaphore object



142   * @param[in] timeout time-out value in milliseconds



143   * @return  One of the possible status codes:



144   * - STATUS_SUCCESS: semaphore wait operation success



145   * - STATUS_TIMEOUT: semaphore wait timed out



146   */



147  status_t OSIF_SemaWait(semaphore_t * const pSem,



148    const uint32_t timeout);



149



150  /*!



151   * @brief Increment a semaphore



152   *



153   * @param[in] pSem reference to the semaphore object



154   * @return  One of the possible status codes:



155   * - STATUS_SUCCESS: semaphore post operation success



156   * - STATUS_ERROR: semaphore could not be incremented



157   */



158  status_t OSIF_SemaPost(semaphore_t * const pSem);



159



160  /*!



161   * @brief Creates a semaphore with a given value.



162   *



163   * @param[in] pSem reference to the semaphore object



164   * @param[in] initValue initial value of the semaphore



165   * @return  One of the possible status codes:



166   * - STATUS_SUCCESS: semaphore created



167   * - STATUS_ERROR: semaphore could not be created



168   */



169  status_t OSIF_SemaCreate(semaphore_t * const pSem,



170    const uint8_t initValue);



171



172  /*!



173   * @brief Destroys a previously created semaphore.



174   *



175   * @param[in] pSem reference to the semaphore object



176   * @return  One of the possible status codes:



177   * - STATUS_SUCCESS: semaphore destroyed



178   */



179  status_t OSIF_SemaDestroy(const semaphore_t * const pSem);



180



181  /*! @}*/



182#if defined (__cplusplus)



183



184}



185#endif



186



187/*! @}*/



188#endif                                 /* OSIF_H */
