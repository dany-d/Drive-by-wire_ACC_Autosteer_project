38/*!



39 * @brief Power modes enumeration.



40 *



41 * Defines power modes. Used in the power mode configuration structure



42 * (power_manager_user_config_t). From ARM core perspective, Power modes



43 * can be generally divided into run modes (High speed run, Run and



44 * Very low power run), sleep (Wait and Very low power wait) and deep sleep modes



45 * (all Stop modes).



46 * List of power modes supported by specific chip along with requirements for entering



47 * and exiting of these modes can be found in chip documentation.



48 * List of all supported power modes:\n



49 *  \li POWER_MANAGER_HSRUN - High speed run mode.



50 *  \li POWER_MANAGER_RUN - Run mode.



51 *  \li POWER_MANAGER_VLPR - Very low power run mode.



52 *  \li POWER_MANAGER_WAIT - Wait mode.



53 *  \li POWER_MANAGER_VLPW - Very low power wait mode.



54 *  \li POWER_MANAGER_PSTOP1 - Partial stop 1 mode.



55 *  \li POWER_MANAGER_PSTOP2 - Partial stop 2 mode.



56 *  \li POWER_MANAGER_PSTOP1 - Stop 1 mode.



57 *  \li POWER_MANAGER_PSTOP2 - Stop 2 mode.



58 *  \li POWER_MANAGER_VLPS - Very low power stop mode.



59 * Implements power_manager_modes_t_Class



60 */



61typedef enum



62{



63#if FEATURE_SMC_HAS_HIGH_SPEED_RUN_MODE



64    POWER_MANAGER_HSRUN,            /*!< High speed run mode.  */



65#endif



66    POWER_MANAGER_RUN,              /*!< Run mode. */



67    POWER_MANAGER_VLPR,             /*!< Very low power run mode.  */



68#if FEATURE_SMC_HAS_WAIT_VLPW



69    POWER_MANAGER_WAIT,             /*!< Wait mode.  */



70    POWER_MANAGER_VLPW,             /*!< Very low power wait mode.  */



71#endif



72#if FEATURE_SMC_HAS_PSTOPO



73    POWER_MANAGER_PSTOP1,           /*!< Partial stop 1 mode. */



74    POWER_MANAGER_PSTOP2,           /*!< Partial stop 2 mode. */



75#endif



76#if FEATURE_SMC_HAS_STOPO



77    POWER_MANAGER_STOP1,           /*!< Stop 1 mode. */



78    POWER_MANAGER_STOP2,           /*!< Stop 2 mode. */



79#endif



80    POWER_MANAGER_VLPS,             /*!< Very low power stop mode.  */



81    POWER_MANAGER_MAX



82} power_manager_modes_t;



83



84/*!



85 * @brief Power mode user configuration structure.



86 *



87 * List of power mode configuration structure members depends on power options available



88 * for the specific chip. Complete list contains:



89 * mode - S32K power mode. List of available modes is chip-specific. See power_manager_modes_t



90 * list of modes.



91 * sleepOnExitOption - Controls whether the sleep-on-exit option value is used(when set to true)



92 * or ignored(when set to false). See sleepOnExitValue.



93 * sleepOnExitValue - When set to true, ARM core returns to sleep (S32K wait modes) or deep sleep



94 * state (S32K stop modes) after interrupt service finishes. When set to false, core stays



95 * woken-up.



96 * Implements power_manager_user_config_t_Class



97 */



98typedef struct



99{



100    power_manager_modes_t powerMode;



101    bool sleepOnExitValue;



102} power_manager_user_config_t;

104/*!



105 * @brief Power Modes in PMSTAT



106 *



107 */



108typedef enum



109{



110    STAT_RUN     = 0x01,              /*!< 0000_0001 - Current power mode is RUN*/



111    STAT_STOP    = 0x02,              /*!< 0000_0010 - Current power mode is STOP*/



112    STAT_VLPR    = 0x04,              /*!< 0000_0100 - Current power mode is VLPR*/



113    STAT_VLPW    = 0x08,              /*!< 0000_1000 - Current power mode is VLPW*/



114    STAT_VLPS    = 0x10,              /*!< 0001_0000 - Current power mode is VLPS*/



115    STAT_HSRUN   = 0x80,              /*!< 1000_0000 - Current power mode is HSRUN*/



116    STAT_INVALID = 0xFF               /*!< 1111_1111 - Non-existing power mode*/



117} power_mode_stat_t;
