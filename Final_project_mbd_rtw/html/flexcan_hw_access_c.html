212/*FUNCTION**********************************************************************



213 *



214 * Function Name : FLEXCAN_GetRxFifoMask



215 * Description   : Calculate the Global Mask as format ID type in FIFO Mode.



216 * Based on the ID format type and Mask Type will calculate the and set accordingly



217 * the Rx FIFO Individual\Global Mask register.



218 *



219 *END**************************************************************************/



220uint32_t FLEXCAN_GetRxFifoMask( flexcan_msgbuff_id_type_t id_type,



221  flexcan_rx_fifo_id_element_format_t formatType,



222  uint32_t mask)



223{



224  uint32_t val= 0;



225  switch (formatType)



226  {



227   case (FLEXCAN_RX_FIFO_ID_FORMAT_A):



228    /* Set RTR bit encoded as bit 31 and IDE bit encoded as bit 30 in mask */



229    val = mask & ((1UL<<FLEXCAN_RX_FIFO_ID_FILTER_FORMATAB_RTR_SHIFT) | (1UL<<



230      FLEXCAN_RX_FIFO_ID_FILTER_FORMATAB_IDE_SHIFT));



231    if (id_type == FLEXCAN_MSG_ID_STD) {



232      /* Set standard global mask for RX FIFO and IDE will be 1 and check the FIFO filter ide */



233      val |= ((mask << FLEXCAN_RX_FIFO_ID_FILTER_FORMATA_STD_SHIFT) &



234              FLEXCAN_RX_FIFO_ID_FILTER_FORMATA_STD_MASK);



235    } else if (id_type == FLEXCAN_MSG_ID_EXT) {



236      /* Set extended global mask for RX FIFO and IDE will be 0 and don't check the FIFO filter ide */



237      val |= ((mask << FLEXCAN_RX_FIFO_ID_FILTER_FORMATA_EXT_SHIFT) &



238              FLEXCAN_RX_FIFO_ID_FILTER_FORMATA_EXT_MASK);



239    } else {



240      /* Should not get here */



241    }



242    break;



243



244   case (FLEXCAN_RX_FIFO_ID_FORMAT_B):



245    /* Set RTR bit encoded as bit 31 and IDE bit encoded as bit 30 in mask */



246    val = mask & ((1UL<<FLEXCAN_RX_FIFO_ID_FILTER_FORMATAB_RTR_SHIFT) | (1UL<<



247      FLEXCAN_RX_FIFO_ID_FILTER_FORMATAB_IDE_SHIFT));



248    if (id_type == FLEXCAN_MSG_ID_STD) {



249      /* Set standard global mask for RX FIFO  */



250      val |= ((mask & FLEXCAN_RX_FIFO_ID_FILTER_FORMATB_STD_MASK) <<



251              FLEXCAN_RX_FIFO_ID_FILTER_FORMATB_STD_SHIFT1);



252    } else if (id_type == FLEXCAN_MSG_ID_EXT) {



253      /* Set extended global mask for RX FIFO  */



254      val |= ((mask & FLEXCAN_RX_FIFO_ID_FILTER_FORMATB_EXT_MASK1) <<



255              FLEXCAN_RX_FIFO_ID_FILTER_FORMATB_EXT_SHIFT1);



256    } else {



257      /* Should not get here */



258    }



259    break;



260



261   case (FLEXCAN_RX_FIFO_ID_FORMAT_C):



262    if ((id_type == FLEXCAN_MSG_ID_EXT) || (id_type == FLEXCAN_MSG_ID_STD)) {



263      val |= ((mask & FLEXCAN_RX_FIFO_ID_FILTER_FORMATB_STD_MASK) <<



264              FLEXCAN_RX_FIFO_ID_FILTER_FORMATC_SHIFT1);



265    } else {



266      /* Should not get here */



267    }



268    break;



269



270   case (FLEXCAN_RX_FIFO_ID_FORMAT_D):



271    /* Format not supported.*/



272    break;



273



274   default:



275    /* Should not get here */



276    break;



277  }                                    /* End Switch */



278



279  return val;



280}

316/*FUNCTION**********************************************************************



317 *



318 * Function Name: FLEXCAN_ComputeDLCValue



319 * Description  : Computes the DLC field value, given a payload size (in bytes).



320 *



321 *END**************************************************************************/



322static uint8_t FLEXCAN_ComputeDLCValue(



323  uint8_t payloadSize)



324{



325  uint8_t ret;



326  if (payloadSize <= 8U) {



327    ret = payloadSize;



328  } else if ((payloadSize > 8U) && (payloadSize <= 12U)) {



329    ret = CAN_DLC_VALUE_12_BYTES;



330  } else if ((payloadSize > 12U) && (payloadSize <= 16U)) {



331    ret = CAN_DLC_VALUE_16_BYTES;



332  } else if ((payloadSize > 16U) && (payloadSize <= 20U)) {



333    ret = CAN_DLC_VALUE_20_BYTES;



334  } else if ((payloadSize > 20U) && (payloadSize <= 24U)) {



335    ret = CAN_DLC_VALUE_24_BYTES;



336  } else if ((payloadSize > 24U) && (payloadSize <= 32U)) {



337    ret = CAN_DLC_VALUE_32_BYTES;



338  } else if ((payloadSize > 32U) && (payloadSize <= 48U)) {



339    ret = CAN_DLC_VALUE_48_BYTES;



340  } else if ((payloadSize > 48U) && (payloadSize <= 64U)) {



341    ret = CAN_DLC_VALUE_64_BYTES;



342  } else {



343    /* The argument is not a valid payload size */



344    ret = 0xFFU;



345  }



346



347  return ret;



348}

495    while (((base->MCR & CAN_MCR_LPMACK_MASK) >> CAN_MCR_LPMACK_SHIFT) != 0U) {



496    }



497  }

716      }

720      }

723    }



724  }
