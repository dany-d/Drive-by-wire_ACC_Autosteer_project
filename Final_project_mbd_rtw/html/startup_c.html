63/*******************************************************************************



64 * Static Variables

66static volatile uint32_t * s_vectors[NUMBER_OF_CORES] =



67  FEATURE_INTERRUPT_INT_VECTORS;

69/*******************************************************************************



70 * Code



71 ******************************************************************************/

73/*FUNCTION**********************************************************************



74 *



75 * Function Name : init_data_bss



76 * Description   : Make necessary initializations for RAM.



77 * - Copy the vector table from ROM to RAM.



78 * - Copy initialized data from ROM to RAM.



79 * - Copy code that should reside in RAM from ROM



80 * - Clear the zero-initialized data section.



81 *



82 * Tool Chains:



83 *   __GNUC__   : GNU Compiler Collection



84 *   __ghs__    : Green Hills ARM Compiler



85 *   __ICCARM__ : IAR ARM Compiler



86 *   __DCC__    : Wind River Diab Compiler



87 *



88 * Implements    : init_data_bss_Activity



89 *END**************************************************************************/



90void init_data_bss(void)



91{



92  uint32_t n;



93  uint8_t coreId;



94



95  /* Declare pointers for various data sections. These pointers



96   * are initialized using values pulled in from the linker file */



97  uint8_t * data_ram;



98  uint8_t * code_ram;



99  uint8_t * bss_start;



100  uint8_t * custom_ram;



101  const uint8_t * data_rom, * data_rom_end;



102  const uint8_t * code_rom, * code_rom_end;



103  const uint8_t * bss_end;



104  const uint8_t * custom_rom, * custom_rom_end;



105



106  /* Addresses for VECTOR_TABLE and VECTOR_RAM come from the linker file */



107#if defined(__ARMCC_VERSION)



108



109  extern uint32_t __RAM_VECTOR_TABLE_SIZE;



110  extern uint32_t __VECTOR_ROM;



111  extern uint32_t __VECTOR_RAM;



112



113#else



114



115  extern uint32_t __RAM_VECTOR_TABLE_SIZE[];



116  extern uint32_t __VECTOR_TABLE[];



117  extern uint32_t __VECTOR_RAM[];



118



119#endif



120



121  /* Get section information from linker files */



122#if defined(__ICCARM__)



123



124  /* Data */



125  data_ram = __section_begin(".data");



126  data_rom = __section_begin(".data_init");



127  data_rom_end = __section_end(".data_init");



128



129  /* CODE RAM */



130#pragma section = "__CODE_ROM"



131#pragma section = "__CODE_RAM"



132



133  code_ram = __section_begin("__CODE_RAM");



134  code_rom = __section_begin("__CODE_ROM");



135  code_rom_end = __section_end("__CODE_ROM");



136



137  /* BSS */



138  bss_start = __section_begin(".bss");



139  bss_end = __section_end(".bss");



140  custom_ram = __section_begin(".customSection");



141  custom_rom = __section_begin(".customSection_init");



142  custom_rom_end = __section_end(".customSection_init");



143



144#elif defined (__ARMCC_VERSION)



145



146  extern uint32_t __DATA_ROM;



147  extern uint32_t __DATA_RAM;



148  extern uint32_t __DATA_END;



149  extern uint32_t __CODE_RAM;



150  extern uint32_t __CODE_ROM;



151  extern uint32_t __CODE_END;



152  extern uint32_t __BSS_START;



153  extern uint32_t __BSS_END;



154



155  /* Data */



156  data_ram = (uint8_t *)__DATA_RAM;



157  data_rom = (uint8_t *)__DATA_ROM;



158  data_rom_end = (uint8_t *)__DATA_END;



159



160  /* CODE RAM */



161  code_ram = (uint8_t *)__CODE_RAM;



162  code_rom = (uint8_t *)__CODE_ROM;



163  code_rom_end = (uint8_t *)__CODE_END;



164



165  /* BSS */



166  bss_start = (uint8_t *)__BSS_START;



167  bss_end = (uint8_t *)__BSS_END;



168



169  /* VECTOR TABLE*/



170  uint8_t * vector_table_size = (uint8_t *)__RAM_VECTOR_TABLE_SIZE;



171  uint32_t * vector_rom = (uint32_t *)__VECTOR_ROM;



172  uint32_t * vector_ram = (uint32_t *)__VECTOR_RAM;



173



174#else



175



176  extern uint32_t __DATA_ROM[];



177  extern uint32_t __DATA_RAM[];



178  extern uint32_t __DATA_END[];



179  extern uint32_t __CODE_RAM[];



180  extern uint32_t __CODE_ROM[];



181  extern uint32_t __CODE_END[];



182  extern uint32_t __BSS_START[];



183  extern uint32_t __BSS_END[];



184  extern uint32_t __CUSTOM_ROM[];



185  extern uint32_t __CUSTOM_END[];



186



187  /* Data */



188  data_ram = (uint8_t *)__DATA_RAM;



189  data_rom = (uint8_t *)__DATA_ROM;



190  data_rom_end = (uint8_t *)__DATA_END;



191



192  /* CODE RAM */



193  code_ram = (uint8_t *)__CODE_RAM;



194  code_rom = (uint8_t *)__CODE_ROM;



195  code_rom_end = (uint8_t *)__CODE_END;



196



197  /* BSS */



198  bss_start = (uint8_t *)__BSS_START;



199  bss_end = (uint8_t *)__BSS_END;



200



201  /* Custom section */



202  custom_ram = CUSTOMSECTION_SECTION_START;



203  custom_rom = (uint8_t *)__CUSTOM_ROM;



204  custom_rom_end = (uint8_t *)__CUSTOM_END;



205



206#endif



207



208  /* Copy initialized data from ROM to RAM */



209  while (data_rom_end != data_rom) {



210    *data_ram = *data_rom;



211    data_ram++;



212    data_rom++;



213  }



214



215  /* Copy functions from ROM to RAM */



216  while (code_rom_end != code_rom) {



217    *code_ram = *code_rom;



218    code_ram++;



219    code_rom++;



220  }



221



222  /* Clear the zero-initialized data section */



223  while (bss_end != bss_start) {



224    *bss_start = 0;



225    bss_start++;



226  }



227



228  /* Copy customsection rom to ram */



229  while (custom_rom_end != custom_rom) {



230    *custom_ram = *custom_rom;



231    custom_rom++;



232    custom_ram++;



233  }



234



235  coreId = (uint8_t)GET_CORE_ID();



236



237#if defined (__ARMCC_VERSION)



238



239  /* Copy the vector table from ROM to RAM */



240  /* Workaround */



241  for (n = 0; n < (((uint32_t)(vector_table_size))/sizeof(uint32_t)); n++) {



242    vector_ram[n] = vector_rom[n];



243  }



244



245  /* Point the VTOR to the position of vector table */



246  *s_vectors[coreId] = (uint32_t) __VECTOR_RAM;



247



248#else



249



250  /* Check if VECTOR_TABLE copy is needed */



251  if (__VECTOR_RAM != __VECTOR_TABLE) {



252    /* Copy the vector table from ROM to RAM */



253    for (n = 0; n < (((uint32_t)__RAM_VECTOR_TABLE_SIZE)/sizeof(uint32_t)); n++)



254    {



255      __VECTOR_RAM[n] = __VECTOR_TABLE[n];



256    }



257



258    /* Point the VTOR to the position of vector table */



259    *s_vectors[coreId] = (uint32_t)__VECTOR_RAM;



260  } else {



261    /* Point the VTOR to the position of vector table */



262    *s_vectors[coreId] = (uint32_t)__VECTOR_TABLE;



263  }



264



265#endif



266



267}
