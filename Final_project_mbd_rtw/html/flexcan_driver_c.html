398/*FUNCTION**********************************************************************



399 *



400 * Function Name : FLEXCAN_DRV_SetRxIndividualMask

403 *



404 * Implements    : FLEXCAN_DRV_SetRxIndividualMask_Activity

406status_t FLEXCAN_DRV_SetRxIndividualMask(



407  uint8_t instance,



408  flexcan_msgbuff_id_type_t id_type,



409  uint8_t mb_idx,



410  uint32_t mask)



411{



412  DEV_ASSERT(instance < CAN_INSTANCE_COUNT);



413  CAN_Type * base = g_flexcanBase[instance];



414  flexcan_rx_fifo_id_element_format_t formatType;



415  FLEXCAN_EnterFreezeMode(base);



416  if ((mb_idx > FLEXCAN_GetMaxMsgBuffNum(base)) || (mb_idx >= CAN_RXIMR_COUNT))



417  {



418    FLEXCAN_ExitFreezeMode(base);



419    return STATUS_CAN_BUFF_OUT_OF_RANGE;



420  }



421



422  if (false == FLEXCAN_IsRxFifoEnabled(base)) {



423    if (id_type == FLEXCAN_MSG_ID_STD) {



424      /* Set standard individual mask*/



425      FLEXCAN_SetRxIndividualStdMask(base, mb_idx, mask);



426    } else if (id_type == FLEXCAN_MSG_ID_EXT) {



427      /* Set extended individual mask*/



428      FLEXCAN_SetRxIndividualExtMask(base, mb_idx, mask);



429    } else {



430      /* Should not get here */



431    }



432  } else {



433    if (mb_idx > FLEXCAN_GetNoOfIndividualMBsRxFIFO(base)) {



434      FLEXCAN_ExitFreezeMode(base);



435      return STATUS_CAN_BUFF_OUT_OF_RANGE;



436    }



437



438    /* In FIFO Mode get the Id Filter Format already configured by FLEXCAN_DRV_ConfigRxFifo */



439    formatType = FLEXCAN_GetRxFifoIdFormat(base);



440    FLEXCAN_SetRxIndividualMask(base, mb_idx,



441      FLEXCAN_GetRxFifoMask(id_type, formatType, mask));



442  }



443



444  FLEXCAN_ExitFreezeMode(base);



445  return STATUS_SUCCESS;



446}

448/*FUNCTION**********************************************************************



449 *



450 * Function Name : FLEXCAN_DRV_Init



451 * Description   : Initialize FlexCAN driver.



452 * This function will select a source clock, reset FlexCAN module, set maximum



453 * number of message buffers, initialize all message buffers as inactive, enable



454 * RX FIFO if needed, mask all mask bits, disable all MB interrupts, enable



455 * FlexCAN normal mode, and enable all the error interrupts if needed.



456 *



457 * Implements    : FLEXCAN_DRV_Init_Activity



458 *END**************************************************************************/



459status_t FLEXCAN_DRV_Init(



460  uint8_t instance,



461  flexcan_state_t *state,



462  const flexcan_user_config_t *data)



463{



464  DEV_ASSERT(instance < CAN_INSTANCE_COUNT);



465  DEV_ASSERT(state != NULL);



466



467#ifdef ERRATA_E10595



468



469  if (instance != 0U) {



470    DEV_ASSERT((MC_ME->GS & MC_ME_GS_S_FXOSC_MASK) != 0U);



471  }



472



473#endif



474



475  status_t result;



476  CAN_Type * base = g_flexcanBase[instance];



477  flexcan_time_segment_t bitrate;



478  status_t osifStat;



479  uint32_t i, j;



480



481  /* To enter Disable Mode requires FreezMode first */



482  FLEXCAN_EnterFreezeMode(base);



483  FLEXCAN_Disable(base);



484



485#if FEATURE_CAN_HAS_PE_CLKSRC_SELECT



486



487  /* Select a source clock for the FlexCAN engine */



488  FLEXCAN_SelectClock(base, data->pe_clock);



489



490#endif



491



492  /* Enable the CAN clock */



493  FLEXCAN_Enable(base);



494  FLEXCAN_EnterFreezeMode(base);



495



496  /* Initialize FLEXCAN device */



497  FLEXCAN_Init(base);



498



499#ifdef ERRATA_E10368



500#if FEATURE_CAN_HAS_FD



501



502  if (data->fd_enable) {



503    flexcan_msgbuff_code_status_t messageBuff= {



504      .code = FLEXCAN_TX_REMOTE,



505      .msgIdType = FLEXCAN_MSG_ID_STD,



506      .dataLen = 1U,



507      .fd_enable = false,



508      .fd_padding = 0U,



509      .enable_brs = false



510    };



511



512    uint8_t dummy_data[16];



513



514    /* Enable Self Reception */



515    FLEXCAN_SetSelfReception(base, true);



516



517    /* Select Loopback mode */



518    FLEXCAN_SetOperationMode(base, FLEXCAN_LOOPBACK_MODE);



519



520    /* Dummy transmission on MB0 */



521    (void)FLEXCAN_SetTxMsgBuff(base, 0U, &messageBuff, 0x1U, dummy_data);



522    FLEXCAN_SetFDEnabled(base, true);



523    bitrate = data->bitrate;



524    FLEXCAN_SetExtendedTimeSegments(base, &bitrate);



525    bitrate = data->bitrate_cbt;



526    FLEXCAN_SetFDTimeSegments(base, &bitrate);



527    FLEXCAN_ExitFreezeMode(base);



528



529    /* Wait transmission of MB0 to complete */



530    while ((base->IFLAG1 & 0x1U) != 0x1U) {



531    }



532



533    ;



534



535    /* Clear Transmission Complete */



536    base->IFLAG1 = base->IFLAG1 | 0x1U;



537    FLEXCAN_EnterFreezeMode(base);



538  }



539



540#endif                                 /* FEATURE_CAN_HAS_FD */



541#endif                                 /* ERRATA_E10368 */



542



543#if FEATURE_CAN_HAS_FD



544



545  /* Enable/Disable FD and check FD was set as expected. Setting FD as enabled



546   * might fail if the current CAN instance does not support FD. */



547  FLEXCAN_SetFDEnabled(base, data->fd_enable);



548  if (FLEXCAN_IsFDEnabled(base) != data->fd_enable) {



549    return STATUS_ERROR;



550  }



551



552  /* If the FD feature is enabled, enable the Stuff Bit Count, in order to be



553   * ISO-compliant. */



554  FLEXCAN_SetStuffBitCount(base, data->fd_enable);



555



556#endif



557



558  /* Disable the self reception feature if FlexCAN is not in loopback mode. */



559  if (data->flexcanMode != FLEXCAN_LOOPBACK_MODE) {



560    FLEXCAN_SetSelfReception(base, false);



561  }



562



563  /* Enable RxFIFO feature, if requested. This might fail if the FD mode is



564   * enabled. */



565  if (data->is_rx_fifo_needed) {



566    result = FLEXCAN_EnableRxFifo(base, (uint32_t)data->num_id_filters);



567    if (result != STATUS_SUCCESS) {



568      return result;



569    }



570  }



571



572#if FEATURE_CAN_HAS_DMA_ENABLE



573



574  /* Enable DMA support for RxFIFO transfer, if requested. */



575  if (data->transfer_type == FLEXCAN_RXFIFO_USING_DMA) {



576    if (FLEXCAN_IsRxFifoEnabled(base)) {



577      FLEXCAN_SetRxFifoDMA(base, true);



578    } else {



579      return STATUS_ERROR;



580    }



581  }



582



583#endif



584



585  /* Select mode */



586  FLEXCAN_SetOperationMode(base, data->flexcanMode);



587



588#if FEATURE_CAN_HAS_FD



589



590  /* Set payload size. */



591  FLEXCAN_SetPayloadSize(base, data->payload);



592



593#endif



594



595  result = FLEXCAN_SetMaxMsgBuffNum(base, data->max_num_mb);



596  if (result != STATUS_SUCCESS) {



597    return result;



598  }



599



600#if FEATURE_CAN_HAS_FD



601



602  /* Set bit rate. */



603  if (FLEXCAN_IsFDEnabled(base)) {



604    bitrate = data->bitrate;



605    FLEXCAN_SetExtendedTimeSegments(base, &bitrate);



606    bitrate = data->bitrate_cbt;



607    FLEXCAN_SetFDTimeSegments(base, &bitrate);



608  } else



609#endif



610



611    {



612      bitrate = data->bitrate;



613      FLEXCAN_SetTimeSegments(base, &bitrate);



614    }



615



616  FLEXCAN_ExitFreezeMode(base);



617



618  /* Enable FlexCAN interrupts.*/



619#if FEATURE_CAN_HAS_WAKE_UP_IRQ



620



621  if (g_flexcanWakeUpIrqId[instance] != NotAvail_IRQn) {



622    INT_SYS_EnableIRQ(g_flexcanWakeUpIrqId[instance]);



623  }



624



625#endif



626



627  INT_SYS_EnableIRQ(g_flexcanErrorIrqId[instance]);



628  INT_SYS_EnableIRQ(g_flexcanBusOffIrqId[instance]);



629  for (i = 0; i < FEATURE_CAN_MB_IRQS_MAX_COUNT; i++) {



630    if (g_flexcanOredMessageBufferIrqId[i][instance] != NotAvail_IRQn) {



631      INT_SYS_EnableIRQ(g_flexcanOredMessageBufferIrqId[i][instance]);



632    }



633  }



634



635  for (i = 0; i < FEATURE_CAN_MAX_MB_NUM; i++) {



636    osifStat = OSIF_SemaCreate(&state->mbs[i].mbSema, 0U);



637    if (osifStat != STATUS_SUCCESS) {



638      for (j = 0; j < i; j++) {



639        (void)OSIF_SemaDestroy(&state->mbs[j].mbSema);



640      }



641



642      return STATUS_ERROR;



643    }



644



645    state->mbs[i].isBlocking = false;



646    state->mbs[i].mb_message = NULL;



647    state->mbs[i].state = FLEXCAN_MB_IDLE;



648  }



649



650  /* Store transfer type and DMA channel number used in transfer */



651  state->transferType = data->transfer_type;



652



653#if FEATURE_CAN_HAS_DMA_ENABLE



654



655  state->rxFifoDMAChannel = data->rxFifoDMAChannel;



656



657#endif



658



659  /* Save runtime structure pointers so irq handler can point to the correct state structure */



660  g_flexcanStatePtr[instance] = state;



661  return (STATUS_SUCCESS);



662}

669 * the function will make sure if the MB requested is not occupied by RX FIFO

1179/*FUNCTION**********************************************************************



1180 *



1181 * Function Name : FLEXCAN_IRQHandler



1182 * Description   : Interrupt handler for FLEXCAN.



1183 * This handler read data from MB or FIFO, and then clear the interrupt flags.



1184 * This is not a public API as it is called whenever an interrupt occurs.



1185 *



1186 *END**************************************************************************/



1187void FLEXCAN_IRQHandler(uint8_t instance)



1188{



1189  DEV_ASSERT(instance < CAN_INSTANCE_COUNT);



1190  uint32_t flag_reg = 0;



1191  CAN_Type * base = g_flexcanBase[instance];



1192  flexcan_state_t * state = g_flexcanStatePtr[instance];



1193  status_t result = STATUS_SUCCESS;



1194



1195  /* Get the interrupts that are enabled and ready */



1196  uint32_t mb_idx = 0;



1197  flag_reg = FLEXCAN_GetMsgBuffIntStatusFlag(base, mb_idx);



1198  while ((flag_reg & 1U) == 0U) {



1199    mb_idx++;



1200    flag_reg = FLEXCAN_GetMsgBuffIntStatusFlag(base, mb_idx);



1201    if (mb_idx >= FEATURE_CAN_MAX_MB_NUM) {



1202      break;



1203    }



1204  }



1205



1206  /* Check Tx/Rx interrupt flag and clear the interrupt */



1207  if (flag_reg != 0U) {



1208    if (FLEXCAN_IsRxFifoEnabled(base)) {



1209      FLEXCAN_IRQHandlerRxFIFO(instance, mb_idx);



1210    } else {



1211      /* Check mailbox completed reception */



1212      if (state->mbs[mb_idx].state == FLEXCAN_MB_RX_BUSY) {



1213        /* Lock RX message buffer and RX FIFO*/



1214        result = FLEXCAN_LockRxMsgBuff(base, mb_idx);



1215        if (result == STATUS_SUCCESS) {



1216          /* Get RX MB field values*/



1217          result = FLEXCAN_GetMsgBuff(base, mb_idx, state->mbs[mb_idx].



1218            mb_message);



1219        }



1220



1221        if (result == STATUS_SUCCESS) {



1222          /* Unlock RX message buffer and RX FIFO*/



1223          FLEXCAN_UnlockRxMsgBuff(base);



1224



1225          /* Complete receive data */



1226          FLEXCAN_CompleteTransfer(instance, mb_idx);



1227          FLEXCAN_ClearMsgBuffIntStatusFlag(base, mb_idx);



1228



1229          /* Invoke callback */



1230          if (state->callback != NULL) {



1231            state->callback(instance, FLEXCAN_EVENT_RX_COMPLETE, mb_idx, state);



1232          }



1233        }



1234      }



1235    }



1236



1237    /* Check mailbox completed transmission */



1238    if (state->mbs[mb_idx].state == FLEXCAN_MB_TX_BUSY) {



1239      /* Complete transmit data */



1240      FLEXCAN_CompleteTransfer(instance, mb_idx);



1241      if (state->mbs[mb_idx].isRemote) {



1242        /* If the frame was a remote frame, clear the flag only if the response was



1243         * not received yet. If the response was received, leave the flag set in order



1244         * to be handled when the user calls FLEXCAN_DRV_RxMessageBuffer. */



1245        flexcan_msgbuff_t mb;



1246        (void) FLEXCAN_LockRxMsgBuff(base, mb_idx);



1247        (void) FLEXCAN_GetMsgBuff(base, mb_idx, &mb);



1248        FLEXCAN_UnlockRxMsgBuff(base);



1249        if (((mb.cs & CAN_CS_CODE_MASK) >> CAN_CS_CODE_SHIFT) == (uint32_t)



1250            FLEXCAN_RX_EMPTY) {



1251          FLEXCAN_ClearMsgBuffIntStatusFlag(base, mb_idx);



1252        }



1253      } else {



1254        FLEXCAN_ClearMsgBuffIntStatusFlag(base, mb_idx);



1255      }



1256



1257      /* Invoke callback */



1258      if (state->callback != NULL) {



1259        state->callback(instance, FLEXCAN_EVENT_TX_COMPLETE, mb_idx, state);



1260      }



1261    }



1262  } else {



1263



1264#if FEATURE_CAN_HAS_PRETENDED_NETWORKING



1265



1266    uint8_t wtof = FLEXCAN_GetWTOF(base);



1267    uint8_t wumf = FLEXCAN_GetWUMF(base);



1268



1269    /* Check if wake-up event occurred */



1270    if ((wtof != 0U) || (wumf != 0U)) {



1271      FLEXCAN_WakeUpHandler(instance);



1272    }



1273



1274#endif



1275



1276    uint32_t ecr = FLEXCAN_GetErrorCounters(base);



1277



1278    /* Check if any error occurred */

1958/*FUNCTION**********************************************************************



1959 *



1960 * Function Name : FLEXCAN_ProccessTSeg



1961 * Description   : Calculate Tseg value



1962 *



1963 *END**************************************************************************/



1964static inline void FLEXCAN_ProccessTSeg(uint32_t * tSeg1, uint32_t * tSeg2)



1965{



1966  /* Adjust time segment 1 and time segment 2 */



1967  while ((*tSeg1 >= FLEXCAN_TSEG1_MAX) || (*tSeg2 < FLEXCAN_TSEG2_MIN)) {



1968    *tSeg2 = *tSeg2 + 1U;



1969    *tSeg1 = *tSeg1 - 1U;



1970  }



1971}

1973/*FUNCTION**********************************************************************



1974 *



1975 * Function Name : FLEXCAN_ProccessPSeg



1976 * Description   : Calculate Pseg value



1977 *



1978 *END**************************************************************************/



1979static inline void FLEXCAN_ProccessPSeg(uint32_t * tmpPropseg, uint32_t *



1980  tmpPseg1)



1981{



1982  while (*tmpPropseg <= 0U) {



1983    *tmpPropseg = *tmpPropseg + 1U;



1984    *tmpPseg1 = *tmpPseg1 - 1U;



1985  }



1986



1987  while (*tmpPropseg >= FLEXCAN_PROPSEG_MAX) {



1988    *tmpPropseg = *tmpPropseg - 1U;



1989    *tmpPseg1 = *tmpPseg1 + 1U;



1990  }



1991}

1993/*FUNCTION**********************************************************************



1994 *



1995 * Function Name : FLEXCAN_BitrateToTimeSeg



1996 * Description   : Converts a bitrate (kbit/s) in time segment values for



1997 *                 standard CAN frame.



1998 *



1999 *END**************************************************************************/



2000static void FLEXCAN_BitrateToTimeSeg(uint32_t bitrate,



2001  uint32_t clkFreq,



2002  flexcan_time_segment_t *timeSeg)



2003{



2004  uint32_t dBitrate, tmpBitrate, dBitrateMin, tmpPresdiv, tmpSample,



2005    dSampleMin, dSample, samplePoint, numTq, pseg1, pseg2, propseg,



2006    presdiv, tSeg1, tSeg2, tmpPseg1, tmpPseg2, tmpPropseg;



2007  presdiv = 0U;



2008  propseg = 0U;



2009  pseg1 = 0U;



2010  pseg2 = 0U;



2011  dSampleMin = 100U;



2012  dBitrateMin = 1000000U;



2013  samplePoint = 88U;



2014  for (tmpPresdiv = 0U; tmpPresdiv < FLEXCAN_PRESDIV_MAX; tmpPresdiv++) {



2015    /* Compute the number of time quanta in 1 bit time */



2016    numTq = clkFreq / ((tmpPresdiv + 1U) * bitrate);



2017



2018    /* Compute the real bitrate resulted */



2019    tmpBitrate = clkFreq / ((tmpPresdiv + 1U) * numTq);



2020



2021    /* The number of time quanta in 1 bit time must be lower than the one supported */



2022    if ((numTq >= FLEXCAN_NUM_TQ_MIN) && (numTq < FLEXCAN_NUM_TQ_MAX)) {



2023      /* Compute time segments based on the value of the sampling point */



2024      tSeg1 = (numTq * samplePoint / 100U) - 1U;



2025      tSeg2 = numTq - 1U - tSeg1;



2026



2027      /* Adjust time segment 1 and time segment 2 */



2028      FLEXCAN_ProccessTSeg(&tSeg1, &tSeg2);



2029      tmpPseg2 = tSeg2 - 1U;



2030



2031      /* Start from pseg1 = pseg2 and adjust until propseg is valid */



2032      tmpPseg1 = tmpPseg2;



2033      tmpPropseg = tSeg1 - tmpPseg1 - 2U;



2034      FLEXCAN_ProccessPSeg(&tmpPropseg, &tmpPseg1);



2035      if (((tSeg1 >= FLEXCAN_TSEG1_MAX) || (tSeg2 >= FLEXCAN_TSEG2_MAX) ||



2036           (tSeg2 < FLEXCAN_TSEG2_MIN) || (tSeg1 < FLEXCAN_TSEG1_MIN)) ||



2037          ((tmpPropseg >= FLEXCAN_PROPSEG_MAX) || (tmpPseg1 >= FLEXCAN_PSEG1_MAX)



2038           || (tmpPseg2 < FLEXCAN_PSEG2_MIN) || (tmpPseg2 >= FLEXCAN_PSEG2_MAX)))



2039      {



2040        continue;



2041      }



2042



2043      tmpSample = ((tSeg1 + 1U) * 100U) / numTq;



2044      dSample = FLEXCAN_CheckDsample(tmpSample , samplePoint);



2045      dBitrate = FLEXCAN_CheckdBitrate(tmpBitrate , bitrate);



2046      if ((dBitrate < dBitrateMin) ||



2047          ((dBitrate == dBitrateMin) && (dSample < dSampleMin))) {



2048        dSampleMin = dSample;



2049        dBitrateMin = dBitrate;



2050        pseg1 = tmpPseg1;



2051        pseg2 = tmpPseg2;



2052        presdiv = tmpPresdiv;



2053        propseg = tmpPropseg;



2054        if ((dBitrate == 0U) && (dSample <= 1U)) {



2055          break;



2056        }



2057      }



2058    }



2059  }



2060



2061  timeSeg->phaseSeg1 = pseg1;



2062  timeSeg->phaseSeg2 = pseg2;



2063  timeSeg->preDivider = presdiv;



2064  timeSeg->propSeg = propseg;



2065  timeSeg->rJumpwidth = FLEXCAN_CheckJumpwidth(pseg1);



2066}
