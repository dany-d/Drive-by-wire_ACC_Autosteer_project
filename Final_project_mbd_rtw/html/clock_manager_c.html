44static clock_manager_state_t g_clockState;



45



46/*FUNCTION**********************************************************************



47 *



48 * Function Name : CLOCK_SYS_Init



49 * Description   : Install pre-defined clock configurations.



50 * This function installs the pre-defined clock configuration table to the



51 * clock manager.



52 *



53 * Implements CLOCK_SYS_Init_Activity



54 *END**************************************************************************/



55status_t CLOCK_SYS_Init(clock_manager_user_config_t const **clockConfigsPtr,



56  uint8_t configsNumber,



57  clock_manager_callback_user_config_t **callbacksPtr,



58  uint8_t callbacksNumber)



59{



60  DEV_ASSERT(clockConfigsPtr != NULL);



61  DEV_ASSERT(callbacksPtr != NULL);



62  g_clockState.configTable = clockConfigsPtr;



63  g_clockState.clockConfigNum = configsNumber;



64  g_clockState.callbackConfig = callbacksPtr;



65  g_clockState.callbackNum = callbacksNumber;



66



67  /*



68   * errorCallbackIndex is the index of the callback which returns error



69   * during clock mode switch. If all callbacks return success, then the



70   * errorCallbackIndex is callbacksNumber.



71   */



72  g_clockState.errorCallbackIndex = callbacksNumber;



73  return STATUS_SUCCESS;



74}



75



76/*FUNCTION**********************************************************************



77 *



78 * Function Name : CLOCK_SYS_UpdateConfiguration



79 * Description   : Send notification and change system clock configuration.



80 * This function sends the notification to all callback functions, if all



81 * callbacks return OK or forceful policy is used, this function will change



82 * system clock configuration. The function should be called only on run mode.



83 *



84 * Implements CLOCK_SYS_UpdateConfiguration_Activity



85 *END**************************************************************************/



86status_t CLOCK_SYS_UpdateConfiguration(uint8_t targetConfigIndex,



87  clock_manager_policy_t policy)



88{



89  uint8_t callbackIdx;



90  bool successfulSetConfig;            /* SetConfiguraiton status */



91  status_t ret = STATUS_SUCCESS;



92  const clock_manager_callback_user_config_t * callbackConfig;



93  clock_notify_struct_t notifyStruct;



94  DEV_ASSERT(targetConfigIndex < g_clockState.clockConfigNum);/* Clock configuration index is out of range. */



95  notifyStruct.targetClockConfigIndex = targetConfigIndex;



96  notifyStruct.policy = policy;



97



98  /* Disable interrupts */



99  INT_SYS_DisableIRQGlobal();



100



101  /* Set errorcallbackindex as callbackNum, which means no callback error now.*/



102  g_clockState.errorCallbackIndex = g_clockState.callbackNum;



103



104  /* First step: Send "BEFORE" notification. */



105  notifyStruct.notifyType = CLOCK_MANAGER_NOTIFY_BEFORE;



106



107  /* Send notification to all callback. */



108  for (callbackIdx=0; callbackIdx<g_clockState.callbackNum; callbackIdx++) {



109    callbackConfig = g_clockState.callbackConfig[callbackIdx];



110    if ((callbackConfig) &&



111        (callbackConfig->callbackType != CLOCK_MANAGER_CALLBACK_AFTER)) {



112      if (STATUS_SUCCESS !=



113          (*callbackConfig->callback)(&notifyStruct,



114           callbackConfig->callbackData)) {



115        g_clockState.errorCallbackIndex = callbackIdx;



116        if (CLOCK_MANAGER_POLICY_AGREEMENT == policy) {



117          /* Save the error callback index. */



118          ret = STATUS_MCU_NOTIFY_BEFORE_ERROR;



119          break;



120        }



121      }



122    }



123  }



124



125  /* If all callback success or forceful policy is used. */



126  if ((STATUS_SUCCESS == ret) ||



127      (policy == CLOCK_MANAGER_POLICY_FORCIBLE)) {



128    /* clock mode switch. */



129    ret = CLOCK_SYS_SetConfiguration(g_clockState.configTable[targetConfigIndex]);



130    successfulSetConfig = (STATUS_SUCCESS == ret);



131    g_clockState.curConfigIndex = targetConfigIndex;



132  } else {



133    /* Unsuccessful setConfiguration */



134    successfulSetConfig = false;



135  }



136



137  if (successfulSetConfig) {



138    notifyStruct.notifyType = CLOCK_MANAGER_NOTIFY_AFTER;



139    for (callbackIdx=0; callbackIdx<g_clockState.callbackNum; callbackIdx++) {



140      callbackConfig = g_clockState.callbackConfig[callbackIdx];



141      if ((callbackConfig) &&



142          (callbackConfig->callbackType != CLOCK_MANAGER_CALLBACK_BEFORE)) {



143        if (STATUS_SUCCESS !=



144            (*callbackConfig->callback)(&notifyStruct,



145             callbackConfig->callbackData)) {



146          g_clockState.errorCallbackIndex = callbackIdx;



147          if (CLOCK_MANAGER_POLICY_AGREEMENT == policy) {



148            /* Save the error callback index. */



149            ret = STATUS_MCU_NOTIFY_AFTER_ERROR;



150            break;



151          }



152        }



153      }



154    }



155  } else                               /* Error occurs, need to send "RECOVER" notification. */



156  {



157    notifyStruct.notifyType = CLOCK_MANAGER_NOTIFY_RECOVER;



158    for (;;) {



159      callbackConfig = g_clockState.callbackConfig[callbackIdx];



160      if (callbackConfig != NULL) {



161        (void)(*callbackConfig->callback)(&notifyStruct,



162          callbackConfig->callbackData);



163      }



164



165      if (callbackIdx == 0U) {



166        break;



167      }



168



169      callbackIdx--;



170    }



171  }



172



173  /* Enable interrupts */



174  INT_SYS_EnableIRQGlobal();



175  return ret;



176}

178/*FUNCTION**********************************************************************



179 *



180 * Function Name : CLOCK_SYS_GetCurrentConfiguration



181 * Description   : Get current clock configuration index.



182 *



183 * Implements CLOCK_SYS_GetCurrentConfiguration_Activity



184 *END**************************************************************************/



185uint8_t CLOCK_SYS_GetCurrentConfiguration(void)



186{



187  return g_clockState.curConfigIndex;



188}



189



190/*FUNCTION**********************************************************************



191 *



192 * Function Name : CLOCK_SYS_GetErrorCallback



193 * Description   : Get the callback which returns error in last clock switch.



194 *



195 * Implements CLOCK_SYS_GetErrorCallback_Activity



196 *END**************************************************************************/



197clock_manager_callback_user_config_t* CLOCK_SYS_GetErrorCallback(void)



198{



199  clock_manager_callback_user_config_t *retValue;



200



201  /* If all callbacks return success. */



202  if (g_clockState.errorCallbackIndex >= g_clockState.clockConfigNum) {



203    retValue = NULL;



204  } else {



205    retValue = g_clockState.callbackConfig[g_clockState.errorCallbackIndex];



206  }



207



208  return retValue;



209}
