41/*FUNCTION**********************************************************************



42 *



43 * Function Name : PINS_Init



44 * Description   : This function configures the pins with the options provided



45 * in the provided structure.



46 *



47 *END**************************************************************************/



48void PINS_Init(const pin_settings_config_t * config)



49{



50  DEV_ASSERT(config->base != NULL);



51  DEV_ASSERT((PORT_MUX_AS_GPIO != config->mux) || (config->gpioBase != NULL));



52  DEV_ASSERT(config->pinPortIdx < PORT_PCR_COUNT);



53  uint32_t regValue = config->base->PCR[config->pinPortIdx];



54  uint32_t directions;



55  uint32_t digitalFilters;



56



57#if FEATURE_PINS_HAS_PULL_SELECTION



58



59  switch (config->pullConfig)



60  {



61   case PORT_INTERNAL_PULL_NOT_ENABLED:



62    {



63      regValue &= ~(PORT_PCR_PE_MASK);



64    }



65    break;



66



67   case PORT_INTERNAL_PULL_DOWN_ENABLED:



68    {



69      regValue &= ~(PORT_PCR_PS_MASK);



70      regValue |= PORT_PCR_PE(1U);



71    }



72    break;



73



74   case PORT_INTERNAL_PULL_UP_ENABLED:



75    {



76      regValue |= PORT_PCR_PE(1U);



77      regValue |= PORT_PCR_PS(1U);



78    }



79    break;



80



81   default:



82    /* invalid command */



83    DEV_ASSERT(false);



84    break;



85  }



86



87#endif                                 /* FEATURE_PINS_HAS_PULL_SELECTION */



88



89#if FEATURE_PINS_HAS_OVER_CURRENT



90



91  switch (config->overCurConfig)



92  {



93   case PORT_OVER_CURRENT_DISABLED:



94    {



95      regValue &= ~(PORT_PCR_OCE_MASK);



96    }



97    break;



98



99   case PORT_OVER_CURRENT_INT_DISABLED:



100    {



101      regValue &= ~(PORT_PCR_OCIE_MASK);



102      regValue |= PORT_PCR_OCE_MASK;



103    }



104    break;



105



106   case PORT_OVER_CURRENT_INT_ENABLED:



107    {



108      regValue |= PORT_PCR_OCIE_MASK;



109      regValue |= PORT_PCR_OCE_MASK;



110    }



111    break;



112



113   default:



114    /* invalid command */



115    DEV_ASSERT(false);



116    break;



117  }



118



119  if (config->clearOCurFlag) {



120    regValue &= ~(PORT_PCR_OCF_MASK);



121    regValue |= PORT_PCR_OCF(1U);



122  }



123



124#endif                                 /* FEATURE_PINS_HAS_OVER_CURRENT */



125



126#if FEATURE_PINS_HAS_SLEW_RATE



127



128  regValue &= ~(PORT_PCR_SRE_MASK);



129  regValue |= PORT_PCR_SRE(config->rateSelect);



130



131#endif



132



133#if FEATURE_PORT_HAS_PASSIVE_FILTER



134



135  regValue &= ~(PORT_PCR_PFE_MASK);



136  regValue |= PORT_PCR_PFE(config->passiveFilter);



137



138#endif



139



140#if FEATURE_PINS_HAS_OPEN_DRAIN



141



142  regValue &= ~(PORT_PCR_ODE_MASK);



143  regValue |= PORT_PCR_ODE(config->openDrain);



144



145#endif



146



147#if FEATURE_PINS_HAS_DRIVE_STRENGTH



148



149  regValue &= ~(PORT_PCR_DSE_MASK);



150  regValue |= PORT_PCR_DSE(config->driveSelect);



151



152#endif



153



154  regValue &= ~(PORT_PCR_MUX_MASK);



155  regValue |= PORT_PCR_MUX(config->mux);



156



157#if FEATURE_PORT_HAS_PIN_CONTROL_LOCK



158



159  regValue &= ~(PORT_PCR_LK_MASK);



160  regValue |= PORT_PCR_LK(config->pinLock);



161



162#endif



163



164  regValue &= ~(PORT_PCR_IRQC_MASK);



165  regValue |= PORT_PCR_IRQC(config->intConfig);



166  if (config->clearIntFlag) {



167    regValue &= ~(PORT_PCR_ISF_MASK);



168    regValue |= PORT_PCR_ISF(1U);



169  }



170



171  config->base->PCR[config->pinPortIdx] = regValue;



172



173  /* Read current digital filter of port */



174  digitalFilters = (uint32_t)(config->base->DFER);



175  digitalFilters &= ~(1UL << (config->pinPortIdx));



176  digitalFilters |= (((uint32_t)(config->digitalFilter)) << (config->pinPortIdx));



177



178  /* Write to digital filter enable register */



179  config->base->DFER = digitalFilters;



180



181  /* If gpioBase address not null setup the direction of pin */



182  if (PORT_MUX_AS_GPIO == config->mux) {



183    /* Read current direction */



184    directions = (uint32_t)(config->gpioBase->PDDR);



185    switch (config->direction)



186    {



187     case GPIO_INPUT_DIRECTION:



188      directions &= ~(1UL << config->pinPortIdx);



189      break;



190



191     case GPIO_OUTPUT_DIRECTION:



192      directions |= (1UL << config->pinPortIdx);



193      break;



194



195     case GPIO_UNSPECIFIED_DIRECTION:



196      /* pass-through */



197     default:



198      /* nothing to configure */



199      DEV_ASSERT(false);



200      break;



201    }



202



203    /* Configure direction */



204    config->gpioBase->PDDR = GPIO_PDDR_PDD(directions);



205



206    /* Configure initial value for output */



207    if (config->direction == GPIO_OUTPUT_DIRECTION) {



208      PINS_GPIO_WritePin(config->gpioBase, config->pinPortIdx, config->initValue);



209    }



210  }



211}

213/*FUNCTION**********************************************************************



214 *



215 * Function Name : PINS_SetGlobalPinControl



216 * Description   : Quickly configures multiple pins with the same pin configuration.



217 *



218 *END**************************************************************************/



219void PINS_SetGlobalPinControl(PORT_Type * const base,



220  uint16_t pins,



221  uint16_t value,



222  port_global_control_pins_t halfPort)



223{



224  uint16_t mask = 0;



225



226  /* keep only available fields */



227  mask |= PORT_PCR_PS_MASK;



228  mask |= PORT_PCR_PE_MASK;



229



230#if FEATURE_PINS_HAS_SLEW_RATE



231



232  mask |= PORT_PCR_SRE_MASK;



233



234#endif                                 /* FEATURE_PINS_HAS_OPEN_DRAIN */



235



236  mask |= PORT_PCR_PFE_MASK;



237



238#if FEATURE_PINS_HAS_OPEN_DRAIN



239



240  mask |= PORT_PCR_ODE_MASK;



241



242#endif                                 /* FEATURE_PINS_HAS_OPEN_DRAIN */



243



244  mask |= PORT_PCR_DSE_MASK;



245  mask |= PORT_PCR_MUX_MASK;



246



247#if FEATURE_PINS_HAS_OVER_CURRENT



248



249  mask |= PORT_PCR_OCE_MASK;



250



251#endif                                 /* FEATURE_PINS_HAS_OVER_CURRENT */



252



253  mask |= PORT_PCR_LK_MASK;



254  mask &= value;



255  switch (halfPort)



256  {



257   case PORT_GLOBAL_CONTROL_LOWER_HALF_PINS:



258    base->GPCLR = (((uint32_t)pins) << PORT_GPCLR_GPWE_SHIFT) | (uint32_t)mask;



259    break;



260



261   case PORT_GLOBAL_CONTROL_UPPER_HALF_PINS:



262    base->GPCHR = (((uint32_t)pins) << PORT_GPCLR_GPWE_SHIFT) | (uint32_t)mask;



263    break;



264



265   default:



266    /* nothing to configure */



267    DEV_ASSERT(false);



268    break;



269  }



270}

272/*FUNCTION**********************************************************************



273 *



274 * Function Name : PINS_SetGlobalIntControl



275 * Description   : Quickly configures multiple pins with the same interrupt configuration.



276 *



277 *END**************************************************************************/



278void PINS_SetGlobalIntControl(PORT_Type * const base,



279  uint16_t pins,



280  uint16_t value,



281  port_global_control_pins_t halfPort)



282{



283  uint32_t mask;



284  mask = (((uint32_t)value) << PORT_GICLR_GIWD_SHIFT) & PORT_PCR_IRQC_MASK;



285  switch (halfPort)



286  {



287   case PORT_GLOBAL_CONTROL_LOWER_HALF_PINS:



288    base->GICLR = ((uint32_t)pins) | mask;



289    break;



290



291   case PORT_GLOBAL_CONTROL_UPPER_HALF_PINS:



292    base->GICHR = ((uint32_t)pins) | mask;



293    break;



294



295   default:



296    /* nothing to configure */



297    DEV_ASSERT(false);



298    break;



299  }



300}
