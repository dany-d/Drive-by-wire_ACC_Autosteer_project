79/*FUNCTION**********************************************************************



80 *



81 * Function Name : INT_SYS_InstallHandler



82 * Description   : Install an interrupt handler routine for a given IRQ number



83 * This function will let application register/replace the interrupt



84 * handler for specified IRQ number. IRQ number is different than Vector



85 * number. IRQ 0 will start from Vector 16 address. Refer to reference



86 * manual for details. Also refer to startup_<CPU>.s file for each chip



87 * family to find out the default interrupt handler for each device. This



88 * function will convert the IRQ number to vector number by adding 16 to



89 * it.



90 *



91 * Note          : This method is applicable only if interrupt vector is



92 *                 copied in RAM, __flash_vector_table__ symbol is used to



93 *                 control this from linker options.



94 * Implements INT_SYS_InstallHandler_Activity



95 *



96 *END**************************************************************************/



97void INT_SYS_InstallHandler(IRQn_Type irqNumber,



98  const isr_t newHandler,



99  isr_t* const oldHandler)



100{



101



102#if (defined(DEV_ERROR_DETECT) || defined(CUSTOM_DEVASSERT))



103



104  /* Check IRQ number */



105  int32_t dev_irqNumber = (int32_t)irqNumber;



106  DEV_ASSERT(FEATURE_INTERRUPT_IRQ_MIN <= irqNumber);



107  DEV_ASSERT(dev_irqNumber <= (int32_t)FEATURE_INTERRUPT_IRQ_MAX);



108  DEV_ASSERT(__VECTOR_RAM != 0U);



109



110  /* Check whether there is vector table in RAM */



111  DEV_ASSERT((uint32_t)__VECTOR_RAM == S32_SCB->VTOR);



112



113#endif                                 /*(defined(DEV_ERROR_DETECT) || defined(CUSTOM_DEVASSERT)) */



114



115  /* Save the former handler pointer */



116  if (oldHandler != (isr_t *) 0) {



117    *oldHandler = (isr_t)__VECTOR_RAM[((int32_t)irqNumber) + 16];



118  }



119



120#if FEATURE_MSCM_HAS_INTERRUPT_ROUTER



121#if (defined(DEV_ERROR_DETECT) || defined(CUSTOM_DEVASSERT))



122



123  DEV_ASSERT((uint32_t)irqNumber < MSCM_IRSPRC_COUNT);



124



125  /* Check routing is not read-only in case it needs to be written */



126  uint16_t cpu_enable = (uint16_t)(1UL << (MSCM->CPXNUM));



127  if ((MSCM->IRSPRC[irqNumber] & cpu_enable) == 0U) {



128    DEV_ASSERT((MSCM->IRSPRC[irqNumber] & (uint16_t)(MSCM_IRSPRC_RO_MASK)) ==



129               (uint16_t)MSCM_IRSPRC_RO(0));



130  }



131



132#endif                                 /*(defined(DEV_ERROR_DETECT) || defined(CUSTOM_DEVASSERT)) */



133#endif                                 /* FEATURE_MSCM_HAS_INTERRUPT_ROUTER */



134



135  /* Set handler into vector table */



136  __VECTOR_RAM[((int32_t)irqNumber) + 16] = (uint32_t)newHandler;



137}

139/*FUNCTION**********************************************************************



140 *



141 * Function Name : INT_SYS_EnableIRQ



142 * Description   : Enables an interrupt for a given IRQ number.



143 * It calls the system NVIC API to access the interrupt control



144 * register and MSCM (if available) API for interrupt routing.



145 * The input IRQ number does not include the core interrupt, only



146 * the peripheral interrupt, from 0 to a maximum supported IRQ.



147 * Implements INT_SYS_EnableIRQ_Activity



148 *END**************************************************************************/



149void INT_SYS_EnableIRQ(IRQn_Type irqNumber)



150{



151  /* Check IRQ number */



152  DEV_ASSERT(0 <= (int32_t)irqNumber);



153  DEV_ASSERT((uint32_t)irqNumber <= (uint32_t)FEATURE_INTERRUPT_IRQ_MAX);



154



155  /* Enable interrupt */



156  S32_NVIC->ISER[(uint32_t)(irqNumber) >> 5U] = (uint32_t)(1UL << ((uint32_t)



157    (irqNumber) & (uint32_t)0x1FU));



158



159#if FEATURE_MSCM_HAS_INTERRUPT_ROUTER



160



161  /* Enable routing to current CPU */



162  uint16_t cpu_enable = (uint16_t)(1UL << (MSCM->CPXNUM));



163  MSCM->IRSPRC[irqNumber] |= cpu_enable;



164



165#endif                                 /* FEATURE_MSCM_HAS_INTERRUPT_ROUTER */



166



167}

172 * Description   : Disable individual interrupt for a specified IRQ

174 * and MSCM (if available) API for interrupt routing.



175 * Implements INT_SYS_DisableIRQ_Activity

177 *END**************************************************************************/



178void INT_SYS_DisableIRQ(IRQn_Type irqNumber)



179{



180  /* Check IRQ number */



181  DEV_ASSERT(0 <= (int32_t)irqNumber);



182  DEV_ASSERT((uint32_t)irqNumber <= (uint32_t)FEATURE_INTERRUPT_IRQ_MAX);



183



184  /* Disable interrupt */



185  S32_NVIC->ICER[((uint32_t)(irqNumber) >> 5U)] = (uint32_t)(1UL << ((uint32_t)



186    (irqNumber) & (uint32_t)0x1FU));



187



188#if FEATURE_MSCM_HAS_INTERRUPT_ROUTER



189



190  /* Disable routing to current CPU */



191  uint16_t cpu_enable = (uint16_t)(1UL << (MSCM->CPXNUM));



192  MSCM->IRSPRC[irqNumber] &= (uint16_t)~(cpu_enable);



193



194#endif                                 /* FEATURE_MSCM_HAS_INTERRUPT_ROUTER */



195



196}

235/*FUNCTION**********************************************************************



236 *



237 * Function Name : INT_SYS_SetPriority



238 * Description   : Set the priority of an interrupt



239 * This function will set the priority of an interrupt.



240 * Note: The priority cannot be set for every core interrupt.



241 * Implements INT_SYS_SetPriority_Activity



242 *



243 *END**************************************************************************/



244void INT_SYS_SetPriority(IRQn_Type irqNumber, uint8_t priority)



245{



246



247#if (defined(DEV_ERROR_DETECT) || defined(CUSTOM_DEVASSERT))



248



249  /* Check IRQ number and priority. */



250  int32_t dev_irqNumber = (int32_t)irqNumber;



251  DEV_ASSERT(dev_irqNumber <= (int32_t)FEATURE_INTERRUPT_IRQ_MAX);



252  DEV_ASSERT(priority < (uint8_t)(1U << FEATURE_NVIC_PRIO_BITS));



253



254#endif                                 /*(defined(DEV_ERROR_DETECT) || defined(CUSTOM_DEVASSERT)) */



255



256  uint8_t shift = (uint8_t) (8U - FEATURE_NVIC_PRIO_BITS);



257  if ((int32_t)irqNumber < 0) {



258    uint32_t intVectorId = ((uint32_t)(irqNumber) & 0xFU);



259    uint32_t regId = intVectorId / 4U;



260



261    /* Compute pointer to SHPR register - avoid MISRA violation. */



262#if defined (S32K11x_SERIES)



263



264    DEV_ASSERT((int32_t)regId != 1);



265    volatile uint32_t * shpr_reg_ptr = ((regId == 2U) ? (volatile uint32_t *)



266      &S32_SCB->SHPR2 : (volatile uint32_t *)&S32_SCB->SHPR3);



267    uint8_t priByteShift = ((uint8_t)(intVectorId) & 0x3U) << 3U;



268



269    /* Clear the old value from the register */



270    *shpr_reg_ptr &= ~(0xFFUL << priByteShift);



271



272    /* Set Priority for Cortex-M0P System Interrupts */



273    *shpr_reg_ptr |= ((uint32_t)(((((uint32_t)priority) << shift)) & 0xFFUL)) <<



274      priByteShift;



275



276#else



277



278    volatile uint8_t * shpr_reg_ptr = ((regId == 1U) ? (volatile uint8_t *)



279      &S32_SCB->SHPR1 : ((regId == 2U) ? (volatile uint8_t *)&S32_SCB->SHPR2 :



280                         (volatile uint8_t *)&S32_SCB->SHPR3));



281



282    /* Set Priority for Cortex-M4 System Interrupts */



283    shpr_reg_ptr[intVectorId % 4U] = (uint8_t)(((((uint32_t)priority) << shift))



284      & 0xffUL);



285



286#endif                                 /* defined (S32K11x_SERIES) */



287



288  } else {



289    /* Set Priority for device specific Interrupts */



290#if defined (S32K11x_SERIES)



291



292    uint32_t iprVectorId = (uint32_t)(irqNumber) >> 2U;



293    uint8_t priByteShift = (((uint8_t)(irqNumber)) & 0x3U) << 3U;



294



295    /* Clear the old value from the register */



296    S32_NVIC->IPR[iprVectorId] &= ~(0xFFUL << priByteShift);



297    S32_NVIC->IPR[iprVectorId] |= ((uint32_t)(((((uint32_t)priority) << shift))



298      & 0xFFUL)) << priByteShift;



299



300#else



301



302    S32_NVIC->IP[(uint32_t)(irqNumber)] = (uint8_t)(((((uint32_t)priority) <<



303      shift)) & 0xFFUL);



304



305#endif                                 /* defined (S32K11x_SERIES) */



306



307  }



308}

310/*FUNCTION**********************************************************************



311 *



312 * Function Name : INT_SYS_GetPriority



313 * Description   : Get the priority of an interrupt



314 * This function will get the priority of an interrupt.



315 * Note: The priority cannot be obtained for every core interrupt.



316 * Implements INT_SYS_GetPriority_Activity



317 *



318 *END**************************************************************************/



319uint8_t INT_SYS_GetPriority(IRQn_Type irqNumber)



320{



321



322#if (defined(DEV_ERROR_DETECT) || defined(CUSTOM_DEVASSERT))



323



324  /* Check IRQ number. */



325  int32_t dev_irqNumber = (int32_t)irqNumber;



326  DEV_ASSERT(dev_irqNumber <= (int32_t)FEATURE_INTERRUPT_IRQ_MAX);



327



328#endif                                 /*(defined(DEV_ERROR_DETECT) || defined(CUSTOM_DEVASSERT)) */



329



330  uint8_t priority = 0U;



331  uint8_t shift = (uint8_t) (8U - FEATURE_NVIC_PRIO_BITS);



332  if ((int32_t)irqNumber < 0) {



333    uint32_t intVectorId = ((uint32_t)(irqNumber) & 0xFU);



334    uint32_t regId = intVectorId / 4U;



335



336    /* Compute pointer to SHPR register - avoid MISRA violation. */



337#if defined (S32K11x_SERIES)



338



339    DEV_ASSERT((int32_t)regId != 1);



340    volatile const uint32_t * shpr_reg_ptr = ((regId == 2U) ? (volatile uint32_t



341      *)&S32_SCB->SHPR2 : (volatile uint32_t *)&S32_SCB->SHPR3);



342    uint8_t priByteShift = ((uint8_t)(intVectorId) & 0x3U) << 3U;



343    priority = ((uint8_t)(*shpr_reg_ptr >> priByteShift)) >> shift;



344



345#else



346



347    volatile const uint8_t * shpr_reg_ptr = ((regId == 1U) ? (volatile uint8_t *)



348      &S32_SCB->SHPR1 : ((regId == 2U) ? (volatile uint8_t *)&S32_SCB->SHPR2 :



349                         (volatile uint8_t *)&S32_SCB->SHPR3));



350



351    /* Get Priority from Cortex-M  System Interrupts */



352    priority = (uint8_t)(shpr_reg_ptr[intVectorId % 4U] >> (shift));



353



354#endif                                 /* defined (S32K11x_SERIES) */



355



356  } else {



357    /* Get Priority for device specific Interrupts  */



358#if defined (S32K11x_SERIES)



359



360    uint32_t iprVectorId = (uint32_t)(irqNumber) >> 2U;



361    uint8_t priByteShift = (((uint8_t)(irqNumber)) & 0x3U) << 3U;



362    priority = ((uint8_t)(S32_NVIC->IPR[iprVectorId] >> priByteShift)) >> shift;



363



364#else



365



366    priority = (uint8_t)(S32_NVIC->IP[(uint32_t)(irqNumber)] >> shift);



367



368#endif                                 /* defined (S32K11x_SERIES) */



369



370  }



371



372  return priority;



373}

498/*!



499 * @brief Generate Directed CPU Interrupt



500 *



501 * The function generates a directed interrupt to (one or more) CPUs defined by target.



502 * Implements INT_SYS_GenerateDirectedCpuInterrupt_Activity



503 *



504 * @param irqNumber  IRQ number



505 * @param cpu_target Target CPUs for the directed interrupt



506 */



507/*FUNCTION**********************************************************************



508 *



509 * Function Name : INT_SYS_GenerateDirectedCpuInterrupt



510 * Description   : Get active state of a peripheral interrupt



511 * This function gets the active state of a peripheral interrupt.



512 * Implements INT_SYS_GenerateDirectedCpuInterrupt_Activity



513 *



514 *END**************************************************************************/



515void INT_SYS_GenerateDirectedCpuInterrupt(IRQn_Type irqNumber,



516  interrupt_manager_cpu_targets_t cpu_target)



517{



518  /* Check IRQ number */



519  DEV_ASSERT(FEATURE_DIRECTED_CPU_INT_MIN <= irqNumber);



520  DEV_ASSERT(irqNumber <= FEATURE_DIRECTED_CPU_INT_MAX);



521  uint32_t reg_val = MSCM_IRCPGIR_INTID((uint32_t)irqNumber - (uint32_t)



522    FEATURE_DIRECTED_CPU_INT_MIN);



523  switch (cpu_target)



524  {



525   case INTERRUPT_MANAGER_TARGET_SELF:



526    reg_val |= MSCM_IRCPGIR_TLF(2);



527    break;



528



529   case INTERRUPT_MANAGER_TARGET_OTHERS:



530    reg_val |= MSCM_IRCPGIR_TLF(1);



531    break;



532



533   case INTERRUPT_MANAGER_TARGET_NONE:



534   case INTERRUPT_MANAGER_TARGET_CP0:



535   case INTERRUPT_MANAGER_TARGET_CP1:



536   case INTERRUPT_MANAGER_TARGET_CP0_CP1:



537    reg_val |= (MSCM_IRCPGIR_TLF(0) | MSCM_IRCPGIR_CPUTL(cpu_target));



538    break;



539



540   default:



541    /* Not treated case ? */



542    break;



543  }



544



545  /* Generate Directed CPU Interrupt */



546  MSCM->IRCPGIR = reg_val;



547}
