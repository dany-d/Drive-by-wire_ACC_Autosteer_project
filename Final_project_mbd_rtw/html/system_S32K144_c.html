54/*FUNCTION**********************************************************************



55 *



56 * Function Name : SystemInit



57 * Description   : This function disables the watchdog, enables FPU



58 * and the power mode protection if the corresponding feature macro



59 * is enabled. SystemInit is called from startup_device file.



60 *



61 * Implements    : SystemInit_Activity



62 *END**************************************************************************/



63void SystemInit(void)



64{



65  /**************************************************************************/



66  /* FPU ENABLE*/



67  /**************************************************************************/



68#ifdef ENABLE_FPU



69



70  /* Enable CP10 and CP11 coprocessors */



71  S32_SCB->CPACR |= (S32_SCB_CPACR_CP10_MASK | S32_SCB_CPACR_CP11_MASK);



72



73#ifdef ERRATA_E6940



74



75  /* Disable lazy context save of floating point state by clearing LSPEN bit



76   * Workaround for errata e6940 */



77  S32_SCB->FPCCR &= ~(S32_SCB_FPCCR_LSPEN_MASK);



78



79#endif



80#endif                                 /* ENABLE_FPU */



81



82  /**************************************************************************/



83  /* WDOG DISABLE*/



84  /**************************************************************************/



85#if (DISABLE_WDOG)



86



87  /* Write of the WDOG unlock key to CNT register, must be done in order to allow any modifications*/



88  WDOG->CNT = (uint32_t ) FEATURE_WDOG_UNLOCK_VALUE;



89



90  /* The dummy read is used in order to make sure that the WDOG registers will be configured only



91   * after the write of the unlock value was completed. */



92  (void)WDOG->CNT;



93



94  /* Initial write of WDOG configuration register:



95   * enables support for 32-bit refresh/unlock command write words,



96   * clock select from LPO, update enable, watchdog disabled */



97  WDOG->CS = (uint32_t ) ( (1UL << WDOG_CS_CMD32EN_SHIFT) |



98    (FEATURE_WDOG_CLK_FROM_LPO << WDOG_CS_CLK_SHIFT) |



99    (0U << WDOG_CS_EN_SHIFT) |



100    (1U << WDOG_CS_UPDATE_SHIFT) );



101



102  /* Configure timeout */



103  WDOG->TOVAL = (uint32_t )0xFFFF;



104



105#endif                                 /* (DISABLE_WDOG) */



106



107  /**************************************************************************/



108  /* ENABLE CACHE */



109  /**************************************************************************/



110#if defined(I_CACHE) && (ICACHE_ENABLE == 1)



111



112  /* Invalidate and enable code cache */



113  LMEM->PCCCR = LMEM_PCCCR_INVW0(1) | LMEM_PCCCR_INVW1(1) | LMEM_PCCCR_GO(1) |



114    LMEM_PCCCR_ENCACHE(1);



115



116#endif                                 /* defined(I_CACHE) && (ICACHE_ENABLE == 1) */



117



118}

120/*FUNCTION**********************************************************************



121 *



122 * Function Name : SystemCoreClockUpdate



123 * Description   : This function must be called whenever the core clock is changed



124 * during program execution. It evaluates the clock register settings and calculates



125 * the current core clock.



126 *



127 * Implements    : SystemCoreClockUpdate_Activity



128 *END**************************************************************************/



129void SystemCoreClockUpdate(void)



130{



131  uint32_t SCGOUTClock = 0U;           /* Variable to store output clock frequency of the SCG module */



132  uint32_t regValue;                   /* Temporary variable */



133  uint32_t divider, prediv, multi;



134  bool validSystemClockSource = true;



135  static const uint32_t fircFreq[] = {



136    FEATURE_SCG_FIRC_FREQ0,



137  };



138



139  divider = ((SCG->CSR & SCG_CSR_DIVCORE_MASK) >> SCG_CSR_DIVCORE_SHIFT) + 1U;



140  switch ((SCG->CSR & SCG_CSR_SCS_MASK) >> SCG_CSR_SCS_SHIFT) {



141   case 0x1:



142    /* System OSC */



143    SCGOUTClock = CPU_XTAL_CLK_HZ;



144    break;



145



146   case 0x2:



147    /* Slow IRC */



148    regValue = (SCG->SIRCCFG & SCG_SIRCCFG_RANGE_MASK) >>



149      SCG_SIRCCFG_RANGE_SHIFT;



150    if (regValue != 0U) {



151      SCGOUTClock = FEATURE_SCG_SIRC_HIGH_RANGE_FREQ;



152    }



153    break;



154



155   case 0x3:



156    /* Fast IRC */



157    regValue = (SCG->FIRCCFG & SCG_FIRCCFG_RANGE_MASK) >>



158      SCG_FIRCCFG_RANGE_SHIFT;



159    SCGOUTClock= fircFreq[regValue];



160    break;



161



162   case 0x6:



163    /* System PLL */



164    SCGOUTClock = CPU_XTAL_CLK_HZ;



165    prediv = ((SCG->SPLLCFG & SCG_SPLLCFG_PREDIV_MASK) >>



166              SCG_SPLLCFG_PREDIV_SHIFT) + 1U;



167    multi = ((SCG->SPLLCFG & SCG_SPLLCFG_MULT_MASK) >> SCG_SPLLCFG_MULT_SHIFT) +



168      16U;



169    SCGOUTClock = SCGOUTClock * multi / (prediv * 2U);



170    break;



171



172   default:



173    validSystemClockSource = false;



174    break;



175  }



176



177  if (validSystemClockSource == true) {



178    SystemCoreClock = (SCGOUTClock / divider);



179  }



180}

182/*FUNCTION**********************************************************************



183 *



184 * Function Name : SystemSoftwareReset



185 * Description   : This function is used to initiate a system reset



186 *



187 * Implements    : SystemSoftwareReset_Activity



188 *END**************************************************************************/



189void SystemSoftwareReset(void)



190{



191  uint32_t regValue;



192



193  /* Read Application Interrupt and Reset Control Register */



194  regValue = S32_SCB->AIRCR;



195



196  /* Clear register key */



197  regValue &= ~( S32_SCB_AIRCR_VECTKEY_MASK);



198



199  /* Configure System reset request bit and Register Key */



200  regValue |= S32_SCB_AIRCR_VECTKEY(FEATURE_SCB_VECTKEY);



201  regValue |= S32_SCB_AIRCR_SYSRESETREQ(0x1u);



202



203  /* Write computed register value */



204  S32_SCB->AIRCR = regValue;



205}
