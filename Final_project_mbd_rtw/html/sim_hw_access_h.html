59/*!



60 * @brief Set the clock selection of LPOCLKSEL.



61 *



62 * This function sets the clock selection of LPOCLKSEL.



63 *



64 * @param[in] base Base address for current SIM instance.



65 * @param[in] setting The value to set.



66 * @note This function ignores initialize member



67 */



68static inline void SIM_SetLpoClocks(SIM_Type* base, bool enableLpo1k, bool enableLpo32k, uint32_t sourceLpoClk, uint32_t sourceRtcClk)



69{



70    uint32_t regValue = base->LPOCLKS;



71



72    regValue &= ~( SIM_LPOCLKS_LPO1KCLKEN_MASK  |



73                   SIM_LPOCLKS_LPO32KCLKEN_MASK |



74                   SIM_LPOCLKS_LPOCLKSEL_MASK   |



75                   SIM_LPOCLKS_RTCCLKSEL_MASK   );



76



77    regValue |= SIM_LPOCLKS_LPO1KCLKEN(enableLpo1k ? 1UL : 0UL);



78    regValue |= SIM_LPOCLKS_LPO32KCLKEN(enableLpo32k ? 1UL : 0UL);



79    regValue |= SIM_LPOCLKS_LPOCLKSEL(sourceLpoClk);



80    regValue |= SIM_LPOCLKS_RTCCLKSEL(sourceRtcClk);



81



82    /* Write value to register. */



83    base->LPOCLKS = regValue;



84}

473/*!



474 * @brief Sets clockout



475 *



476 * @param[in] base        sim base pointer



477 * @param[in] source      clock source



478 * @param[in] divider     clock divider



479 */



480static inline void SIM_SetClockout(SIM_Type* base, bool enable, uint32_t source, uint32_t divider)



481{



482    uint32_t regValue = base->CHIPCTL;



483



484    regValue &= ~( SIM_CHIPCTL_CLKOUTEN_MASK  |



485    		       SIM_CHIPCTL_CLKOUTDIV_MASK |



486			       SIM_CHIPCTL_CLKOUTSEL_MASK );



487



488    regValue |= SIM_CHIPCTL_CLKOUTEN(enable ? 1UL : 0UL);



489    regValue |= SIM_CHIPCTL_CLKOUTSEL(source);



490    regValue |= SIM_CHIPCTL_CLKOUTDIV(divider);



491



492    base->CHIPCTL = regValue;



493}
