34/*FUNCTION**********************************************************************



35 *



36 * Function Name : FTM_DRV_InitPwm



37 * Description   : Configures duty cycle and frequency and starts outputting



38 * PWM on specified channels.



39 *



40 * Implements    : FTM_DRV_InitPwm_Activity



41 *END**************************************************************************/



42status_t FTM_DRV_InitPwm(uint32_t instance,



43  const ftm_pwm_param_t * param)



44{



45  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);



46  DEV_ASSERT(param != NULL);



47  status_t retVal = STATUS_SUCCESS;



48  uint8_t index = 0U;



49  uint8_t hwChannel = 0U;



50  uint8_t fltChannel = 0U;



51  uint8_t chnlPairNum = 0U;



52  uint8_t channelId = 0U;



53  uint8_t faultChnNum = 0U;



54  uint32_t tempInst = instance;



55  ftm_state_t * state = ftmStatePtr[instance];



56  FTM_Type * ftmBase = g_ftmBase[instance];



57  if ((NULL != state) && (FTM_MODE_NOT_INITIALIZED == state->ftmMode)) {



58    /* Disable counter clock */



59    FTM_DRV_SetClockSource(ftmBase, FTM_CLOCK_SOURCE_NONE);



60



61    /* Clear the overflow flag */



62    FTM_DRV_ClearTimerOverflow(ftmBase);



63



64    /* Disable write protection */



65    FTM_DRV_SetWriteProtectionCmd(ftmBase, false);



66



67    /* Configure FTM mode */



68    state->ftmMode = param->mode;



69



70    /* Configure independent PWM channels */



71    for (index = 0U; index < param->nNumIndependentPwmChannels; index++) {



72      channelId = param->pwmIndependentChannelConfig[index].hwChannelId;



73      chnlPairNum = (uint8_t)(channelId >> 1U);



74      FTM_DRV_SetDualEdgeCaptureCmd(ftmBase, chnlPairNum, false);



75



76      /* Configure the ELSB bit and ELSA bit from user */



77      FTM_DRV_SetChnEdgeLevel(ftmBase, channelId, (uint8_t)



78        param->pwmIndependentChannelConfig[index].levelSelect);



79



80      /* Set MSB and MSA bits*/



81      FTM_DRV_SetChnMSnBAMode(ftmBase, channelId, 3U);



82



83      /* Write FTMn_PWMLOAD register to enable synchronized loading points for the given channel */



84      FTM_DRV_EnablePwmChannelOutputs(ftmBase, channelId);



85      FTM_DRV_SetDualChnFaultCmd(ftmBase, chnlPairNum, ((param->faultConfig)



86        ->faultMode != FTM_FAULT_CONTROL_DISABLED) ? true : false);



87



88      /* Enable sync control for channels*/



89      FTM_DRV_SetDualChnPwmSyncCmd(ftmBase, chnlPairNum, true);



90      FTM_DRV_SetDualChnCombineCmd(ftmBase, chnlPairNum, false);



91      FTM_DRV_SetDualChnMofCombineCmd(ftmBase, chnlPairNum, false);



92



93      /* Enable the generation a trigger on chip module */



94      FTM_DRV_SetChnTriggerCmd(ftmBase, channelId,



95        param->pwmIndependentChannelConfig[index].enableExternalTrigger);



96



97      /* Enable channel (n+1) output */



98      if (param->pwmIndependentChannelConfig[index].enableSecondChannelOutput) {



99        /* Check if the channel id is even number */



100        DEV_ASSERT((channelId % 2U) == 0U);



101        channelId++;



102



103        /* Enable output */



104        FTM_DRV_EnablePwmChannelOutputs(ftmBase, channelId);



105



106        /* Set Edges to enable channel output */



107        FTM_DRV_SetChnEdgeLevel(ftmBase, channelId, (uint8_t)



108          param->pwmIndependentChannelConfig[index].levelSelect);



109



110        /* Set MSB and MSA bits*/



111        FTM_DRV_SetChnMSnBAMode(ftmBase, channelId, 3U);



112



113        /* Check if the channel (n+1) is inverse the channel (n) */



114        if (param->pwmIndependentChannelConfig[index].secondChannelPolarity ==



115            FTM_MAIN_INVERTED) {



116          FTM_DRV_SetDualChnCompCmd(ftmBase, chnlPairNum, true);



117        }



118      } else {



119        FTM_DRV_SetChnOutputPolarityCmd(ftmBase, channelId,



120          ((param->pwmIndependentChannelConfig[index].polarity ==



121            FTM_POLARITY_HIGH)? true : false));



122      }



123



124      /* Enable dead-time insertion */



125      FTM_DRV_SetDualChnDeadtimeCmd(ftmBase, chnlPairNum,



126        param->pwmIndependentChannelConfig[index].deadTime);



127    }



128



129    /* Configure combined PWM channels */



130    for (index = 0U; index < param->nNumCombinedPwmChannels; index++) {



131      channelId = param->pwmCombinedChannelConfig[index].hwChannelId;



132      chnlPairNum = (uint8_t)(channelId >> 1U);



133



134      /* Check if the channel id is even number */



135      DEV_ASSERT((channelId % 2U) == 0U);



136      FTM_DRV_SetDualEdgeCaptureCmd(ftmBase, chnlPairNum, false);



137



138      /* Configure the ELSB bit and ELSA bit from user */



139      FTM_DRV_SetChnEdgeLevel(ftmBase, channelId, (uint8_t)



140        param->pwmCombinedChannelConfig[index].levelSelect);



141



142      /* Set MSB and MSA bits */



143      FTM_DRV_SetChnMSnBAMode(ftmBase, channelId, 3U);



144



145      /* Enable channel (n) output */



146      FTM_DRV_EnablePwmChannelOutputs(ftmBase, channelId);



147



148      /* Enable channel (n+1) output */



149      if (param->pwmCombinedChannelConfig[index].enableSecondChannelOutput) {



150        FTM_DRV_EnablePwmChannelOutputs(ftmBase, (uint8_t)(channelId + 1U));



151



152        /* Configure the ELSB bit and ELSA bit from user */



153        FTM_DRV_SetChnEdgeLevel(ftmBase, (uint8_t)(channelId + 1U), (uint8_t)



154          param->pwmCombinedChannelConfig[index].levelSelectOnNextChn);



155



156        /* Configure complementary mode for channel (n+1) */



157        FTM_DRV_SetDualChnCompCmd(ftmBase, chnlPairNum,



158          ((param->pwmCombinedChannelConfig[index].secondChannelPolarity ==



159            FTM_MAIN_INVERTED)? true : false));



160      } else {



161        /* Set polarity for the main channel */



162        FTM_DRV_SetChnOutputPolarityCmd(ftmBase, channelId,



163          ((param->pwmCombinedChannelConfig[index].mainChannelPolarity ==



164            FTM_POLARITY_HIGH)? true : false));



165        FTM_DRV_DisablePwmChannelOutputs(ftmBase, (uint8_t)(channelId + 1U));



166      }



167



168      /* Set fault control for the channel */



169      FTM_DRV_SetDualChnFaultCmd(ftmBase, chnlPairNum, ((param->faultConfig)



170        ->faultMode != FTM_FAULT_CONTROL_DISABLED) ? true : false);



171



172      /* Enable sync control for channels */



173      FTM_DRV_SetDualChnPwmSyncCmd(ftmBase, chnlPairNum, true);



174



175      /* Enable the combine mode */



176      FTM_DRV_SetDualChnCombineCmd(ftmBase, chnlPairNum, true);



177



178      /* Configure the modified combine mode */



179      FTM_DRV_SetDualChnMofCombineCmd(ftmBase, chnlPairNum,



180        param->pwmCombinedChannelConfig[index].enableModifiedCombine);



181



182      /* Configure dead time */



183      FTM_DRV_SetDualChnDeadtimeCmd(ftmBase, chnlPairNum,



184        param->pwmCombinedChannelConfig[index].deadTime);



185



186      /* Enable the generation a trigger on the channel (n) */



187      FTM_DRV_SetChnTriggerCmd(ftmBase, (chnlPairNum << 1U),



188        param->pwmCombinedChannelConfig[index].enableExternalTrigger);



189



190      /* Enable the generation a trigger on the channel (n+1) */



191      FTM_DRV_SetChnTriggerCmd(ftmBase, (chnlPairNum << 1U) + 1U,



192        param->pwmCombinedChannelConfig[index].enableExternalTriggerOnNextChn);



193    }



194



195    /* Set enable outputs to be set to Initial/default value */



196    FTM_DRV_SetInitChnOutputCmd(ftmBase, true);



197



198    /* Enable faults (if faults were configured) */



199    if ((param->faultConfig)->faultMode != FTM_FAULT_CONTROL_DISABLED) {



200      /* Configure PWM Output behavior */



201      FTM_DRV_SetPwmFaultBehavior(ftmBase, ((param->faultConfig)



202        ->pwmOutputStateOnFault) ? true : false);



203



204      /* Configure fault filter value */



205      FTM_DRV_SetFaultInputFilterVal(ftmBase, ((param->faultConfig)



206        ->faultFilterValue));



207



208      /* Check the FTM instances */



209      if (tempInst <= 3U) {



210        faultChnNum = (uint8_t)FTM_FEATURE_FAULT_CHANNELS;



211      } else {



212        faultChnNum = (uint8_t)(FTM_FEATURE_FAULT_CHANNELS >> 1U);



213      }



214



215      for (fltChannel = 0U; fltChannel < faultChnNum; fltChannel++) {



216        if (true == (param->faultConfig)->ftmFaultChannelParam[fltChannel].



217            faultChannelEnabled) {



218          /* Enable fault channel */



219          FTM_DRV_SetFaultInputCmd(ftmBase, fltChannel, true);



220



221          /* Configure fault filter */



222          FTM_DRV_SetFaultInputFilterCmd(ftmBase,



223            fltChannel,



224            ((param->faultConfig)->ftmFaultChannelParam[fltChannel].



225             faultFilterEnabled) ? true : false);



226



227          /* Configure fault outputs */



228          FTM_DRV_SetChnFaultInputPolarityCmd(ftmBase,



229            fltChannel,



230            (((param->faultConfig)->ftmFaultChannelParam[fltChannel].



231              ftmFaultPinPolarity == FTM_POLARITY_HIGH)? true : false));



232        }



233      }



234



235      /* Set fault interrupt */



236      if (true == ((param->faultConfig)->pwmFaultInterrupt)) {



237        FTM_DRV_SetFaultInt(ftmBase, true);



238      }



239



240      /* Enable fault control */



241      FTM_DRV_SetFaultControlMode(ftmBase, (uint32_t)(param->faultConfig)



242        ->faultMode);



243    }



244



245    /* Configure PWM mode: edge or center aligned */



246    FTM_DRV_SetCpwms(ftmBase, (param->mode == FTM_MODE_CEN_ALIGNED_PWM) ? true :



247                     false);



248



249    /* Calculate frequency of the give FTM hardware module - all channels will run at the same frequency */



250    state->ftmPeriod = FTM_DRV_ConvertFreqToPeriodTicks(instance,



251      param->uFrequencyHZ);



252



253    /* Based on reference manual, in PWM mode CNTIN is to be set 0*/



254    FTM_DRV_SetCounterInitVal(ftmBase, 0U);



255



256    /* Write MOD register with the value of the period */



257    /* For center aligned mode MOD register should be divided by 2 */



258    /* For edge aligned mode period is determined by: MOD-CNTIN+1 */



259    if (param->mode == FTM_MODE_CEN_ALIGNED_PWM) {



260      FTM_DRV_SetMod(ftmBase, (uint16_t)(state->ftmPeriod >> 1U));



261    } else {



262      FTM_DRV_SetMod(ftmBase, (uint16_t)(state->ftmPeriod - 1U));



263    }



264



265    for (index = 0U; index < param->nNumIndependentPwmChannels; index++) {



266      hwChannel = param->pwmIndependentChannelConfig[index].hwChannelId;



267



268      /* Write CnV registers and setup duty cycle and phase values */



269      retVal = FTM_DRV_UpdatePwmChannel(instance,



270        hwChannel,



271        FTM_PWM_UPDATE_IN_DUTY_CYCLE,



272        param->pwmIndependentChannelConfig[index].uDutyCyclePercent,



273        0U,



274        false);



275



276      /* Enable channel (n+1) output for the same channel (n) */



277      if ((param->pwmIndependentChannelConfig[index].enableSecondChannelOutput) &&



278          (param->pwmIndependentChannelConfig[index].secondChannelPolarity ==



279           FTM_MAIN_DUPLICATED)) {



280        /* Write CnV registers and setup duty cycle and phase values */



281        retVal = FTM_DRV_UpdatePwmChannel(instance,



282          hwChannel + 1U,



283          FTM_PWM_UPDATE_IN_DUTY_CYCLE,



284          param->pwmIndependentChannelConfig[index].uDutyCyclePercent,



285          0U,



286          false);



287      }



288    }



289



290    for (index = 0U; index < param->nNumCombinedPwmChannels; index++) {



291      hwChannel = param->pwmCombinedChannelConfig[index].hwChannelId;



292



293      /* Write CnV registers and setup duty cycle and phase values */



294      retVal = FTM_DRV_UpdatePwmChannel(instance,



295        hwChannel,



296        FTM_PWM_UPDATE_IN_DUTY_CYCLE,



297        param->pwmCombinedChannelConfig[index].firstEdge,



298        param->pwmCombinedChannelConfig[index].secondEdge,



299        false);



300    }



301



302    if (STATUS_SUCCESS == retVal) {



303      /* Configure dead time for combine mode */



304      FTM_DRV_SetDeadtimeCount(ftmBase, param->deadTimeValue);



305      FTM_DRV_SetDeadtimePrescale(ftmBase, param->deadTimePrescaler);



306      FTM_DRV_Enable(ftmBase, true);



307      FTM_DRV_SetPwmSyncMode(ftmBase, true);



308



309      /* Set clock source to start counter */



310      FTM_DRV_SetClockSource(ftmBase, state->ftmClockSource);



311    } else {



312      /* Restore FTM mode if initialization fails */



313      state->ftmMode = FTM_MODE_NOT_INITIALIZED;



314    }



315  } else {



316    retVal = STATUS_ERROR;



317  }



318



319  return retVal;



320}

322/*FUNCTION**********************************************************************



323 *



324 * Function Name : FTM_DRV_DeinitPwm



325 * Description   : Stops all PWM channels.



326 *



327 * Implements    : FTM_DRV_DeinitPwm_Activity



328 *END**************************************************************************/



329status_t FTM_DRV_DeinitPwm(uint32_t instance)



330{



331  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);



332  FTM_Type * ftmBase = g_ftmBase[instance];



333  uint8_t channel;



334  uint8_t chnlPairNum;



335  ftm_state_t * state = ftmStatePtr[instance];



336



337  /* Stop the FTM counter */



338  FTM_DRV_SetClockSource(ftmBase, FTM_CLOCK_SOURCE_NONE);



339  for (channel = 0U; channel < FEATURE_FTM_CHANNEL_COUNT; channel++) {



340    chnlPairNum = (uint8_t)(channel >> 1U);



341



342    /* Disable PWM mode in hardware */



343    FTM_DRV_SetChnCountVal(ftmBase, channel, 0U);



344    FTM_DRV_SetChnEdgeLevel(ftmBase, channel, 0U);



345    FTM_DRV_SetChnMSnBAMode(ftmBase, channel, 0U);



346    FTM_DRV_SetCpwms(ftmBase, false);



347



348    /* Configure polarity bit */



349    FTM_DRV_SetChnOutputPolarityCmd(ftmBase, channel, false);



350    FTM_DRV_DisablePwmChannelOutputs(ftmBase, channel);



351



352    /* Clear the PWM synchronization */



353    FTM_DRV_SetDualChnPwmSyncCmd(ftmBase, chnlPairNum, false);



354



355    /* Clear combination for each pair of channels */



356    FTM_DRV_SetDualChnMofCombineCmd(ftmBase, chnlPairNum, false);



357    FTM_DRV_SetDualChnCombineCmd(ftmBase, chnlPairNum, false);



358    FTM_DRV_SetDualChnDeadtimeCmd(ftmBase, chnlPairNum, false);



359    FTM_DRV_SetDualChnFaultCmd(ftmBase, chnlPairNum, false);



360  }



361



362  /* Clear the dead-time pre-scaler and value */



363  FTM_DRV_SetExtDeadtimeValue(ftmBase, 0U);



364  FTM_DRV_SetDeadtimePrescale(ftmBase, FTM_DEADTIME_DIVID_BY_1);



365  FTM_DRV_SetDeadtimeCount(ftmBase, 0U);



366



367  /* Clear fault control register */



368  FTM_DRV_ClearFaultControl(ftmBase);



369



370  /* Disable fault interrupt */



371  FTM_DRV_SetFaultInt(ftmBase, false);



372



373  /* Disable fault control */



374  FTM_DRV_SetFaultControlMode(ftmBase, (uint32_t)FTM_FAULT_CONTROL_DISABLED);



375



376  /* Clear the module value of the registers */



377  FTM_DRV_SetMod(ftmBase, 0U);



378  FTM_DRV_SetCounter(ftmBase, 0U);



379  state->ftmMode = FTM_MODE_NOT_INITIALIZED;



380  return STATUS_SUCCESS;



381}

383/*FUNCTION**********************************************************************



384 *



385 * Function Name : FTM_DRV_UpdatePwmChannel



386 * Description   : This function will update the duty cycle of PWM output.



387 * - If the type of update in the duty cycle, this function will convert the input parameters representing



388 * frequency in Hz to a period value in ticks needed by the hardware timer. Period is calculated depending



389 * on the operating mode of the FTM module and is stored in internal state structure.



390 * firstEdge and secondEdge can have value between 0 - FTM_MAX_DUTY_CYCLE(0 = 0% duty



391 * and FTM_MAX_DUTY_CYCLE = 100% duty). secondEdge value is used only whenFTM module is used in PWM combine mode.



392 * - If the type of update in ticks, this function will get value in ticks to the hardware timer.



393 * firstEdge and secondEdge variables can have value between 0 and ftmPeriod is stored in the state structure.



394 * - in the modified combine mode, the firstEdge parameter is fixed value and only can modify the secondEdge variables



395 * which is the initial value in the channel (n+1) match edge when the FTM counter has been ran.



396 *



397 * Implements    : FTM_DRV_UpdatePwmChannel_Activity



398 *END**************************************************************************/



399status_t FTM_DRV_UpdatePwmChannel(uint32_t instance,



400  uint8_t channel,



401  ftm_pwm_update_option_t typeOfUpdate,



402  uint16_t firstEdge,



403  uint16_t secondEdge,



404  bool softwareTrigger)



405{



406  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);



407  DEV_ASSERT(channel < FEATURE_FTM_CHANNEL_COUNT);



408  uint16_t hwFirstEdge = 0U;



409  uint16_t hwSecondEdge = 0U;



410  uint16_t ftmPeriod = 0U;



411  uint8_t chnlPairNum = (uint8_t)(channel >> 1U);



412  FTM_Type * ftmBase = g_ftmBase[instance];



413  ftm_state_t * state = ftmStatePtr[instance];



414  status_t retStatus = STATUS_SUCCESS;



415



416  /* Get the newest period in the MOD register */



417  ftmPeriod = FTM_DRV_GetMod(ftmBase);



418



419  /* Check the mode operation in FTM module */



420  if (state->ftmMode == FTM_MODE_CEN_ALIGNED_PWM) {



421    ftmPeriod = (uint16_t)(ftmPeriod << 1U);



422  } else if (state->ftmMode == FTM_MODE_EDGE_ALIGNED_PWM) {



423    ftmPeriod = (uint16_t)(ftmPeriod + 1U);



424  } else {



425    retStatus = STATUS_ERROR;



426  }



427



428  /* Check the type of update for PWM */



429  if (FTM_PWM_UPDATE_IN_DUTY_CYCLE == typeOfUpdate) {



430    if ((firstEdge <= FTM_MAX_DUTY_CYCLE) && (secondEdge <= FTM_MAX_DUTY_CYCLE))



431    {



432      /* Calculate DutyCycle based of the previously calculated frequency*/



433      /* For greater resolution the DutyCycle values are in the range [0. FTM_MAX_DUTY_CYCLE]



434       *  where 0 = 0% or PWM always at Low and FTM_MAX_DUTY_CYCLE = 100% or PWM always HIGH;



435       *  a value of 0x4000 is equivalent of 50% DutyCycle. */



436      hwFirstEdge = (uint16_t)((ftmPeriod * firstEdge) >>



437        FTM_DUTY_TO_TICKS_SHIFT);



438      hwSecondEdge = (uint16_t)((ftmPeriod * secondEdge) >>



439        FTM_DUTY_TO_TICKS_SHIFT);



440



441      /* adjust DutyCycle if 100% value is to be achieved. */



442      if (FTM_MAX_DUTY_CYCLE == firstEdge) {



443        /* If expected duty is 100% then increase by 1 the value that is to be written



444         *  to Hardware so it will exceed value of period */



445        hwFirstEdge = (uint16_t)(hwFirstEdge + 1U);



446      }



447    } else {



448      retStatus = STATUS_ERROR;



449    }



450  } else {



451    if ((firstEdge <= ftmPeriod) && (secondEdge <= ftmPeriod)) {



452      hwFirstEdge = firstEdge;



453      hwSecondEdge = secondEdge;



454    } else {



455      retStatus = STATUS_ERROR;



456    }



457  }



458



459  if (STATUS_SUCCESS == retStatus) {



460    if (true == FTM_DRV_GetDualChnCombineCmd(ftmBase, chnlPairNum)) {



461      if (true == FTM_DRV_GetDualChnMofCombineCmd(ftmBase, chnlPairNum)) {



462        /* Check the clock source for FTM counter is disabled or not */



463        if (FTM_DRV_GetClockSource(ftmBase) == 0U) {



464          FTM_DRV_SetChnCountVal(ftmBase, (uint8_t)(chnlPairNum * 2U),



465            hwFirstEdge);



466        }



467      } else {



468        FTM_DRV_SetChnCountVal(ftmBase, (uint8_t)(chnlPairNum * 2U), hwFirstEdge);



469      }



470



471      /* Modify the initial value in the channel (n+1) match edge */



472      FTM_DRV_SetChnCountVal(ftmBase, (uint8_t)((chnlPairNum * 2U) + 1U),



473        hwSecondEdge);



474    } else {



475      /* Channel value is divided by 2 for up down counter mode to keep same duty */



476      if (true == FTM_DRV_GetCpwms(ftmBase)) {



477        FTM_DRV_SetChnCountVal(ftmBase, channel, (uint16_t)(hwFirstEdge >> 1U));



478      } else {



479        FTM_DRV_SetChnCountVal(ftmBase, channel, hwFirstEdge);



480      }



481    }



482



483    /* Software trigger is generated to change CnV registers */



484    /* Before this please configure sync mechanism to use software trigger */



485    FTM_DRV_SetSoftwareTriggerCmd(ftmBase, softwareTrigger);



486



487    /* Store the PWM period in the state structure */



488    state->ftmPeriod = ftmPeriod;



489  }



490



491  return retStatus;



492}

494/*FUNCTION**********************************************************************



495 *



496 * Function Name : FTM_DRV_UpdatePwmPeriod



497 * Description   : This function will update the new period in the frequency or



498 * in the counter value into mode register which modify the period of PWM signal



499 * on the channel output.



500 * - If the type of update in the duty cycle which is reused in FTM_DRV_UpdatePwmChannel



501 * function to convert the newValue parameters representing frequency in Hz to



502 * a period value to update the MOD register. The newValue parameter must be value



503 * between 1U and maximum is the frequency of the FTM counter.



504 * - If the type of update in ticks, this function will get value in counting to



505 * the MOD register. The newValue parameter must be value between 1U and 0xFFFFU



506 *



507 * Implements : FTM_DRV_UpdatePwmPeriod_Activity



508 *END**************************************************************************/



509status_t FTM_DRV_UpdatePwmPeriod(uint32_t instance,



510  ftm_pwm_update_option_t typeOfUpdate,



511  uint32_t newValue,



512  bool softwareTrigger)



513{



514  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);



515  DEV_ASSERT(newValue != 0U);



516  uint32_t ftmPeriod = 0U;



517  FTM_Type * ftmBase = g_ftmBase[instance];



518  ftm_state_t * state = ftmStatePtr[instance];



519  status_t retStatus = STATUS_SUCCESS;



520



521  /* Check the type of update for period in PWM mode */



522  if (FTM_PWM_UPDATE_IN_TICKS == typeOfUpdate) {



523    ftmPeriod = newValue;



524  } else {



525    if (newValue <= state->ftmSourceClockFrequency) {



526      ftmPeriod = (uint32_t)FTM_DRV_ConvertFreqToPeriodTicks(instance, newValue);



527    } else {



528      retStatus = STATUS_ERROR;



529    }



530  }



531



532  if (STATUS_SUCCESS == retStatus) {



533    /* Check the ftmPeriod is invalid */



534    DEV_ASSERT(ftmPeriod <= 0xFFFFU);



535



536    /* Check the signal operation in which PWM mode */



537    DEV_ASSERT((FTM_MODE_CEN_ALIGNED_PWM == state->ftmMode) ||



538               (FTM_MODE_EDGE_ALIGNED_PWM == state->ftmMode));



539    if (FTM_MODE_CEN_ALIGNED_PWM == state->ftmMode) {



540      ftmPeriod = (ftmPeriod >> 1U);



541    } else {



542      ftmPeriod = (ftmPeriod - 1U);



543    }



544



545    /* Set the new modulo value into MOD register */



546    FTM_DRV_SetMod(ftmBase, (uint16_t)(ftmPeriod));



547



548    /* Software trigger is generated to change MOD registers */



549    FTM_DRV_SetSoftwareTriggerCmd(ftmBase, softwareTrigger);



550



551    /* Store the PWM period in the state structure */

646/*FUNCTION**********************************************************************



647 *



648 * Function Name : FTM_PWM_DRV_IrqHandler



649 * Description   : This function is used to workaround an errata which the safe state



650 * is not removed from channel outputs after fault condition ends if SWOCTRL is being



651 * used to control the pin. The FTM_MODE[FAULTM] should be configured for manual fault



652 * clearing (0b10)



653 * This function must be used in the TOF interrupt handler when a fault is detected to



654 * ensure that the outputs return to the value configured by FTM_SWOCTR



655 *



656 * Implements    : FTM_PWM_DRV_IrqHandler_Activity



657 *END**************************************************************************/



658void FTM_PWM_DRV_IrqHandler(uint32_t instance,



659  uint32_t chnOutCtrlVal)



660{



661  FTM_Type * ftmBase = g_ftmBase[instance];



662



663  /* 1. Check the value of FTM_FMS[FAULTF] */



664  if (FTM_DRV_GetDetectedFaultInput(ftmBase) == true) {



665    faultDetection = true;



666



667    /* 2. Write the FTM_OUTMASK register to set the bit */



668    FTM_DRV_SetOutmaskReg(ftmBase, chnOutCtrlVal);



669



670    /* 3. Clear fault conditions by reading the FTM_FMS register and then writing FTM_FMS with all zeroes */



671    FTM_DRV_ClearFaultsIsr(ftmBase);



672



673    /* 4. Clear the FTM_SC[TOF] bit by reading the FTM_SC register, then writing a 0 to FTM_SC[TOF] */



674    FTM_DRV_ClearTimerOverflow(ftmBase);



675



676    /* 5. Exit the interrupt handler to skip following steps */



677  } else if (faultDetection == true) {



678    /* 6. Clear the FTM_SWOCTRL by writing all zeroes to it */



679    FTM_DRV_SetAllChnSoftwareCtrlCmd(ftmBase, 0x00U);



680    FTM_DRV_SetAllChnSoftwareCtrlVal(ftmBase, 0x00U);



681



682    /* 7. Write FTM_SWOCTRL with the desired value again */



683    uint8_t u8chnOutCtrlVal = (uint8_t)(chnOutCtrlVal & 0xFFu);



684    FTM_DRV_SetAllChnSoftwareCtrlCmd(ftmBase, u8chnOutCtrlVal);



685    FTM_DRV_SetAllChnSoftwareCtrlVal(ftmBase, u8chnOutCtrlVal);



686



687    /* 8. Clear the FTM_OUTMASK bits that were set in step 2 */



688    FTM_DRV_SetOutmaskReg(ftmBase, 0x00U);



689



690    /* 9. Clear the fault variable that was set in step 1 when the fault condition was originally detected */



691    faultDetection = false;



692



693    /* 10. Clear the FTM_SC[TOF] bit by reading the FTM_SC register, then writing a 0 to FTM_SC[TOF] */



694    FTM_DRV_ClearTimerOverflow(ftmBase);



695  } else {



696    /* Nothing to do */



697  }



698}
