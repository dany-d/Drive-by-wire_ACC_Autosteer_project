145/*FUNCTION**********************************************************************



146 *



147 * Function Name : LPUART_DRV_Init



148 * Description   : This function initializes a LPUART instance for operation.



149 * This function will initialize the run-time state structure to keep track of



150 * the on-going transfers, ungate the clock to the LPUART module, initialize the



151 * module to user defined settings and default settings, configure the IRQ state



152 * structure and enable the module-level interrupt to the core, and enable the



153 * LPUART module transmitter and receiver.



154 * The following is an example of how to set up the lpuart_state_t and the



155 * lpuart_user_config_t parameters and how to call the LPUART_DRV_Init function



156 * by passing in these parameters:



157 *    lpuart_user_config_t lpuartConfig;



158 *    lpuartConfig.baudRate = 9600;



159 *    lpuartConfig.bitCountPerChar = LPUART_8_BITS_PER_CHAR;



160 *    lpuartConfig.parityMode = LPUART_PARITY_DISABLED;



161 *    lpuartConfig.stopBitCount = LPUART_ONE_STOP_BIT;



162 *    lpuartConfig.transferType = LPUART_USING_INTERRUPTS;



163 *    lpuart_state_t lpuartState;



164 *    LPUART_DRV_Init(instance, &lpuartState, &lpuartConfig);



165 *



166 * Implements    : LPUART_DRV_Init_Activity



167 *END**************************************************************************/



168status_t LPUART_DRV_Init(uint32_t instance, lpuart_state_t * lpuartStatePtr,



169  const lpuart_user_config_t * lpuartUserConfig)



170{



171  DEV_ASSERT(instance < LPUART_INSTANCE_COUNT);



172  DEV_ASSERT(lpuartStatePtr != NULL);



173  DEV_ASSERT(lpuartUserConfig != NULL);



174  status_t lpuartStatus;



175  status_t osStatusRxSem;



176  status_t osStatusTxSem;



177  LPUART_Type * base = s_lpuartBase[instance];



178  uint32_t idx;



179  uint32_t lpuartSourceClock;



180  clock_names_t instanceClkName = s_lpuartClkNames[instance];



181



182  /* Get the LPUART clock as configured in the clock manager */



183  (void)CLOCK_SYS_GetFreq(instanceClkName, &lpuartSourceClock);



184



185  /* Check if current instance is clock gated off. */



186  DEV_ASSERT(lpuartSourceClock > 0U);



187



188  /* Check if current instance is already initialized. */



189  DEV_ASSERT(s_lpuartStatePtr[instance] == NULL);



190



191#if FEATURE_LPUART_HAS_DMA_ENABLE



192



193  /* In DMA mode, only 8-bits chars are supported */



194  DEV_ASSERT((lpuartUserConfig->transferType != LPUART_USING_DMA) ||



195             (lpuartUserConfig->bitCountPerChar == LPUART_8_BITS_PER_CHAR));



196



197#endif



198



199  /* For 10 bits per char, parity bit cannot be enabled */



200  DEV_ASSERT((lpuartUserConfig->bitCountPerChar != LPUART_10_BITS_PER_CHAR) ||



201             (lpuartUserConfig->parityMode == LPUART_PARITY_DISABLED));



202



203  /* Clear the state struct for this instance. */



204  uint8_t *clearStructPtr = (uint8_t *)lpuartStatePtr;



205  for (idx = 0; idx < sizeof(lpuart_state_t); idx++) {



206    clearStructPtr[idx] = 0;



207  }



208



209  /* Save runtime structure pointer.*/



210  s_lpuartStatePtr[instance] = lpuartStatePtr;



211



212  /* Save the transfer information for runtime retrieval */



213  lpuartStatePtr->transferType = lpuartUserConfig->transferType;



214  lpuartStatePtr->bitCountPerChar = lpuartUserConfig->bitCountPerChar;



215



216#if FEATURE_LPUART_HAS_DMA_ENABLE



217



218  lpuartStatePtr->rxDMAChannel = lpuartUserConfig->rxDMAChannel;



219  lpuartStatePtr->txDMAChannel = lpuartUserConfig->txDMAChannel;



220



221#endif



222



223  /* initialize the LPUART instance */



224  LPUART_Init(base);



225



226  /* initialize the parameters of the LPUART config structure with desired data */



227  lpuartStatus = LPUART_DRV_SetBaudRate(instance, lpuartUserConfig->baudRate);



228  if (lpuartStatus != STATUS_SUCCESS) {



229    return STATUS_ERROR;



230  }



231



232  if (lpuartUserConfig->parityMode != LPUART_PARITY_DISABLED) {



233    LPUART_SetBitCountPerChar(base, lpuartUserConfig->bitCountPerChar, true);



234  } else {



235    LPUART_SetBitCountPerChar(base, lpuartUserConfig->bitCountPerChar, false);



236  }



237



238  LPUART_SetParityMode(base, lpuartUserConfig->parityMode);



239  LPUART_SetStopBitCount(base, lpuartUserConfig->stopBitCount);



240



241  /* initialize last driver operation status */



242  lpuartStatePtr->transmitStatus = STATUS_SUCCESS;



243  lpuartStatePtr->receiveStatus = STATUS_SUCCESS;



244



245  /* Create the synchronization objects */



246  osStatusRxSem = OSIF_SemaCreate(&lpuartStatePtr->rxComplete, 0);



247  osStatusTxSem = OSIF_SemaCreate(&lpuartStatePtr->txComplete, 0);



248  if ((osStatusRxSem == STATUS_ERROR) || (osStatusTxSem == STATUS_ERROR)) {



249    return STATUS_ERROR;



250  }



251



252  /* Install LPUART irq handler */



253  INT_SYS_InstallHandler(s_lpuartRxTxIrqId[instance], g_lpuartIsr[instance],



254    (isr_t*) 0);



255



256  /* Enable LPUART interrupt. */



257  INT_SYS_EnableIRQ(s_lpuartRxTxIrqId[instance]);



258  return STATUS_SUCCESS;



259}

261/*FUNCTION**********************************************************************



262 *



263 * Function Name : LPUART_DRV_Deinit



264 * Description   : This function shuts down the UART by disabling interrupts and



265 *                 transmitter/receiver.



266 *



267 * Implements    : LPUART_DRV_Deinit_Activity



268 *END**************************************************************************/



269status_t LPUART_DRV_Deinit(uint32_t instance)



270{



271  DEV_ASSERT(instance < LPUART_INSTANCE_COUNT);



272  clock_names_t instanceClkName = s_lpuartClkNames[instance];



273  uint32_t lpuartSourceClock;



274  const LPUART_Type * base = s_lpuartBase[instance];



275  const lpuart_state_t * lpuartState = (lpuart_state_t *)



276    s_lpuartStatePtr[instance];



277  (void)CLOCK_SYS_GetFreq(instanceClkName, &lpuartSourceClock);



278



279  /* Check if current instance is already de-initialized or is gated.*/



280  DEV_ASSERT(s_lpuartStatePtr[instance] != NULL);



281  DEV_ASSERT(lpuartSourceClock > 0U);



282



283  /* Wait until the data is completely shifted out of shift register */



284  while (!LPUART_GetStatusFlag(base, LPUART_TX_COMPLETE)) {



285  }



286



287  /* Destroy the synchronization objects */



288  (void)OSIF_SemaDestroy(&lpuartState->rxComplete);



289  (void)OSIF_SemaDestroy(&lpuartState->txComplete);



290



291  /* Disable LPUART interrupt. */



292  INT_SYS_DisableIRQ(s_lpuartRxTxIrqId[instance]);



293



294  /* Restore default handler. */



295  INT_SYS_InstallHandler(s_lpuartRxTxIrqId[instance], DefaultISR, (isr_t*) 0);



296



297  /* Clear our saved pointer to the state structure */



298  s_lpuartStatePtr[instance] = NULL;



299  return STATUS_SUCCESS;



300}

341/*FUNCTION**********************************************************************



342 *



343 * Function Name : LPUART_DRV_SendDataBlocking



344 * Description   : This function sends data out through the LPUART module using



345 * blocking method. The function does not return until the transmit is complete.



346 *



347 * Implements    : LPUART_DRV_SendDataBlocking_Activity



348 *END**************************************************************************/



349status_t LPUART_DRV_SendDataBlocking(uint32_t instance,



350  const uint8_t * txBuff,



351  uint32_t txSize,



352  uint32_t timeout)



353{



354  DEV_ASSERT(instance < LPUART_INSTANCE_COUNT);



355  DEV_ASSERT(txBuff != NULL);



356  lpuart_state_t * lpuartState = (lpuart_state_t *)s_lpuartStatePtr[instance];



357  status_t retVal = STATUS_SUCCESS;



358  status_t syncStatus;



359



360  /* Indicates this is a blocking transaction. */



361  lpuartState->isTxBlocking = true;



362  DEV_ASSERT((lpuartState->transferType == LPUART_USING_INTERRUPTS) ||



363             (lpuartState->transferType == LPUART_USING_DMA));



364  if (lpuartState->transferType == LPUART_USING_INTERRUPTS) {



365    /* Start the transmission process using interrupts */



366    retVal = LPUART_DRV_StartSendDataUsingInt(instance, txBuff, txSize);



367  }



368



369#if FEATURE_LPUART_HAS_DMA_ENABLE



370



371  else



372  {



373    /* Start the transmission process using DMA */



374    retVal = LPUART_DRV_StartSendDataUsingDma(instance, txBuff, txSize);



375  }



376#endif



377



378  if (retVal == STATUS_SUCCESS) {



379    /* Wait until the transmit is complete. */



380    syncStatus = OSIF_SemaWait(&lpuartState->txComplete, timeout);



381



382    /* Finish the transmission if timeout expired */



383    if (syncStatus == STATUS_TIMEOUT) {



384      lpuartState->isTxBlocking = false;



385      lpuartState->transmitStatus = STATUS_TIMEOUT;



386      if (lpuartState->transferType == LPUART_USING_INTERRUPTS) {



387        LPUART_DRV_CompleteSendDataUsingInt(instance);



388      }



389



390#if FEATURE_LPUART_HAS_DMA_ENABLE



391



392      else



393      {



394        LPUART_DRV_StopTxDma(instance);



395      }



396#endif



397



398    }



399  }



400



401  return lpuartState->transmitStatus;



402}

404/*FUNCTION**********************************************************************



405 *



406 * Function Name : LPUART_DRV_SendDataPolling



407 * Description   : Send out multiple bytes of data using polling method.



408 *



409 * Implements    : LPUART_DRV_SendDataPolling_Activity



410 *END**************************************************************************/



411void LPUART_DRV_SendDataPolling(uint32_t instance,



412  const uint8_t *txBuff,



413  uint32_t txSize)



414{



415  DEV_ASSERT(instance < LPUART_INSTANCE_COUNT);



416  DEV_ASSERT(txBuff != NULL);



417  LPUART_Type * base = s_lpuartBase[instance];



418  lpuart_state_t * lpuartState = (lpuart_state_t *)s_lpuartStatePtr[instance];



419



420  /* Check the validity of the parameters */



421  DEV_ASSERT(txSize > 0U);



422  DEV_ASSERT((lpuartState->bitCountPerChar == LPUART_8_BITS_PER_CHAR) ||



423             ((txSize & 1U) == 0U));



424



425  /* Enable the LPUART transmitter */



426  LPUART_SetTransmitterCmd(base, true);



427  while (txSize > 0U) {



428    while (!LPUART_GetStatusFlag(base, LPUART_TX_DATA_REG_EMPTY)) {



429    }



430



431    lpuartState->txBuff = txBuff;



432    LPUART_DRV_PutData(instance);



433    if (lpuartState->bitCountPerChar == LPUART_8_BITS_PER_CHAR) {



434      ++txBuff;



435      --txSize;



436    } else {



437      ++txBuff;



438      ++txBuff;



439      txSize -= 2U;



440    }



441  }



442



443  /* Disable the LPUART transmitter */



444  LPUART_SetTransmitterCmd(base, false);



445}

486/*FUNCTION**********************************************************************



487 *



488 * Function Name : LPUART_DRV_GetTransmitStatus



489 * Description   : This function returns whether the previous LPUART transmit has



490 * finished. When performing non-blocking transmit, the user can call this



491 * function to ascertain the state of the current transmission:



492 * in progress (or busy) or complete (success). In addition, if the transmission



493 * is still in progress, the user can obtain the number of words that have been



494 * currently transferred.



495 *



496 * Implements    : LPUART_DRV_GetTransmitStatus_Activity



497 *END**************************************************************************/



498status_t LPUART_DRV_GetTransmitStatus(uint32_t instance, uint32_t *



499  bytesRemaining)



500{



501  DEV_ASSERT(instance < LPUART_INSTANCE_COUNT);



502  const lpuart_state_t * lpuartState = (lpuart_state_t *)



503    s_lpuartStatePtr[instance];



504  if (bytesRemaining != NULL) {



505    if (lpuartState->isTxBusy) {



506      /* Fill in the bytes not transferred yet. */



507      if (lpuartState->transferType == LPUART_USING_INTERRUPTS) {



508        /* In interrupt-based communication, the remaining bytes are retrieved



509         * from the state structure



510         */



511        *bytesRemaining = lpuartState->txSize;



512        ;



513      }



514



515#if FEATURE_LPUART_HAS_DMA_ENABLE



516



517      else



518      {



519        /* In DMA-based communication, the remaining bytes are retrieved



520         * from the current DMA major loop count



521         */



522        *bytesRemaining = EDMA_DRV_GetRemainingMajorIterationsCount



523          (lpuartState->txDMAChannel);



524      }



525#endif



526



527    } else {



528      *bytesRemaining = 0;



529    }



530  }



531



532  return lpuartState->transmitStatus;



533}

636/*FUNCTION**********************************************************************



637 *



638 * Function Name : LPUART_DRV_ReceiveDataPolling



639 * Description   : Receive multiple bytes of data using polling method.



640 *



641 * Implements    : LPUART_DRV_ReceiveDataPolling_Activity



642 *END**************************************************************************/



643status_t LPUART_DRV_ReceiveDataPolling(uint32_t instance,



644  uint8_t *rxBuff,



645  uint32_t rxSize)



646{



647  DEV_ASSERT(instance < LPUART_INSTANCE_COUNT);



648  DEV_ASSERT(rxBuff != NULL);



649  uint8_t tmpByte;



650  status_t retVal = STATUS_SUCCESS;



651  status_t tmpState = STATUS_SUCCESS;



652  LPUART_Type * base = s_lpuartBase[instance];



653  lpuart_state_t * lpuartState = (lpuart_state_t *)s_lpuartStatePtr[instance];



654



655  /* Check the validity of the parameters */



656  DEV_ASSERT(rxSize > 0U);



657  DEV_ASSERT((lpuartState->bitCountPerChar == LPUART_8_BITS_PER_CHAR) ||



658             ((rxSize & 1U) == 0U));



659



660  /* Enable the LPUART receiver */



661  LPUART_SetReceiverCmd((LPUART_Type *)base, true);



662  while (rxSize > 0U) {



663    while (!LPUART_GetStatusFlag(base, LPUART_RX_DATA_REG_FULL)) {



664    }



665



666    lpuartState->rxBuff = rxBuff;



667    LPUART_DRV_GetData(instance);



668    if (lpuartState->bitCountPerChar == LPUART_8_BITS_PER_CHAR) {



669      ++rxBuff;



670      --rxSize;



671    } else {



672      ++rxBuff;



673      ++rxBuff;



674      rxSize -= 2U;



675    }



676



677    /* Check for errors on received data */



678    if (LPUART_GetStatusFlag(base, LPUART_FRAME_ERR)) {



679      tmpState = STATUS_UART_FRAMING_ERROR;



680



681      /* Disable the LPUART receiver */



682      LPUART_SetReceiverCmd((LPUART_Type *)base, false);



683



684      /* Clear the flag */



685      (void)LPUART_ClearStatusFlag(base, LPUART_FRAME_ERR);



686      break;



687    } else if (LPUART_GetStatusFlag(base, LPUART_NOISE_DETECT)) {



688      tmpState = STATUS_UART_NOISE_ERROR;



689



690      /* Disable the LPUART receiver */



691      LPUART_SetReceiverCmd((LPUART_Type *)base, false);



692



693      /* Clear the flag */



694      (void)LPUART_ClearStatusFlag(base, LPUART_NOISE_DETECT);



695      break;



696    } else if (LPUART_GetStatusFlag(base, LPUART_PARITY_ERR)) {



697      tmpState = STATUS_UART_PARITY_ERROR;



698



699      /* Disable the LPUART receiver */



700      LPUART_SetReceiverCmd((LPUART_Type *)base, false);



701



702      /* Clear the flag */



703      (void)LPUART_ClearStatusFlag(base, LPUART_PARITY_ERR);



704      break;



705    } else if (LPUART_GetStatusFlag(base, LPUART_RX_OVERRUN)) {



706      tmpState = STATUS_UART_RX_OVERRUN;



707



708      /* Disable the LPUART receiver */



709      LPUART_SetReceiverCmd((LPUART_Type *)base, false);



710



711      /* Clear the flag */



712      (void)LPUART_ClearStatusFlag(base, LPUART_RX_OVERRUN);



713      break;



714    } else {



715      /* No error condition - avoid MISRA violation */



716    }



717  }



718



719  /* Update received status */



720  if ((rxSize == 0U) && (tmpState == STATUS_UART_RX_OVERRUN)) {



721    retVal = STATUS_SUCCESS;



722  } else {



723    retVal = tmpState;



724  }



725



726  if (retVal == STATUS_SUCCESS) {



727    /* Disable the LPUART receiver */



728    LPUART_SetReceiverCmd((LPUART_Type *)base, false);



729  }



730



731  /* Read dummy to clear RDRF flag */



732  LPUART_Getchar(base, &tmpByte);



733  return retVal;



734}

861/*FUNCTION**********************************************************************



862 *



863 * Function Name : LPUART_DRV_SetBaudRate



864 * Description   : Configures the LPUART baud rate.



865 * In some LPUART instances the user must disable the transmitter/receiver



866 * before calling this function.



867 * Generally, this may be applied to all LPUARTs to ensure safe operation.



868 *



869 * Implements    : LPUART_DRV_SetBaudRate_Activity



870 *END**************************************************************************/



871status_t LPUART_DRV_SetBaudRate(uint32_t instance, uint32_t desiredBaudRate)



872{



873  DEV_ASSERT(instance < LPUART_INSTANCE_COUNT);



874  uint16_t sbr, sbrTemp, i;



875  uint32_t osr, tempDiff, calculatedBaud, baudDiff;



876  uint32_t lpuartSourceClock;



877  clock_names_t instanceClkName = s_lpuartClkNames[instance];



878  LPUART_Type * base = s_lpuartBase[instance];



879  const lpuart_state_t * lpuartState;



880  lpuartState = (lpuart_state_t *)s_lpuartStatePtr[instance];



881  if (lpuartState != NULL) {



882    /* Check if there is an ongoing transfer */



883    if (lpuartState->isTxBusy == true) {



884      return STATUS_BUSY;



885    }



886



887    if (lpuartState->isRxBusy == true) {



888      return STATUS_BUSY;



889    }



890  }



891



892  /* Get the LPUART clock as configured in the clock manager */



893  (void)CLOCK_SYS_GetFreq(instanceClkName, &lpuartSourceClock);



894



895  /* Check if current instance is clock gated off. */



896  DEV_ASSERT(lpuartSourceClock > 0U);



897



898  /* Check if the desired baud rate can be configured with the current protocol clock. */



899  DEV_ASSERT(lpuartSourceClock >= (desiredBaudRate * 4U));



900



901  /* This lpuart instantiation uses a slightly different baud rate calculation



902   * The idea is to use the best OSR (over-sampling rate) possible



903   * Note, osr is typically hard-set to 16 in other lpuart instantiations



904   * First calculate the baud rate using the minimum OSR possible (4) */



905  osr = 4;



906  sbr = (uint16_t)(lpuartSourceClock / (desiredBaudRate * osr));



907  calculatedBaud = (lpuartSourceClock / (osr * sbr));



908  if (calculatedBaud > desiredBaudRate) {



909    baudDiff = calculatedBaud - desiredBaudRate;



910  } else {



911    baudDiff = desiredBaudRate - calculatedBaud;



912  }



913



914  /* loop to find the best osr value possible, one that generates minimum baudDiff



915   * iterate through the rest of the supported values of osr */



916  for (i = 5U; i <= 32U; i++) {



917    /* calculate the temporary sbr value   */



918    sbrTemp = (uint16_t)(lpuartSourceClock / (desiredBaudRate * i));



919



920    /* calculate the baud rate based on the temporary osr and sbr values */



921    calculatedBaud = (lpuartSourceClock / (i * sbrTemp));



922    if (calculatedBaud > desiredBaudRate) {



923      tempDiff = calculatedBaud - desiredBaudRate;



924    } else {



925      tempDiff = desiredBaudRate - calculatedBaud;



926    }



927



928    if (tempDiff <= baudDiff) {



929      baudDiff = tempDiff;



930      osr = i;                         /* update and store the best osr value calculated */



931      sbr = sbrTemp;                   /* update store the best sbr value calculated */



932    }



933  }



934



935  /* Check if osr is between 4x and 7x oversampling.



936   * If so, then "BOTHEDGE" sampling must be turned on */



937  if ((osr > 3U) && (osr < 8U)) {



938    LPUART_SetBothEdgeSamplingCmd(base, true);



939  }



940



941  /* program the osr value (bit value is one less than actual value) */



942  LPUART_SetOversamplingRatio(base, (osr - 1U));



943



944  /* write the sbr value to the BAUD registers */



945  LPUART_SetBaudRateDivisor(base, sbr);



946  return STATUS_SUCCESS;



947}
