103/*FUNCTION**********************************************************************



104 *



105 * Function Name : FTM_DRV_InitInputCapture



106 * Description   : Configures Channel Input Capture for either getting time-stamps on edge detection



107 * or on signal measurement . When the edge specified in the captureMode



108 * argument occurs on the channel the FTM counter is captured into the CnV register.



109 * The user will have to read the CnV register separately to get this value. The filter



110 * function is disabled if the filterVal argument passed in is 0. The filter function



111 * is available only on channels 0,1,2,3.



112 *



113 * Implements    : FTM_DRV_InitInputCapture_Activity



114 *END**************************************************************************/



115status_t FTM_DRV_InitInputCapture(uint32_t instance,



116  const ftm_input_param_t * param)



117{



118  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);



119  DEV_ASSERT(param != NULL);



120  FTM_Type * ftmBase = g_ftmBase[instance];



121  uint8_t chnlPairNum = 0U;



122  uint8_t index = 0U;



123  uint8_t hwChannel = 0U;



124  ftm_signal_measurement_mode_t measurementType;



125  ftm_state_t * state = ftmStatePtr[instance];



126  status_t retStatus = STATUS_SUCCESS;



127  if ((NULL != state) && (FTM_MODE_NOT_INITIALIZED == state->ftmMode)) {



128    FTM_DRV_SetClockSource(ftmBase, FTM_CLOCK_SOURCE_NONE);



129    FTM_DRV_SetCounterInitVal(ftmBase, 0U);



130    FTM_DRV_SetMod(ftmBase, param->nMaxCountValue);



131    FTM_DRV_SetCpwms(ftmBase, false);



132



133    /* Disable the combine mode */



134    FTM_DRV_SetDualChnMofCombineCmd(ftmBase, chnlPairNum, false);



135    FTM_DRV_SetDualChnCombineCmd(ftmBase, chnlPairNum, false);



136    for (index = 0U; index < param->nNumChannels; index++) {



137      hwChannel = param->inputChConfig[index].hwChannelId;



138      chnlPairNum = (uint8_t)(hwChannel >> 1U);



139



140      /* Save in state structure user define handlers */



141      state->channelsCallbacksParams[hwChannel] = param->inputChConfig[index].



142        channelsCallbacksParams;



143      state->channelsCallbacks[hwChannel] = param->inputChConfig[index].



144        channelsCallbacks;



145      state->enableNotification[hwChannel] = true;



146



147      /* Enable filtering for input channels */



148      if (hwChannel < CHAN4_IDX) {



149        if (true == param->inputChConfig[index].filterEn) {



150          FTM_DRV_SetChnInputCaptureFilter(ftmBase, hwChannel, (uint8_t)



151            param->inputChConfig[index].filterValue);



152        } else {



153          FTM_DRV_SetChnInputCaptureFilter(ftmBase, hwChannel, 0U);



154        }



155      }



156



157      if (FTM_EDGE_DETECT == param->inputChConfig[index].inputMode) {



158        /* Disable the dual edge mode */



159        FTM_DRV_SetDualEdgeCaptureCmd(ftmBase, chnlPairNum, false);



160



161        /* Set input capture mode */



162        FTM_DRV_SetChnMSnBAMode(ftmBase, hwChannel, 0U);



163



164        /* Check if no edge is selected */



165        DEV_ASSERT (FTM_NO_PIN_CONTROL != param->inputChConfig[index].



166                    edgeAlignement);



167



168        /* Set the event which will generate the interrupt */



169        FTM_DRV_SetChnEdgeLevel(ftmBase, hwChannel, (uint8_t)



170          param->inputChConfig[index].edgeAlignement);



171



172        /* Enable interrupt request for the current channel */



173        FTM_DRV_EnableChnInt(ftmBase, hwChannel);



174        INT_SYS_EnableIRQ(g_ftmIrqId[instance][hwChannel]);



175      } else if (FTM_SIGNAL_MEASUREMENT == param->inputChConfig[index].inputMode)



176      {



177        /* Check the channel is even number */



178        DEV_ASSERT((hwChannel % 2U) == 0U);



179



180        /* Enable the dual edge mode */



181        FTM_DRV_SetDualEdgeCaptureCmd(ftmBase, chnlPairNum, true);



182



183        /* Enable dual edge input capture */



184        FTM_DRV_SetDualChnDecapCmd(ftmBase, chnlPairNum, true);



185



186        /* If continuous mode is set*/



187        if (true == param->inputChConfig[index].continuousModeEn) {



188          /* Set MSnA and MSnB bit*/



189          FTM_DRV_SetChnMSnBAMode(ftmBase, hwChannel, 3U);



190        } else {



191          /* Clear MSnA and Set MSnB bit*/



192          FTM_DRV_SetChnMSnBAMode(ftmBase, hwChannel, 2U);



193        }



194



195        measurementType = param->inputChConfig[index].measurementType;



196



197        /* Check If want to measure a pulse width or period of the signal */



198        if ((FTM_PERIOD_ON_MEASUREMENT == measurementType) ||



199            (FTM_RISING_EDGE_PERIOD_MEASUREMENT== measurementType)) {



200          FTM_DRV_SetChnEdgeLevel(ftmBase, hwChannel, 1U);



201          if (FTM_PERIOD_ON_MEASUREMENT == measurementType) {



202            /* Measure time between rising and falling edge - positive duty */



203            FTM_DRV_SetChnEdgeLevel(ftmBase, (uint8_t)(hwChannel + 1U), 2U);



204          } else {



205            /* If channel (n) is configured to capture falling edges (ELS(n)B:ELS(n)A = 0:1)



206             * then channel (n+1) also captures falling edges (ELS(n+1)B:ELS(n+1)A = 0:1) */



207            FTM_DRV_SetChnEdgeLevel(ftmBase, (uint8_t)(hwChannel + 1U), 1U);



208          }



209        } else if ((FTM_PERIOD_OFF_MEASUREMENT == measurementType) ||



210                   (FTM_FALLING_EDGE_PERIOD_MEASUREMENT == measurementType)) {



211          FTM_DRV_SetChnEdgeLevel(ftmBase, hwChannel, 2U);



212          if (FTM_PERIOD_OFF_MEASUREMENT == measurementType) {



213            /* Measure time between falling and rising edge - negative duty */



214            FTM_DRV_SetChnEdgeLevel(ftmBase, (uint8_t)(hwChannel + 1U), 1U);



215          } else {



216            /* If channel (n) is configured to capture rising edges (ELS(n)B:ELS(n)A = 1:0) than



217             * channel (n+1) is setup to capture also raising edges (ELS(n+1)B:ELS(n+1)A = 1:0) */



218            FTM_DRV_SetChnEdgeLevel(ftmBase, (uint8_t)(hwChannel + 1U), 2U);



219          }



220        } else {



221          retStatus = STATUS_ERROR;



222          break;



223        }



224



225        /* Enable the interrupt request for the channel which will indicate that the measurement is done. */



226        FTM_DRV_EnableChnInt(ftmBase, (uint8_t)(hwChannel + 1U));



227        INT_SYS_EnableIRQ(g_ftmIrqId[instance][hwChannel]);



228      } else {



229        /* Do nothing */



230      }



231    }



232



233    if (STATUS_SUCCESS == retStatus) {



234      state->ftmMode = FTM_MODE_INPUT_CAPTURE;



235



236      /* Set clock source to start the counter */



237      FTM_DRV_SetClockSource(ftmBase, state->ftmClockSource);



238    }



239  } else {



240    retStatus = STATUS_ERROR;



241  }



242



243  return retStatus;



244}

246/*FUNCTION**********************************************************************



247 *



248 * Function Name : FTM_DRV_DeinitInputCapture



249 * Description   : Disables Channel Input Capture



250 *



251 * Implements    : FTM_DRV_DeinitInputCapture_Activity



252 *END**************************************************************************/



253status_t FTM_DRV_DeinitInputCapture(uint32_t instance,



254  const ftm_input_param_t * param)



255{



256  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);



257  DEV_ASSERT(param != NULL);



258  FTM_Type * ftmBase = g_ftmBase[instance];



259  uint8_t chnlPairNum = 0U;



260  uint8_t index = 0U;



261  uint8_t hwChannel = 0U;



262  ftm_state_t * state = ftmStatePtr[instance];



263



264  /* FTM counter is disabled */



265  FTM_DRV_SetClockSource(ftmBase, FTM_CLOCK_SOURCE_NONE);



266  FTM_DRV_SetCounterInitVal(ftmBase, 0U);



267  FTM_DRV_SetMod(ftmBase, 0xFFFFU);



268  FTM_DRV_SetCpwms(ftmBase, false);



269  for (index = 0U; index < param->nNumChannels; index++) {



270    hwChannel = param->inputChConfig[index].hwChannelId;



271    chnlPairNum = (uint8_t)(hwChannel >> 1U);



272



273    /* Disable filtering for input channels */



274    if (hwChannel < CHAN4_IDX) {



275      FTM_DRV_SetChnInputCaptureFilter(ftmBase, hwChannel, 0U);



276    }



277



278    FTM_DRV_SetDualChnCombineCmd(ftmBase, chnlPairNum, false);



279    FTM_DRV_SetDualEdgeCaptureCmd(ftmBase, chnlPairNum, false);



280    FTM_DRV_SetChnEdgeLevel(ftmBase, hwChannel, (uint8_t)0U);



281    FTM_DRV_DisableChnInt(ftmBase, hwChannel);



282  }



283



284  /* Clear Callbacks function from the state structure */



285  for (index = 0U; index < FEATURE_FTM_CHANNEL_COUNT; index++) {



286    state->channelsCallbacksParams[index] = NULL;



287    state->channelsCallbacks[index] = NULL;



288  }



289



290  state->ftmMode = FTM_MODE_NOT_INITIALIZED;



291  return STATUS_SUCCESS;



292}

294/*FUNCTION**********************************************************************



295 *



296 * Function Name : FTM_DRV_GetInputCaptureMeasurement



297 * Description   : This function is used to calculate the measurement and/or time stamps values



298 * which are read from the C(n, n+1)V registers and stored to the static buffers.



299 *



300 * Implements    : FTM_DRV_GetInputCaptureMeasurement_Activity



301 *END**************************************************************************/



302uint16_t FTM_DRV_GetInputCaptureMeasurement(uint32_t instance,



303  uint8_t channel)



304{



305  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);



306  DEV_ASSERT(channel < FEATURE_FTM_CHANNEL_COUNT);



307  const ftm_state_t * state = ftmStatePtr[instance];



308  uint16_t value = 0U;



309  if (state != NULL) {



310    value = state->measurementResults[channel];



311  }



312



313  return value;



314}

316/*FUNCTION**********************************************************************



317 *



318 * Function Name : FTM_DRV_StartNewSignalMeasurement



319 * Description   : This function starts new Signal Measurements on a dual input compare channel



320 * that is configured as single-shot measurement.



321 *



322 * Implements    : FTM_DRV_StartNewSignalMeasurement_Activity



323 *END**************************************************************************/



324status_t FTM_DRV_StartNewSignalMeasurement(uint32_t instance,



325  uint8_t channel)



326{



327  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);



328  DEV_ASSERT(channel < FEATURE_FTM_CHANNEL_COUNT);



329



330  /* Clear CH(n)F and CH(n+1)F  flags and Set DECAP bit */



331  FTM_Type * ftmBase = g_ftmBase[instance];



332  uint8_t chnlPairNum = (uint8_t)(channel >> 1U);



333



334  /* Get channel mode */



335  if (FTM_FEATURE_INPUT_CAPTURE_SINGLE_SHOT == FTM_DRV_GetChnMode(ftmBase,



336       channel)) {



337    if (FTM_DRV_GetDualEdgeCaptureBit(ftmBase, chnlPairNum)) {



338      /* Clear event flags for channel n and n + 1 */



339      FTM_DRV_ClearChnEventFlag(ftmBase, (uint8_t)(channel + 1U));



340      FTM_DRV_ClearChnEventFlag(ftmBase, channel);



341



342      /* Set DECAP bit to start measurement */



343      FTM_DRV_SetDualChnDecapCmd(ftmBase, chnlPairNum, true);



344    }



345  } else {



346    /* Nothing to do */



347  }



348



349  return STATUS_SUCCESS;



350}

352/*FUNCTION**********************************************************************



353 *



354 * Function Name : FTM_IC_DRV_SetChannelMode



355 * Description   : This function is used to change the channel mode at run time or



356 * when stopping channel. The channel mode is selected in the ftm_ic_op_mode_t



357 * enumeration type.



358 *



359 * Implements    : FTM_IC_DRV_SetChannelMode_Activity



360 *END**************************************************************************/



361status_t FTM_IC_DRV_SetChannelMode(uint32_t instance,



362  uint8_t channel,



363  ftm_ic_op_mode_t inputMode,



364  bool enableContinuousCapture)



365{



366  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);



367  DEV_ASSERT(channel < FEATURE_FTM_CHANNEL_COUNT);



368  FTM_Type * ftmBase = g_ftmBase[instance];



369  uint8_t level = 1U;



370  uint8_t nextChnlevel = 1U;



371  uint8_t channelPair = (uint8_t)(channel >> 1U);



372  if (inputMode == FTM_DISABLE_OPERATION) {



373    /* Set the edge level to disable operation on the channel input */



374    FTM_DRV_SetChnEdgeLevel(ftmBase, channel, 0x0U);



375  } else if ((inputMode == FTM_TIMESTAMP_RISING_EDGE) ||



376             (inputMode == FTM_TIMESTAMP_FALLING_EDGE) ||



377             (inputMode == FTM_TIMESTAMP_BOTH_EDGES)) {



378    /* Disable the dual edge mode */



379    FTM_DRV_SetDualEdgeCaptureCmd(ftmBase, channelPair, false);



380



381    /* Set input capture mode */



382    FTM_DRV_SetChnMSnBAMode(ftmBase, channel, 0U);



383



384    /* Set the event which will generate the interrupt */



385    FTM_DRV_SetChnEdgeLevel(ftmBase, channel, (uint8_t)inputMode);



386



387    /* Enable interrupt request for the current channel */



388    FTM_DRV_EnableChnInt(ftmBase, channel);



389  } else {



390    /* Check the channel is even number */



391    DEV_ASSERT((channel % 2U) == 0U);



392



393    /* Enable the dual edge mode */



394    FTM_DRV_SetDualEdgeCaptureCmd(ftmBase, channelPair, true);



395



396    /* Enable dual edge input capture */



397    FTM_DRV_SetDualChnDecapCmd(ftmBase, channelPair, true);



398



399    /* If continuous mode is set*/



400    if (true == enableContinuousCapture) {



401      /* Set MSnA and MSnB bit*/



402      FTM_DRV_SetChnMSnBAMode(ftmBase, channel, 3U);



403    } else {



404      /* Clear MSnA and Set MSnB bit*/



405      FTM_DRV_SetChnMSnBAMode(ftmBase, channel, 2U);



406    }



407



408    /* Check If want to measure a pulse width or period of the signal */



409    if ((FTM_MEASURE_PULSE_HIGH == inputMode) || (FTM_MEASURE_RISING_EDGE_PERIOD



410         == inputMode)) {



411      if (FTM_MEASURE_PULSE_HIGH == inputMode) {



412        /* Measure time between rising and falling edge - positive duty */



413        nextChnlevel = 2U;



414      }



415    } else {



416      level = 2U;



417      if (FTM_MEASURE_FALLING_EDGE_PERIOD == inputMode) {



418        /* If channel (n) is configured to capture rising edges (ELS(n)B:ELS(n)A = 1:0) than



419         * channel (n+1) is setup to capture also raising edges (ELS(n+1)B:ELS(n+1)A = 1:0) */



420        nextChnlevel = 2U;



421      }



422    }



423



424    FTM_DRV_SetChnEdgeLevel(ftmBase, channel, level);



425    FTM_DRV_SetChnEdgeLevel(ftmBase, (uint8_t)(channel + 1U), nextChnlevel);



426



427    /* Disable interrupt request for the current channel */



428    FTM_DRV_DisableChnInt(ftmBase, channel);



429    FTM_DRV_EnableChnInt(ftmBase, (uint8_t)(channel + 1U));



430  }



431



432  return STATUS_SUCCESS;



433}

437/* Implementation of FTM0_Ch0_Ch7_IRQHandler master handler named in startup code. */



438void FTM0_Ch0_7_IRQHandler(void)



439{



440  uint32_t statusFlag = FTM_DRV_GetStatusFlags(0U);



441  uint32_t enabledInt = FTM_DRV_GetEnabledInterrupts(0U);



442  uint32_t index = 0U;



443  uint32_t shift = 0U;



444  bool isIntOnChn = false;



445  bool isIntOnNextChn = false;



446  for (index = 0U; index < (uint32_t)(FEATURE_FTM_CHANNEL_COUNT / 2U); index++)



447  {



448    shift = index << 1U;



449    isIntOnChn = (((statusFlag & enabledInt) & (0x1UL << shift)) != 0U) ? true :



450      false;



451    shift++;



452    isIntOnNextChn = (((statusFlag & enabledInt) & (0x1UL << shift)) != 0U) ?



453      true : false;



454    if ((isIntOnChn == true) || (isIntOnNextChn == true)) {



455      FTM_DRV_IrqHandler(0U, (uint8_t)index);



456    }



457  }



458}

692static void FTM_DRV_IrqHandler(uint32_t instance,



693  uint8_t channelPair)



694{



695  const ftm_state_t * state = ftmStatePtr[instance];



696  switch (state->ftmMode)



697  {



698   case FTM_MODE_INPUT_CAPTURE:



699    FTM_DRV_InputCaptureHandler(instance, channelPair);



700    break;



701



702   default:



703    /* Nothing to do */



704    break;



705  }



706}



707



708static void FTM_DRV_InputCaptureHandler(uint32_t instance,



709  uint8_t channelPair)



710{



711  DEV_ASSERT(instance < FTM_INSTANCE_COUNT);



712  DEV_ASSERT(channelPair < (FEATURE_FTM_CHANNEL_COUNT >> 1U));



713  ftm_state_t * state = ftmStatePtr[instance];



714  FTM_Type * ftmBase = g_ftmBase[instance];



715  uint8_t channel = (uint8_t)(channelPair << 1U);



716



717  /* Verify the mode for current pair of channels */



718  if (FTM_DRV_GetDualEdgeCaptureBit(ftmBase, channelPair)) {



719    /* Dual edge input capture case */



720    uint16_t first_event_time = FTM_DRV_GetChnCountVal(ftmBase, channel);



721    uint16_t second_event_time = FTM_DRV_GetChnCountVal(ftmBase, (uint8_t)



722      (channel + 1U));



723    if (second_event_time < first_event_time) {



724      /* Measurement when overflow occurred */



725      state->measurementResults[channel] = (uint16_t)(second_event_time +



726        (FTM_DRV_GetMod(ftmBase) - first_event_time));



727    } else {



728      /* Measurement when overflow doesn't occurred */



729      state->measurementResults[channel] = (uint16_t)(second_event_time -



730        first_event_time);



731    }



732



733    /* Clear flags for channels n and n+1 */



734    FTM_DRV_ClearChnEventFlag(ftmBase, channel);



735    FTM_DRV_ClearChnEventFlag(ftmBase, (uint8_t)(channel + 1U));



736  } else {



737    /* To get the channel interrupt source the both channels flag must be checked */



738    if (false == FTM_DRV_HasChnEventOccurred(ftmBase, channel)) {



739      channel++;



740    }



741



742    /* Get the time stamp of the event */



743    state->measurementResults[channel] = FTM_DRV_GetChnCountVal(ftmBase, channel);



744



745    /* Clear the flag for C(n+1) channel */



746    FTM_DRV_ClearChnEventFlag(ftmBase, channel);



747  }



748



749  /* If the callback is defined to use it */



750  if (((state->channelsCallbacks[channel]) != NULL) &&



751      (state->enableNotification[channel] == true)) {



752    state->channelsCallbacks[channel](IC_EVENT_MEASUREMENT_COMPLETE,



753      state->channelsCallbacksParams[channel]);



754  }



755}
