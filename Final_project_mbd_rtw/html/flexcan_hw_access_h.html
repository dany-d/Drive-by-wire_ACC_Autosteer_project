47/*! @brief FlexCAN message buffer CODE for Rx buffers*/



48enum {



49  FLEXCAN_RX_INACTIVE = 0x0,           /*!< MB is not active.*/



50  FLEXCAN_RX_FULL = 0x2,               /*!< MB is full.*/



51  FLEXCAN_RX_EMPTY = 0x4,              /*!< MB is active and empty.*/



52  FLEXCAN_RX_OVERRUN = 0x6,            /*!< MB is overwritten into a full buffer.*/



53  FLEXCAN_RX_BUSY = 0x8,               /*!< FlexCAN is updating the contents of the MB.*/



54



55  /*!  The CPU must not access the MB.*/



56  FLEXCAN_RX_RANSWER = 0xA,            /*!< A frame was configured to recognize a Remote Request Frame*/



57



58  /*!  and transmit a Response Frame in return.*/



59  FLEXCAN_RX_NOT_USED = 0xF            /*!< Not used*/



60};

62/*! @brief FlexCAN message buffer CODE FOR Tx buffers*/



63enum {



64  FLEXCAN_TX_INACTIVE = 0x08,          /*!< MB is not active.*/



65  FLEXCAN_TX_ABORT = 0x09,             /*!< MB is aborted.*/



66  FLEXCAN_TX_DATA = 0x0C,              /*!< MB is a TX Data Frame(MB RTR must be 0).*/



67  FLEXCAN_TX_REMOTE = 0x1C,            /*!< MB is a TX Remote Request Frame (MB RTR must be 1).*/



68  FLEXCAN_TX_TANSWER = 0x0E,           /*!< MB is a TX Response Request Frame from.*/



69



70  /*!  an incoming Remote Request Frame.*/

72};

549  /*!



550   * @brief Return Rx FIFO Id Format.



551   *



552   * @param   base     The FlexCAN base address



553   * @return  RxFifo Id Format



554   */



555  static inline flexcan_rx_fifo_id_element_format_t FLEXCAN_GetRxFifoIdFormat(



556    const CAN_Type * base)



557  {



558    flexcan_rx_fifo_id_element_format_t id_type= FLEXCAN_RX_FIFO_ID_FORMAT_A;



559    switch ((base->MCR & CAN_MCR_IDAM_MASK) >> CAN_MCR_IDAM_SHIFT)



560    {



561     case 0U:



562      id_type= FLEXCAN_RX_FIFO_ID_FORMAT_A;



563      break;



564



565     case 1U:



566      id_type= FLEXCAN_RX_FIFO_ID_FORMAT_B;



567      break;



568



569     case 2U:



570      id_type= FLEXCAN_RX_FIFO_ID_FORMAT_C;



571      break;



572



573     case 3U:



574      id_type= FLEXCAN_RX_FIFO_ID_FORMAT_D;



575      break;



576



577     default:



578      {



579        /* This case should never be reached */



580        DEV_ASSERT(false);



581        break;



582      }



583    }



584



585    return id_type;



586  }
