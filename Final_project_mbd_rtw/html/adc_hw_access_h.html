148  /*!



149   * @brief Sets the Sample time in AD clock cycles



150   *



151   * This function configures the sample time for the ADC (in



152   * ADCK clocks). The actual sample time will be the value



153   * provided plus 1.  Selection of 2 to 256 ADCK is possible.



154   * A real sample time of 1 is not supported (a parameter value of 0



155   * will be automatically be changed to 1).



156   *



157   * @param[in] baseAddr adc base pointer



158   * @param[in] sampletime Sample time in AD Clocks



159   */



160  static inline void ADC_SetSampleTime(ADC_Type * const baseAddr,



161    uint8_t sampletime)



162  {



163    /* Clip sample time to minimum value */



164    uint8_t rsampletime = (uint8_t)((sampletime > 0U) ? sampletime : 1U);



165    uint32_t tmp = baseAddr->CFG2;



166    tmp &= ~(ADC_CFG2_SMPLTS_MASK);



167    tmp |= ADC_CFG2_SMPLTS(rsampletime);



168    baseAddr->CFG2 = tmp;



169  }

1032   * measured data before being written in the result register.

1038    const uint16_t value)
