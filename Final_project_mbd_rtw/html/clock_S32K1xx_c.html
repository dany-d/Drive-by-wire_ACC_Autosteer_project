559/*FUNCTION**********************************************************************



560 *



561 * Function Name : CLOCK_SYS_SetConfiguration



562 * Description   : This function sets the system to target configuration, it



563 * only sets the clock modules registers for clock mode change, but not send



564 * notifications to drivers.



565 *



566 * Implements CLOCK_SYS_SetConfiguration_Activity



567 * END**************************************************************************/



568status_t CLOCK_SYS_SetConfiguration(clock_manager_user_config_t const * config)



569{



570  status_t result;



571  clock_manager_user_config_t config_default;



572  clock_manager_user_config_t const * cfg = config;



573  DEV_ASSERT(CLOCK_SYS_GetCurrentRunMode() == SCG_SYSTEM_CLOCK_MODE_RUN);



574  if (config == NULL) {



575    /* Get default configuration */



576    CLOCK_SYS_GetDefaultConfiguration(&config_default);



577    cfg = &config_default;



578  }



579



580  /* Set SCG settings. */



581  result = CLOCK_SYS_SetScgConfiguration(&cfg->scgConfig);



582  if (STATUS_SUCCESS == result) {



583    /* Set PCC settings. */



584    CLOCK_SYS_SetPccConfiguration(&cfg->pccConfig);



585



586    /* Set SIM settings. */



587    CLOCK_SYS_SetSimConfiguration(&cfg->simConfig);



588



589    /* Set PMC settings. */



590    CLOCK_SYS_SetPmcConfiguration(&cfg->pmcConfig);



591  }



592



593  return result;



594}

596/*FUNCTION**********************************************************************



597 *



598 * Function Name : CLOCK_SYS_SetScgConfiguration



599 * Description   : This function configures the SCG blocks



600 *



601 * END**************************************************************************/



602static status_t CLOCK_SYS_SetScgConfiguration(const scg_config_t * scgConfig)



603{



604  status_t status = STATUS_SUCCESS;



605  DEV_ASSERT(scgConfig != NULL);



606  if (scgConfig != NULL) {



607    /* Configure a temporary system clock source: FIRC */



608    status = CLOCK_SYS_ConfigureTemporarySystemClock();



609    if (status == STATUS_SUCCESS) {



610      /* Configure clock sources from SCG */



611      status = CLOCK_SYS_ConfigureModulesFromScg(scgConfig);



612    }



613



614    if (status == STATUS_SUCCESS) {



615      /* Configure RTC. */



616      if (scgConfig->rtcConfig.initialize) {



617        /* RTC Clock settings. */



618        g_RtcClkInFreq = scgConfig->rtcConfig.rtcClkInFreq;



619      }



620



621      /* Configure SCG ClockOut. */



622      if (scgConfig->clockOutConfig.initialize) {



623        /* ClockOut settings. */



624        SCG_SetClockoutSourceSel(SCG, (uint32_t)scgConfig->clockOutConfig.source);



625      }



626



627      /* Configure SCG clock modes. */



628      if (scgConfig->clockModeConfig.initialize) {



629        /* Configure SCG clock modes */
