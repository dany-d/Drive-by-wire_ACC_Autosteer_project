667   * it has no effect.

696      base->TCD[channel].NBYTES.MLOFFYES = regValTemp;



697    }



698



699#else



700



701    if (((base->CR >> DMA_CR_EMLM_SHIFT) & 1U) != 0U) {



702      uint32_t regValTemp;



703      regValTemp = base->TCD[channel].NBYTES.MLOFFYES;



704      regValTemp &= ~(DMA_TCD_NBYTES_MLOFFYES_SMLOE_MASK);



705      regValTemp |= DMA_TCD_NBYTES_MLOFFYES_SMLOE(enable ? 1UL : 0UL);
