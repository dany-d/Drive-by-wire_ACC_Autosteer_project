113/*!



114 * @brief Sets SCG run system clock



115 *



116 * This function sets the RUN clock control (system clock source, bus, core and slow dividers).



117 *



118 * @param[in] base Register base address for the SCG instance.



119 * @param[in] base System clock configuration address.



120 */



121static inline void SCG_SetRunClockControl(SCG_Type * base, uint32_t source, uint32_t coreDivider, uint32_t busDivider, uint32_t slowDivider)



122{



123    uint32_t value = (uint32_t)(((source      << SCG_RCCR_SCS_SHIFT)     & SCG_RCCR_SCS_MASK)     |



124                                ((coreDivider << SCG_RCCR_DIVCORE_SHIFT) & SCG_RCCR_DIVCORE_MASK) |



125                                ((busDivider  << SCG_RCCR_DIVBUS_SHIFT)  & SCG_RCCR_DIVBUS_MASK)  |



126                                ((slowDivider << SCG_RCCR_DIVSLOW_SHIFT) & SCG_RCCR_DIVSLOW_MASK) );



127    base->RCCR = value;



128}
