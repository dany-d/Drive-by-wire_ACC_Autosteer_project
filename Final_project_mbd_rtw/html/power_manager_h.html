61/*!



62 * @brief Power manager policies.



63 *



64 * Defines whether the mode switch initiated by the POWER_SYS_SetMode() is agreed upon



65 * (depending on the result of notification callbacks), or forced.



66 * For POWER_MANAGER_POLICY_FORCIBLE the power mode is changed



67 * regardless of the callback results, while for POWER_MANAGER_POLICY_AGREEMENT policy



68 * any error code returned by one of the callbacks aborts the mode change.



69 * See also POWER_SYS_SetMode() description.



70 * Implements power_manager_policy_t_Class



71 */



72typedef enum



73{



74    POWER_MANAGER_POLICY_AGREEMENT,      /*!< Power mode is changed if all of the callbacks return success. */



75    POWER_MANAGER_POLICY_FORCIBLE        /*!< Power mode is changed regardless of the result of callbacks. */



76} power_manager_policy_t;



77



78/*!



79 * @brief The PM notification type. Used to notify registered callbacks.



80 * Callback notifications can be invoked in following situations:



81 *  - before a power mode change (Callback return value can affect POWER_SYS_SetMode()



82 *    execution. Refer to the  POWER_SYS_SetMode() and power_manager_policy_t documentation).



83 *  - after a successful change of the power mode.



84 *  - after an unsuccessful attempt to switch power mode, in order to recover to a working state.



85 * Implements power_manager_notify_t_Class



86 */



87typedef enum



88{



89    POWER_MANAGER_NOTIFY_RECOVER = 0x00U,  /*!< Notify IP to recover to previous work state.      */



90    POWER_MANAGER_NOTIFY_BEFORE  = 0x01U,  /*!< Notify IP that the system will change the power setting.  */



91    POWER_MANAGER_NOTIFY_AFTER   = 0x02U   /*!< Notify IP that the system has changed to a new power setting. */



92} power_manager_notify_t;



93



94/*!



95 * @brief The callback type indicates when a callback will be invoked.



96 *



97 * Used in the callback configuration structures (power_manager_callback_user_config_t)



98 * to specify when the registered callback will be called during power mode change initiated by



99 * POWER_SYS_SetMode().



100 *



101 * Implements power_manager_callback_type_t_Class



102 */



103typedef enum



104{



105    POWER_MANAGER_CALLBACK_BEFORE       = 0x01U, /*!< Before callback. */



106    POWER_MANAGER_CALLBACK_AFTER        = 0x02U, /*!< After callback. */



107    POWER_MANAGER_CALLBACK_BEFORE_AFTER = 0x03U  /*!< Before-After callback. */



108} power_manager_callback_type_t;



109



110/*!



111 * @brief Callback-specific data.



112 *



113 * Pointer to data of this type is passed during callback registration. The pointer is



114 * part of the power_manager_callback_user_config_t structure and is passed to the callback during



115 * power mode change notifications.



116 * Implements power_manager_callback_data_t_Class



117 */



118typedef void power_manager_callback_data_t;



119



120/*!



121 * @brief Power mode user configuration structure.



122 *



123 * This structure defines power mode with additional power options.



124 * This structure is implementation-defiend. Please refer to actual definition based on the



125 * underlying HAL (SMC, MC_ME etc). Applications may define multiple power modes and



126 * switch between them. A list of all defined power modes is passed to the Power manager during



127 * initialization as an array of references to structures of this type (see POWER_SYS_Init()).



128 * Power modes can be switched by calling POWER_SYS_SetMode(), which takes as argument the index



129 * of the reqested power mode in the list passed during manager initialization.



130 * The power mode currently in use can be retrieved by calling POWER_SYS_GetLastMode(), which provides



131 * the index of the current power mode, or by calling POWER_SYS_GetLastModeConfig(), which provides a



132 * pointer to the configuration structure of the current power mode.



133 * The members of the power mode configuration structure depend on power options available



134 * for a specific chip, and includes at least the power mode. The available power modes are chip-specific.



135 * See power_manager_modes_t defined in the underlying HAL for a list of all supported modes.



136 */



137



138/*!



139 * @brief Power notification structure passed to registered callback function



140 *



141 * Implements power_manager_notify_struct_t_Class



142 */



143typedef struct



144{



145    power_manager_user_config_t * targetPowerConfigPtr; /*!< Pointer to target power configuration */



146    uint8_t targetPowerConfigIndex;    /*!< Target power configuration index. */



147    power_manager_policy_t policy;     /*!< Power mode transition policy.          */



148    power_manager_notify_t notifyType; /*!< Power mode notification type.          */



149} power_manager_notify_struct_t;



150



151/*!



152 * @brief Callback prototype.



153 *



154 * Declaration of callback. It is common for all registered callbacks.



155 * Function pointer of this type is part of power_manager_callback_user_config_t callback



156 * configuration structure.



157 * Depending on the callback type, the callback function is invoked during power mode change



158 * (see POWER_SYS_SetMode()) before the mode change, after it, or in both cases to notify about



159 * the change progress (see power_manager_callback_type_t). When called, the type of the notification



160 * is passed as parameter along with a pointer to power mode configuration structure



161 * (see power_manager_notify_struct_t) and any data passed during the callback registration (see



162 * power_manager_callback_data_t).



163 * When notified before a mode change, depending on the power mode change policy (see



164 * power_manager_policy_t) the callback may deny the mode change by returning any error code other



165 * than STATUS_SUCCESS (see POWER_SYS_SetMode()).



166 * @param notify Notification structure.



167 * @param dataPtr Callback data. Pointer to the data passed during callback registration. Intended to



168 *  pass any driver or application data such as internal state information.



169 * @return An error code or STATUS_SUCCESS.



170 * Implements power_manager_callback_t_Class



171 */



172typedef status_t (* power_manager_callback_t)(power_manager_notify_struct_t * notify,



173                                              power_manager_callback_data_t * dataPtr);



174



175/*!



176 * @brief callback configuration structure



177 *



178 * This structure holds configuration of callbacks passed



179 * to the Power manager during its initialization.



180 * Structures of this type are expected to be statically



181 * allocated.



182 * This structure contains following application-defined data:



183 *  callback - pointer to the callback function



184 *  callbackType - specifies when the callback is called



185 *  callbackData - pointer to the data passed to the callback



186 * Implements power_manager_callback_user_config_t_Class



187 */



188typedef struct



189{



190    power_manager_callback_t callbackFunction;



191    power_manager_callback_type_t callbackType;



192    power_manager_callback_data_t * callbackData;



193} power_manager_callback_user_config_t;



194



195/*!



196 * @brief Power manager internal state structure.



197 *



198 * Power manager internal structure. Contains data necessary for Power manager proper



199 * functionality. Stores references to registered power mode configurations,



200 * callbacks, and other internal data.



201 * This structure is statically allocated and initialized by POWER_SYS_Init().



202 * Implements power_manager_state_t_Class



203 */



204typedef struct



205{



206    power_manager_user_config_t * (*configs)[];             /*!< Pointer to power configure table.*/



207    uint8_t configsNumber;                                  /*!< Number of power configurations */



208    power_manager_callback_user_config_t * (*staticCallbacks)[]; /*!< Pointer to callback table. */



209    uint8_t staticCallbacksNumber;                          /*!< Max. number of callback configurations */



210    uint8_t errorCallbackIndex;                             /*!< Index of callback returns error. */



211    uint8_t currentConfig;                                  /*!< Index of current configuration.  */



212} power_manager_state_t;

222/*!



223 * @brief Power manager initialization for operation.



224 *



225 * This function initializes the Power manager and its run-time state structure.



226 * Pointer to an array of Power mode configuration structures needs to be passed



227 * as a parameter along with a parameter specifying its size. At least one power mode



228 * configuration is required. Optionally, pointer to the array of predefined



229 * callbacks can be passed with its corresponding size parameter.



230 * For details about callbacks, refer to the power_manager_callback_user_config_t.



231 * As Power manager stores only pointers to arrays of these structures, they need



232 * to exist and be valid for the entire life cycle of Power manager.



233 *



234 * @param[in] powerConfigsPtr A pointer to an array of pointers to all power



235 *  configurations which will be handled by Power manager.



236 * @param[in] configsNumber Number of power configurations. Size of powerConfigsPtr



237 *  array.



238 * @param[in] callbacksPtr A pointer to an array of pointers to callback configurations.



239 *  If there are no callbacks to register during Power manager initialization, use NULL value.



240 * @param[in] callbacksNumber Number of registered callbacks. Size of callbacksPtr



241 *  array.



242 * @return An error code or STATUS_SUCCESS.



243 */



244status_t POWER_SYS_Init(power_manager_user_config_t * (*powerConfigsPtr)[],



245                        uint8_t configsNumber,



246                        power_manager_callback_user_config_t * (*callbacksPtr)[],



247                        uint8_t callbacksNumber);

256/*!



257 * @brief This function configures the power mode.



258 *



259 * This function switches to one of the defined power modes. Requested mode number is passed



260 * as an input parameter. This function notifies all registered callback functions before



261 * the mode change (using  POWER_MANAGER_CALLBACK_BEFORE set as callback type parameter),



262 * sets specific power options defined in the power mode configuration and enters the specified



263 * mode. In case of run modes (for example, Run, Very low power run, or High speed run), this function



264 * also invokes all registered callbacks after the mode change (using POWER_MANAGER_CALLBACK_AFTER).



265 * In case of sleep or deep sleep modes, if the requested mode is not exited through



266 * a reset, these notifications are sent after the core wakes up.



267 * Callbacks are invoked in the following order: All registered callbacks are notified



268 * ordered by index in the callbacks array (see callbacksPtr parameter of POWER_SYS_Init()).



269 * The same order is used for before and after switch notifications.



270 * The notifications before the power mode switch can be used to obtain confirmation about



271 * the change from registered callbacks. If any registered callback denies the power



272 * mode change, further execution of this function depends on mode change policy: the mode



273 * change is either forced(POWER_MANAGER_POLICY_FORCIBLE) or aborted(POWER_MANAGER_POLICY_AGREEMENT).



274 * When mode change is forced, the results of the before switch notifications are ignored. If



275 * agreement is requested, in case any callback returns an error code then further



276 * before switch notifications are cancelled and all already notified callbacks are re-invoked



277 * with POWER_MANAGER_CALLBACK_AFTER set as callback type parameter. The index of the callback



278 * which returned error code during pre-switch notifications is stored and can be obtained by using



279 * POWER_SYS_GetErrorCallback(). Any error codes during callbacks re-invocation (recover phase) are ignored.



280 * POWER_SYS_SetMode() returns an error code denoting the phase in which a callback failed.



281 * It is possible to enter any mode supported by the processor. Refer to the chip reference manual



282 * for the list of available power modes. If it is necessary to switch into an intermediate power mode prior to



283 * entering the requested mode (for example, when switching from Run into Very low power wait through Very low



284 * power run mode), then the intermediate mode is entered without invoking the callback mechanism.



285 *



286 * @param[in] powerModeIndex Requested power mode represented as an index into



287 * array of user-defined power mode configurations passed to the POWER_SYS_Init().



288 * @param[in] policy Transaction policy



289 * @return An error code or STATUS_SUCCESS.



290 */



291status_t POWER_SYS_SetMode(uint8_t powerModeIndex,



292                           power_manager_policy_t policy);



293
