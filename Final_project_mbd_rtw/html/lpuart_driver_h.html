41/*! @brief Type of LPUART transfer (based on interrupts or DMA).



42 *



43 * Implements : lpuart_transfer_type_t_Class



44 */



45typedef enum {



46  LPUART_USING_DMA = 0,                /*!< The driver will use DMA to perform UART transfer */



47  LPUART_USING_INTERRUPTS              /*!< The driver will use interrupts to perform UART transfer */



48} lpuart_transfer_type_t;



49



50/*! @brief LPUART number of bits in a character



51 *



52 * Implements : lpuart_bit_count_per_char_t_Class



53 */



54typedef enum {



55  LPUART_8_BITS_PER_CHAR = 0x0U,       /*!< 8-bit data characters */



56  LPUART_9_BITS_PER_CHAR = 0x1U,       /*!< 9-bit data characters */



57  LPUART_10_BITS_PER_CHAR = 0x2U       /*!< 10-bit data characters */



58} lpuart_bit_count_per_char_t;



59



60/*! @brief LPUART parity mode



61 *



62 * Implements : lpuart_parity_mode_t_Class



63 */



64typedef enum {



65  LPUART_PARITY_DISABLED = 0x0U,       /*!< parity disabled */



66  LPUART_PARITY_EVEN = 0x2U,           /*!< parity enabled, type even, bit setting: PE|PT = 10 */



67  LPUART_PARITY_ODD = 0x3U             /*!< parity enabled, type odd,  bit setting: PE|PT = 11 */



68} lpuart_parity_mode_t;



69



70/*! @brief LPUART number of stop bits



71 *



72 * Implements : lpuart_stop_bit_count_t_Class



73 */



74typedef enum {



75  LPUART_ONE_STOP_BIT = 0x0U,          /*!< one stop bit */



76  LPUART_TWO_STOP_BIT = 0x1U           /*!< two stop bits */



77} lpuart_stop_bit_count_t;



78



79/*!



80 * @brief Runtime state of the LPUART driver.



81 *



82 * Note that the caller provides memory for the driver state structures during



83 * initialization because the driver does not statically allocate memory.



84 *



85 * Implements : lpuart_state_t_Class



86 */



87typedef struct {



88  const uint8_t * txBuff;              /*!< The buffer of data being sent.*/



89  uint8_t * rxBuff;                    /*!< The buffer of received data.*/



90  volatile uint32_t txSize;            /*!< The remaining number of bytes to be transmitted. */



91  volatile uint32_t rxSize;            /*!< The remaining number of bytes to be received. */



92  volatile bool isTxBusy;              /*!< True if there is an active transmit.*/



93  volatile bool isRxBusy;              /*!< True if there is an active receive.*/



94  volatile bool isTxBlocking;          /*!< True if transmit is blocking transaction. */



95  volatile bool isRxBlocking;          /*!< True if receive is blocking transaction. */



96  lpuart_bit_count_per_char_t bitCountPerChar;/*!< number of bits in a char (8/9/10) */



97  uart_callback_t rxCallback;          /*!< Callback to invoke for data receive



98                                          Note: when the transmission is interrupt based, the callback



99                                          is being called upon receiving a byte; when DMA transmission



100                                          is used, the bytes are copied to the rx buffer by the DMA engine



101                                          and the callback is called when all the bytes have been transferred. */



102  void * rxCallbackParam;              /*!< Receive callback parameter pointer.*/



103  uart_callback_t txCallback;          /*!< Callback to invoke for data send



104                                          Note: when the transmission is interrupt based, the callback



105                                          is being called upon sending a byte; when DMA transmission



106                                          is used, the bytes are copied to the tx buffer by the DMA engine



107                                          and the callback is called when all the bytes have been transferred. */



108  void * txCallbackParam;              /*!< Transmit callback parameter pointer.*/



109  lpuart_transfer_type_t transferType; /*!< Type of LPUART transfer (interrupt/dma based) */



110



111#if FEATURE_LPUART_HAS_DMA_ENABLE



112



113  uint8_t rxDMAChannel;                /*!< DMA channel number for DMA-based rx. */



114  uint8_t txDMAChannel;                /*!< DMA channel number for DMA-based tx. */



115



116#endif



117



118  semaphore_t rxComplete;              /*!< Synchronization object for blocking Rx timeout condition */



119  semaphore_t txComplete;              /*!< Synchronization object for blocking Tx timeout condition */



120  volatile status_t transmitStatus;    /*!< Status of last driver transmit operation */



121  volatile status_t receiveStatus;     /*!< Status of last driver receive operation */



122} lpuart_state_t;



123



124/*! @brief LPUART configuration structure



125 *



126 * Implements : lpuart_user_config_t_Class



127 */



128typedef struct {



129  uint32_t baudRate;                   /*!< LPUART baud rate */



130  lpuart_parity_mode_t parityMode;     /*!< parity mode, disabled (default), even, odd */



131  lpuart_stop_bit_count_t stopBitCount;/*!< number of stop bits, 1 stop bit (default) or 2 stop bits */



132  lpuart_bit_count_per_char_t bitCountPerChar;/*!< number of bits in a character (8-default, 9 or 10);



133                                                 for 9/10 bits chars, users must provide appropriate buffers



134                                                 to the send/receive functions (bits 8/9 in subsequent bytes);



135                                                 for DMA transmission only 8-bit char is supported. */



136  lpuart_transfer_type_t transferType; /*!< Type of LPUART transfer (interrupt/dma based) */



137  uint8_t rxDMAChannel;                /*!< Channel number for DMA rx channel.



138                                          If DMA mode isn't used this field will be ignored. */



139  uint8_t txDMAChannel;                /*!< Channel number for DMA tx channel.



140                                          If DMA mode isn't used this field will be ignored. */



141} lpuart_user_config_t;

188   * @param function The LPUART receive callback function.



189   * @param rxBuff The receive buffer used inside IRQHandler. This buffer must be kept as long as the callback is alive.

191   * @return Former LPUART receive callback function pointer.



192   */

243  /*!



244   * @brief Sends data out through the LPUART module using a non-blocking method.



245   *  This enables an a-sync method for transmitting data. When used with



246   *  a non-blocking receive, the LPUART can perform a full duplex operation.



247   *  Non-blocking  means that the function returns immediately.



248   *  The application has to get the transmit status to know when the transmit is complete.



249   *



250   * @param instance  LPUART instance number



251   * @param txBuff  source buffer containing 8-bit data chars to send



252   * @param txSize  the number of bytes to send



253   * @return STATUS_SUCCESS if successful;



254   *         STATUS_BUSY if a resource is busy;



255   *         STATUS_ERROR if an error occurred

258    const uint8_t * txBuff,



259    uint32_t txSize);

261  /*!



262   * @brief Returns whether the previous transmit is complete.



263   *



264   * @param instance  LPUART instance number



265   * @param bytesRemaining Pointer to value that is populated with the number of bytes that



266   *        have been sent in the active transfer



267   *        @note In DMA mode, this parameter may not be accurate, in case the transfer completes



268   *              right after calling this function; in this edge-case, the parameter will reflect



269   *              the initial transfer size, due to automatic reloading of the major loop count



270   *              in the DMA transfer descriptor.



271   * @return The transmit status.



272   * @retval STATUS_SUCCESS The transmit has completed successfully.



273   * @retval STATUS_BUSY The transmit is still in progress. @a bytesTransmitted will be



274   *     filled with the number of bytes that have been transmitted so far.



275   * @retval STATUS_UART_ABORTED The transmit was aborted.



276   * @retval STATUS_TIMEOUT A timeout was reached.



277   * @retval STATUS_ERROR An error occurred.



278   */



279  status_t LPUART_DRV_GetTransmitStatus(uint32_t instance, uint32_t *



280    bytesRemaining);

301   *         STATUS_BUSY if a resource is busy;



302   *         STATUS_UART_FRAMING_ERROR if a framing error occurred;

305   *         STATUS_UART_RX_OVERRUN if an overrun error occurred;



306   *         STATUS_ERROR if a DMA error occurred;



307   */
