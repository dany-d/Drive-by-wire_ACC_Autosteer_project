32

35  /******************************************************************************

46#define ICACHE_ENABLE                  0



47#endif



48



49  /* Value of the external crystal or oscillator clock frequency in Hz */



50#ifndef CPU_XTAL_CLK_HZ



51#define CPU_XTAL_CLK_HZ                8000000u



52#endif



53



54  /* Value of the fast internal oscillator clock frequency in Hz  */



55#ifndef CPU_INT_FAST_CLK_HZ



56#define CPU_INT_FAST_CLK_HZ            48000000u



57#endif



58



59  /* Default System clock value */



60#ifndef DEFAULT_SYSTEM_CLOCK



61#define DEFAULT_SYSTEM_CLOCK           48000000u



62#endif



63



64  /**



65   * @brief System clock frequency (core clock)



66   *



67   * The system clock frequency supplied to the SysTick timer and the processor



68   * core clock. This variable can be used by the user application to setup the



69   * SysTick timer or configure other parameters. It may also be used by debugger to



70   * query the frequency of the debug timer or configure the trace clock speed



71   * SystemCoreClock is initialized with a correct predefined value.



72   */



73  extern uint32_t SystemCoreClock;



74



75  /**



76   * @brief Setup the SoC.



77   *



78   * This function disables the watchdog, enables FPU.



79   * if the corresponding feature macro is enabled.



80   * SystemInit is called from startup_device file.



81   */



82  void SystemInit(void);



83



84  /**



85   * @brief Updates the SystemCoreClock variable.



86   *



87   * It must be called whenever the core clock is changed during program



88   * execution. SystemCoreClockUpdate() evaluates the clock register settings and calculates



89   * the current core clock.



90   * This function must be called when user does not want to use clock manager component.



91   * If clock manager is used, the CLOCK_SYS_GetFreq function must be used with CORE_CLOCK



92   * parameter.



93   *



94   */



95  void SystemCoreClockUpdate(void);



96



97  /**



98   * @brief Initiates a system reset.



99   *



100   * This function is used to initiate a system reset



101   */



102  void SystemSoftwareReset(void);
