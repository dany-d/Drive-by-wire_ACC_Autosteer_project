227  /*!

230   * This function enables or disables the LPUART transmitter, based on the



231   * parameter received.



232   *



233   *



234   * @param base LPUART base pointer.



235   * @param enable Enable(true) or disable(false) transmitter.



236   */



237  static inline void LPUART_SetTransmitterCmd(LPUART_Type * base, bool enable)



238  {



239    base->CTRL = (base->CTRL & ~LPUART_CTRL_TE_MASK) | ((enable ? 1UL : 0UL) <<



240      LPUART_CTRL_TE_SHIFT);



241



242    /* Wait for the register write operation to complete */



243    while ((bool)((base->CTRL & LPUART_CTRL_TE_MASK) != 0U) != enable) {



244    }



245  }

267  /*!



268   * @brief Sets the LPUART baud rate modulo divisor.



269   *



270   * This function sets the LPUART baud rate modulo divisor.



271   *



272   *



273   * @param base LPUART base pointer.



274   * @param baudRateDivisor The baud rate modulo division "SBR"



275   */



276  static inline void LPUART_SetBaudRateDivisor(LPUART_Type * base, uint32_t



277    baudRateDivisor)



278  {



279



280#ifdef DEV_ERROR_DETECT



281



282    DEV_ASSERT((baudRateDivisor <= 0x1FFFU) && (baudRateDivisor >= 1U));



283



284#endif



285



286    uint32_t baudRegValTemp;



287    baudRegValTemp = base->BAUD;



288    baudRegValTemp &= ~(LPUART_BAUD_SBR_MASK);



289



290    /* Removed the shift operation as the SBR field position is zero; shifting with 0 violates MISRA */



291    baudRegValTemp |= baudRateDivisor & LPUART_BAUD_SBR_MASK;



292    base->BAUD = baudRegValTemp;



293  }

302   * @return The baud rate modulo division "SBR"



303   */



304  static inline uint16_t LPUART_GetBaudRateDivisor(const LPUART_Type * base)



305  {



306    return ((uint16_t)((base->BAUD & LPUART_BAUD_SBR_MASK) >>



307                       LPUART_BAUD_SBR_SHIFT));



308  }

312  /*!



313   * @brief Sets the LPUART baud rate oversampling ratio



314   *



315   * This function sets the LPUART baud rate oversampling ratio.



316   * (Note: Feature available on select LPUART instances used together with baud rate programming)



317   * The oversampling ratio should be set between 4x (00011) and 32x (11111). Writing



318   * an invalid oversampling ratio results in an error and is set to a default



319   * 16x (01111) oversampling ratio.



320   * Disable the transmitter/receiver before calling this function.



321   *



322   *



323   * @param base LPUART base pointer.



324   * @param overSamplingRatio The oversampling ratio "OSR"



325   */



326  static inline void LPUART_SetOversamplingRatio(LPUART_Type * base, uint32_t



327    overSamplingRatio)



328  {



329



330#ifdef DEV_ERROR_DETECT



331



332    DEV_ASSERT(overSamplingRatio <= 0x1FU);



333



334#endif



335

337    baudRegValTemp = base->BAUD;



338    baudRegValTemp &= ~(LPUART_BAUD_OSR_MASK);

666  /*!

668   *



669   * This function configures the break char length.



670   * In some LPUART instances, the user should disable the transmitter before calling



671   * this function. Generally, this may be applied to all LPUARTs to ensure safe operation.



672   *



673   * @param base LPUART base pointer



674   * @param length   LPUART break character length setting: 0 - minimum 10-bit times (default),



675   *                   1 - minimum 13-bit times

677  static inline void LPUART_SetBreakCharTransmitLength(LPUART_Type * base,
