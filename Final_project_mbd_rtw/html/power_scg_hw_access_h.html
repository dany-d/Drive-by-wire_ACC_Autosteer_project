56  /*!



57   * @brief Set the SPLL clock source was enabled or disabled



58   *



59   * This function sets SPLL clock source enable or disabled.



60   * This one is used in the switching very low power mode sequence.



61   *



62   * @param[in] enable This variable select the SPLL available or not.



63   */



64  static inline void SCG_SetEnableSPLL(bool enable)



65  {



66    uint32_t regValue = SCG->SPLLCSR;



67    regValue &= ~SCG_SPLLCSR_SPLLEN_MASK;



68    if (enable == true) {



69      regValue |= SCG_SPLLCSR_SPLLEN(1U);



70    } else {



71      regValue |= SCG_SPLLCSR_SPLLEN(0U);



72    }



73



74    SCG->SPLLCSR = regValue;



75  }



76



77  /*!



78   * @brief Get status of SPLL clock source was enabled or disabled



79   *



80   * This function gets SPLL clock source enable or disabled.



81   * This one is used in the switching very low power mode sequence.



82   *



83   * @return true  : SPLL enabled



84     false : SPLL disable



85   */



86  static inline bool SCG_GetEnableSPLL(void)



87  {



88    uint32_t regValue = SCG->SPLLCSR;



89    regValue = (regValue & SCG_SPLLCSR_SPLLEN_MASK) >> SCG_SPLLCSR_SPLLEN_SHIFT;



90    return (regValue == 0U) ? false : true;



91  }
