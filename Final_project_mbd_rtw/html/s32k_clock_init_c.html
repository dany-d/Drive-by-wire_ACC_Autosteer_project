241void UpdateSystickConfig(uint32_t ticks)



242{



243  S32_SysTick->RVR = S32_SysTick_RVR_RELOAD(ticks - 1);



244  S32_SysTick->CVR = 0;



245  S32_SysTick->CSR = S32_SysTick_CSR_ENABLE(1u) | S32_SysTick_CSR_TICKINT(1u) |



246    S32_SysTick_CSR_CLKSOURCE(1u);



247}
