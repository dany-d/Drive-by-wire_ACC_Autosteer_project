96static inline void osif_UpdateTickConfig(void)



97{



98  uint32_t core_freq = 0u;



99



100  /* Get the correct name of the core clock */



101  clock_names_t coreclk = CORE_CLK;



102  status_t clk_status = CLOCK_SYS_GetFreq(coreclk, &core_freq);



103  DEV_ASSERT(clk_status == STATUS_SUCCESS);



104  DEV_ASSERT(core_freq > 0u);



105  (void)clk_status;



106



107  /* For Cortex-M0 devices the systick counter is initialized with an undefined



108     value, so make sure to initialize it to 0 before starting */



109  S32_SysTick->CSR = S32_SysTick_CSR_ENABLE(0u);



110  S32_SysTick->RVR = S32_SysTick_RVR_RELOAD(core_freq / 1000u);



111  S32_SysTick->CVR = S32_SysTick_CVR_CURRENT(0U);



112  S32_SysTick->CSR = S32_SysTick_CSR_ENABLE(1u) | S32_SysTick_CSR_TICKINT(1u) |



113    S32_SysTick_CSR_CLKSOURCE(1u);



114}

135static inline void osif_UpdateTickConfig(void)



136{



137  uint32_t tick_freq = 0u;



138  PIT_Type * base = OSIF_PIT;



139



140  /* get the clock frequency for the timer and compute ticks for 1 ms */



141  status_t clk_status = CLOCK_SYS_GetFreq(PITRTI0_CLK, &tick_freq);



142  DEV_ASSERT(clk_status == STATUS_SUCCESS);



143  DEV_ASSERT(tick_freq > 0u);



144  (void)clk_status;



145  uint32_t tick_1ms = tick_freq / 1000u;



146



147  /* setup timer and enable interrupt */



148  base->MCR &= ~PIT_MCR_MDIS(1u);      /* make sure module is started */



149  base->MCR |= PIT_MCR_FRZ(1u);        /* stop in Debug */



150  base->TIMER[OSIF_PIT_CHAN_ID].LDVAL = tick_1ms;



151  base->TIMER[OSIF_PIT_CHAN_ID].TCTRL |= PIT_TCTRL_TEN(1u) | PIT_TCTRL_TIE(1u);



152  const IRQn_Type pitIrqId[PIT_INSTANCE_COUNT][PIT_IRQS_CH_COUNT] = PIT_IRQS;



153  INT_SYS_InstallHandler(pitIrqId[0U][OSIF_PIT_CHAN_ID], OSIF_PIT_IRQHandler,



154    NULL);



155  INT_SYS_EnableIRQ(pitIrqId[0U][OSIF_PIT_CHAN_ID]);



156}

190/*FUNCTION**********************************************************************



191 *



192 * Function Name : OSIF_TimeDelay



193 * Description   : This function blocks execution for a number of milliseconds.



194 *



195 * Implements : OSIF_TimeDelay_baremetal_Activity



196 *END**************************************************************************/



197void OSIF_TimeDelay(const uint32_t delay)



198{



199  osif_UpdateTickConfig();



200  uint32_t start = osif_GetCurrentTickCount();



201  uint32_t crt_ticks = osif_GetCurrentTickCount();



202  uint32_t delta = crt_ticks - start;



203  uint32_t delay_ticks = MSEC_TO_TICK(delay);



204  while (delta < delay_ticks) {



205    crt_ticks = osif_GetCurrentTickCount();



206    delta = crt_ticks - start;



207  }



208}

210/*FUNCTION**********************************************************************



211 *



212 * Function Name : OSIF_GetMilliseconds



213 * Description   : This function returns the number of miliseconds elapsed since



214 *                  starting the internal timer. To initialize the internal timer



215 *                  (Systick) in bare-metal, call either OSIF_TimeDelay or



216 *                  OSIF_SemaWait functions. Calling OSIF_TimeDelay(0) will initialize



217 *                  the timer without any side-effects (no delay).



218 *



219 * Implements : OSIF_GetMilliseconds_baremetal_Activity



220 *END**************************************************************************/



221uint32_t OSIF_GetMilliseconds(void)



222{



223  /*



224   * Please make sure the timer is initialized before calling this function.



225   * For example, calling OSIF_TimeDelay(0) ensures that the timer is initialized



226   * without any other side-effects. If OSIF_TimeDelay or OSIF_SemaWait functions



227   * have been called, the timer is already initialized.



228   */



229  return osif_GetCurrentTickCount();   /* This assumes that 1 tick = 1 millisecond */



230}

286/*FUNCTION**********************************************************************



287 *



288 * Function Name : OSIF_SemaWait



289 * Description   : This function performs the 'wait' (decrement) operation on a semaphore.



290 *      When timeout value is 0, it's the equivalent of TryWait - try to decrement but return



291 *      immediately if it fails (counter is 0).



292 *



293 * Implements : OSIF_SemaWait_baremetal_Activity



294 *END**************************************************************************/



295status_t OSIF_SemaWait(semaphore_t * const pSem,



296  const uint32_t timeout)



297{



298  DEV_ASSERT(pSem != NULL);



299  status_t osif_ret_code = STATUS_SUCCESS;



300  osif_UpdateTickConfig();



301  if ((timeout == 0u)) {



302    /* when the timeout is 0 the wait operation is the equivalent of try_wait,



303       meaning that if the semaphore is 0 return immediately with an error code



304     */



305    if (*pSem == 0u) {



306      osif_ret_code = STATUS_TIMEOUT;



307    }



308  } else {



309    /* timeout is not 0 */



310    uint32_t timeoutTicks;



311    if (timeout == OSIF_WAIT_FOREVER) {



312      timeoutTicks = OSIF_WAIT_FOREVER;



313    } else {



314      /* Convert timeout from milliseconds to ticks. */



315      timeoutTicks = MSEC_TO_TICK(timeout);



316    }



317



318    uint32_t start = osif_GetCurrentTickCount();



319    uint32_t end = (uint32_t)(start + timeoutTicks);



320    uint32_t max = end - start;



321    while (*pSem == 0u) {



322      uint32_t crt_ticks = osif_GetCurrentTickCount();



323      uint32_t delta = crt_ticks - start;



324      if ((timeoutTicks != OSIF_WAIT_FOREVER) && (delta > max)) {



325        /* Timeout occured, stop waiting and return fail code */



326        osif_ret_code = STATUS_TIMEOUT;



327        break;



328      }



329    }



330  }



331



332  if (osif_ret_code == STATUS_SUCCESS) {



333    osif_DisableIrqGlobal();



334    --(*pSem);



335    osif_EnableIrqGlobal();



336  }



337



338  return osif_ret_code;



339}

341/*FUNCTION**********************************************************************



342 *



343 * Function Name : OSIF_SemaPost



344 * Description   : This function performs the 'post' (increment) operation on a semaphore.



345 *



346 * Implements : OSIF_SemaPost_baremetal_Activity



347 *END**************************************************************************/



348status_t OSIF_SemaPost(semaphore_t * const pSem)



349{



350  DEV_ASSERT(pSem != NULL);



351  status_t osif_ret_code = STATUS_SUCCESS;



352  osif_DisableIrqGlobal();



353  if (*pSem != 255u) {



354    ++(*pSem);



355  } else {



356    osif_ret_code = STATUS_ERROR;



357  }



358



359  osif_EnableIrqGlobal();



360  return osif_ret_code;



361}
