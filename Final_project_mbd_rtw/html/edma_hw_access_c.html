28/*FUNCTION**********************************************************************



29 *



30 * Function Name : EDMA_Init



31 * Description   : Initializes eDMA module to known state.



32 *END**************************************************************************/



33void EDMA_Init(DMA_Type * base)



34{



35  uint8_t i;



36



37  /* Clear the bit of CR register */



38#ifdef FEATURE_DMA_HWV3



39



40  uint32_t regValTemp;



41  regValTemp = base->MP_CSR;



42  regValTemp &= ~(DMA_MP_CSR_EBW_MASK);



43  regValTemp &= ~(DMA_MP_CSR_EDBG_MASK);



44  regValTemp &= ~(DMA_MP_CSR_ERCA_MASK);



45  regValTemp &= ~(DMA_MP_CSR_HAE_MASK);



46  regValTemp &= ~(DMA_MP_CSR_HALT_MASK);



47  regValTemp &= ~(DMA_MP_CSR_ECL_MASK);



48  regValTemp &= ~(DMA_MP_CSR_EMI_MASK);



49  regValTemp &= ~(DMA_MP_CSR_ECX_MASK);



50  regValTemp &= ~(DMA_MP_CSR_CX_MASK);



51  regValTemp &= ~(DMA_MP_CSR_ACTIVE_ID_MASK);



52  regValTemp &= ~(DMA_MP_CSR_ACTIVE_MASK);



53  base->MP_CSR = regValTemp;



54  for (i = 0; i < FEATURE_DMA_CHANNELS; i++) {



55    base->CH_GRPRI[i] = 0;



56  }



57



58#else



59



60  uint32_t regValTemp;



61  regValTemp = base->CR;



62  regValTemp &= ~(DMA_CR_CLM_MASK);



63  regValTemp &= ~(DMA_CR_CX_MASK);



64  regValTemp &= ~(DMA_CR_ECX_MASK);



65  regValTemp &= ~(DMA_CR_EDBG_MASK);



66  regValTemp &= ~(DMA_CR_EMLM_MASK);



67  regValTemp &= ~(DMA_CR_ERCA_MASK);



68  base->CR = regValTemp;



69



70#endif



71



72  for (i = 0; i < FEATURE_DMA_CHANNELS; i++) {



73    EDMA_TCDClearReg(base, i);



74  }



75}

77/*FUNCTION**********************************************************************



78 *



79 * Function Name : EDMA_CancelTransfer



80 * Description   : Cancels the remaining data transfer.



81 *END**************************************************************************/



82void EDMA_CancelTransfer(DMA_Type * base)



83{



84  uint32_t regValTemp;



85



86#ifdef FEATURE_DMA_HWV3



87



88  regValTemp = base->MP_CSR;



89  regValTemp &= ~(DMA_MP_CSR_CX_MASK);



90  regValTemp |= DMA_MP_CSR_CX(1U);



91  base->MP_CSR = regValTemp;



92  while ((base->MP_CSR & DMA_MP_CSR_CX_MASK) == DMA_MP_CSR_CX_MASK) {



93  }



94



95#else



96



97  regValTemp = base->CR;



98  regValTemp &= ~(DMA_CR_CX_MASK);



99  regValTemp |= DMA_CR_CX(1U);



100  base->CR = regValTemp;



101  while (((base->CR & DMA_CR_CX_MASK) >> DMA_CR_CX_SHIFT) != 0UL) {



102  }



103



104#endif



105



106}

242/*FUNCTION**********************************************************************



243 *



244 * Function Name : EDMA_TCDClearReg



245 * Description   : Set registers to 0 for hardware TCD of eDMA channel.



246 *END**************************************************************************/



247void EDMA_TCDClearReg(DMA_Type * base, uint8_t channel)



248{



249



250#ifdef FEATURE_DMA_HWV3



251



252  base->TCD[channel].CH_CSR = 0U;



253  base->TCD[channel].CH_ES |= DMA_TCD_CH_ES_ERR_MASK;



254  base->TCD[channel].CH_INT |= DMA_TCD_CH_INT_INT_MASK;



255  base->TCD[channel].CH_SBR = 0U;



256  base->TCD[channel].NBYTES.MLOFFNO = 0U;



257



258#else



259



260  base->TCD[channel].NBYTES.MLNO = 0U;



261



262#endif



263



264  base->TCD[channel].SADDR = 0U;



265  base->TCD[channel].SOFF = 0;



266  base->TCD[channel].ATTR = 0U;



267  base->TCD[channel].SLAST = 0;



268  base->TCD[channel].DADDR = 0U;



269  base->TCD[channel].DOFF = 0;



270  base->TCD[channel].CITER.ELINKNO = 0U;



271  base->TCD[channel].DLASTSGA = 0;



272  base->TCD[channel].CSR = 0U;



273  base->TCD[channel].BITER.ELINKNO = 0U;



274}

301/*FUNCTION**********************************************************************



302 *



303 * Function Name : EDMA_TCDSetNbytes



304 * Description   : Configures the nbytes for eDMA channel.



305 *END**************************************************************************/



306void EDMA_TCDSetNbytes(DMA_Type * base, uint8_t channel, uint32_t nbytes)



307{



308



309#ifdef DEV_ERROR_DETECT



310



311  DEV_ASSERT(channel < FEATURE_DMA_CHANNELS);



312



313#endif



314



315#ifdef FEATURE_DMA_HWV3



316



317  uint32_t smloe = (base->TCD[channel].NBYTES.MLOFFNO &



318                    DMA_TCD_NBYTES_MLOFFNO_SMLOE_MASK) >>



319    DMA_TCD_NBYTES_MLOFFNO_SMLOE_SHIFT;



320  uint32_t dmloe = (base->TCD[channel].NBYTES.MLOFFNO &



321                    DMA_TCD_NBYTES_MLOFFNO_DMLOE_MASK) >>



322    DMA_TCD_NBYTES_MLOFFNO_DMLOE_SHIFT;



323  if ((smloe == 0UL) && (dmloe == 0UL)) {



324    base->TCD[channel].NBYTES.MLOFFNO = (nbytes &



325      DMA_TCD_NBYTES_MLOFFNO_NBYTES_MASK);



326  } else {



327    uint32_t regValTemp;



328    regValTemp = base->TCD[channel].NBYTES.MLOFFYES;



329    regValTemp &= ~(DMA_TCD_NBYTES_MLOFFYES_NBYTES_MASK);



330    regValTemp |= DMA_TCD_NBYTES_MLOFFYES_NBYTES(nbytes);



331    base->TCD[channel].NBYTES.MLOFFYES = regValTemp;



332  }



333



334#else



335



336  if (((base->CR & DMA_CR_EMLM_MASK) >> DMA_CR_EMLM_SHIFT) != 0UL) {



337    bool mlOffNo = false;



338    if (((base->TCD[channel].NBYTES.MLOFFNO & DMA_TCD_NBYTES_MLOFFNO_SMLOE_MASK)



339         >> DMA_TCD_NBYTES_MLOFFNO_SMLOE_SHIFT) == 0UL) {



340      if (((base->TCD[channel].NBYTES.MLOFFNO &



341            DMA_TCD_NBYTES_MLOFFNO_DMLOE_MASK) >>



342           DMA_TCD_NBYTES_MLOFFNO_DMLOE_SHIFT) == 0UL) {



343        base->TCD[channel].NBYTES.MLOFFNO = (nbytes &



344          DMA_TCD_NBYTES_MLOFFNO_NBYTES_MASK);



345        mlOffNo = true;



346      }



347    }



348



349    if (!mlOffNo) {



350      uint32_t regValTemp;



351      regValTemp = base->TCD[channel].NBYTES.MLOFFYES;



352      regValTemp &= ~(DMA_TCD_NBYTES_MLOFFYES_NBYTES_MASK);



353      regValTemp |= DMA_TCD_NBYTES_MLOFFYES_NBYTES(nbytes);



354      base->TCD[channel].NBYTES.MLOFFYES = regValTemp;



355    }



356  } else {



357    base->TCD[channel].NBYTES.MLNO = nbytes;



358  }



359



360#endif



361



362}

364/*FUNCTION**********************************************************************



365 *



366 * Function Name : EDMA_TCDSetMinorLoopOffset



367 * Description   : Configures the minor loop offset for the TCD.



368 *END**************************************************************************/



369void EDMA_TCDSetMinorLoopOffset(DMA_Type * base, uint8_t channel, int32_t offset)



370{



371



372#ifdef DEV_ERROR_DETECT



373



374  DEV_ASSERT(channel < FEATURE_DMA_CHANNELS);



375



376#endif



377



378#ifdef FEATURE_DMA_HWV3



379



380  uint32_t smloe = (base->TCD[channel].NBYTES.MLOFFNO &



381                    DMA_TCD_NBYTES_MLOFFNO_SMLOE_MASK) >>



382    DMA_TCD_NBYTES_MLOFFNO_SMLOE_SHIFT;



383  uint32_t dmloe = (base->TCD[channel].NBYTES.MLOFFNO &



384                    DMA_TCD_NBYTES_MLOFFNO_DMLOE_MASK) >>



385    DMA_TCD_NBYTES_MLOFFNO_DMLOE_SHIFT;



386  if ((smloe != 0UL) || (dmloe != 0UL)) {



387    uint32_t regValTemp;



388    regValTemp = base->TCD[channel].NBYTES.MLOFFYES;



389    regValTemp &= ~(DMA_TCD_NBYTES_MLOFFYES_MLOFF_MASK);



390    regValTemp |= DMA_TCD_NBYTES_MLOFFYES_MLOFF(offset);



391    base->TCD[channel].NBYTES.MLOFFYES = regValTemp;



392  }



393



394#else



395



396  if (((base->CR & DMA_CR_EMLM_MASK) >> DMA_CR_EMLM_SHIFT) != 0UL) {



397    bool mlOffNo = false;



398    if (((base->TCD[channel].NBYTES.MLOFFNO & DMA_TCD_NBYTES_MLOFFNO_SMLOE_MASK)



399         >> DMA_TCD_NBYTES_MLOFFNO_SMLOE_SHIFT) != 0UL) {



400      mlOffNo = true;



401    }



402



403    if (((base->TCD[channel].NBYTES.MLOFFNO & DMA_TCD_NBYTES_MLOFFNO_DMLOE_MASK)



404         >> DMA_TCD_NBYTES_MLOFFNO_DMLOE_SHIFT) != 0UL) {



405      mlOffNo = true;



406    }



407



408    if (mlOffNo) {



409      uint32_t regValTemp;



410      regValTemp = base->TCD[channel].NBYTES.MLOFFYES;



411      regValTemp &= ~(DMA_TCD_NBYTES_MLOFFYES_MLOFF_MASK);



412      regValTemp |= DMA_TCD_NBYTES_MLOFFYES_MLOFF(offset);



413      base->TCD[channel].NBYTES.MLOFFYES = regValTemp;



414    }



415  }



416



417#endif



418



419}

440/*FUNCTION**********************************************************************



441 *



442 * Function Name : EDMA_TCDSetChannelMinorLink



443 * Description   : Set Channel minor link for hardware TCD.



444 *END**************************************************************************/



445void EDMA_TCDSetChannelMinorLink(



446  DMA_Type * base, uint8_t channel, uint32_t linkChannel, bool enable)



447{



448



449#ifdef DEV_ERROR_DETECT



450



451  DEV_ASSERT(channel < FEATURE_DMA_CHANNELS);



452  DEV_ASSERT(linkChannel < FEATURE_DMA_CHANNELS);



453



454#endif



455



456#ifdef FEATURE_DMA_HWV3



457



458  uint16_t regValTemp;



459  if (enable) {



460    regValTemp = base->TCD[channel].BITER.ELINKYES;



461    regValTemp &= (uint16_t)~(DMA_TCD_BITER_ELINKYES_ELINK_MASK);



462    regValTemp |= (uint16_t)DMA_TCD_BITER_ELINKYES_ELINK(1);



463    regValTemp &= (uint16_t)~(DMA_TCD_BITER_ELINKYES_LINKCH_MASK);



464    regValTemp |= (uint16_t)DMA_TCD_BITER_ELINKYES_LINKCH(linkChannel);



465    base->TCD[channel].BITER.ELINKYES = regValTemp;



466    regValTemp = base->TCD[channel].CITER.ELINKYES;



467    regValTemp &= (uint16_t)~(DMA_TCD_CITER_ELINKYES_ELINK_MASK);



468    regValTemp |= (uint16_t)DMA_TCD_CITER_ELINKYES_ELINK(1);



469    regValTemp &= (uint16_t)~(DMA_TCD_CITER_ELINKYES_LINKCH_MASK);



470    regValTemp |= (uint16_t)DMA_TCD_CITER_ELINKYES_LINKCH(linkChannel);



471    base->TCD[channel].CITER.ELINKYES = regValTemp;



472  } else {



473    regValTemp = base->TCD[channel].BITER.ELINKNO;



474    regValTemp &= (uint16_t)~(DMA_TCD_BITER_ELINKYES_ELINK_MASK);



475    regValTemp |= (uint16_t)DMA_TCD_BITER_ELINKYES_ELINK(0);



476    regValTemp &= (uint16_t)~(DMA_TCD_BITER_ELINKYES_LINKCH_MASK);



477    regValTemp |= (uint16_t)DMA_TCD_BITER_ELINKYES_LINKCH(linkChannel);



478    base->TCD[channel].BITER.ELINKNO = regValTemp;



479    regValTemp = base->TCD[channel].CITER.ELINKNO;



480    regValTemp &= (uint16_t)~(DMA_TCD_CITER_ELINKYES_ELINK_MASK);



481    regValTemp |= (uint16_t)DMA_TCD_CITER_ELINKYES_ELINK(0);



482    regValTemp &= (uint16_t)~(DMA_TCD_CITER_ELINKYES_LINKCH_MASK);



483    regValTemp |= (uint16_t)DMA_TCD_CITER_ELINKYES_LINKCH(linkChannel);



484    base->TCD[channel].CITER.ELINKNO = regValTemp;



485  }



486



487#else



488



489  uint16_t regValTemp;



490  regValTemp = base->TCD[channel].BITER.ELINKYES;



491  regValTemp &= (uint16_t)~(DMA_TCD_BITER_ELINKYES_ELINK_MASK);



492  regValTemp |= (uint16_t)DMA_TCD_BITER_ELINKYES_ELINK(enable ? 1UL : 0UL);



493  base->TCD[channel].BITER.ELINKYES = regValTemp;



494  regValTemp = base->TCD[channel].CITER.ELINKYES;



495  regValTemp &= (uint16_t)~(DMA_TCD_CITER_ELINKYES_ELINK_MASK);



496  regValTemp |= (uint16_t)DMA_TCD_CITER_ELINKYES_ELINK(enable ? 1UL : 0UL);



497  base->TCD[channel].CITER.ELINKYES = regValTemp;



498  if (enable) {



499    regValTemp = base->TCD[channel].BITER.ELINKYES;



500    regValTemp &= (uint16_t)~(DMA_TCD_BITER_ELINKYES_LINKCH_MASK);



501    regValTemp |= (uint16_t)DMA_TCD_BITER_ELINKYES_LINKCH(linkChannel);



502    base->TCD[channel].BITER.ELINKYES = regValTemp;



503    regValTemp = base->TCD[channel].CITER.ELINKYES;



504    regValTemp &= (uint16_t)~(DMA_TCD_CITER_ELINKYES_LINKCH_MASK);



505    regValTemp |= (uint16_t)DMA_TCD_CITER_ELINKYES_LINKCH(linkChannel);



506    base->TCD[channel].CITER.ELINKYES = regValTemp;



507  }



508



509#endif



510



511}

513/*FUNCTION**********************************************************************



514 *



515 * Function Name : EDMA_TCD_TCDSetMajorCount



516 * Description   : Sets the major iteration count according to minor loop



517 * channel link setting.



518 *END**************************************************************************/



519void EDMA_TCDSetMajorCount(DMA_Type * base, uint8_t channel, uint32_t count)



520{



521



522#ifdef DEV_ERROR_DETECT



523



524  DEV_ASSERT(channel < FEATURE_DMA_CHANNELS);



525



526#endif



527



528  uint16_t regValTemp = 0;



529  if ((base->TCD[channel].BITER.ELINKNO & DMA_TCD_BITER_ELINKNO_ELINK_MASK) ==



530      DMA_TCD_BITER_ELINKNO_ELINK_MASK) {



531    regValTemp = base->TCD[channel].BITER.ELINKYES;



532    regValTemp &= (uint16_t)~(DMA_TCD_BITER_ELINKYES_BITER_MASK);



533    regValTemp |= (uint16_t)DMA_TCD_BITER_ELINKYES_BITER(count);



534    base->TCD[channel].BITER.ELINKYES = regValTemp;



535    regValTemp = base->TCD[channel].CITER.ELINKYES;



536    regValTemp &= (uint16_t)~(DMA_TCD_CITER_ELINKYES_CITER_LE_MASK);



537    regValTemp |= (uint16_t)DMA_TCD_CITER_ELINKYES_CITER_LE(count);



538    base->TCD[channel].CITER.ELINKYES = regValTemp;



539  } else {



540    regValTemp = base->TCD[channel].BITER.ELINKNO;



541    regValTemp &= (uint16_t)~(DMA_TCD_BITER_ELINKNO_BITER_MASK);



542    regValTemp |= (uint16_t)DMA_TCD_BITER_ELINKNO_BITER(count);



543    base->TCD[channel].BITER.ELINKNO = regValTemp;



544    regValTemp = base->TCD[channel].CITER.ELINKNO;



545    regValTemp &= (uint16_t)~(DMA_TCD_CITER_ELINKNO_CITER_MASK);



546    regValTemp |= (uint16_t)DMA_TCD_CITER_ELINKNO_CITER(count);



547    base->TCD[channel].CITER.ELINKNO = regValTemp;



548  }



549}

551/*FUNCTION**********************************************************************



552 *



553 * Function Name : EDMA_TCDGetCurrentMajorCount



554 * Description   : Gets the current major iteration count according to minor



555 * loop channel link setting.

557uint32_t EDMA_TCDGetCurrentMajorCount(const DMA_Type * base, uint8_t channel)



558{



559



560#ifdef DEV_ERROR_DETECT



561

577



578  return (uint32_t) result;
