54  const power_mode_stat_t mode,

57/*! Timeout used for waiting to set new mode */



58#define SMC_TIMEOUT                    1000U

64/*FUNCTION**********************************************************************



65 *



66 * Function Name : SMC_SetPowerMode



67 * Description   : Configure the power mode



68 * This function will configure the power mode control for any run, stop and



69 * stop submode if needed. It will also configure the power options for specific



70 * power mode. Application should follow the proper procedure to configure and



71 * switch power mode between the different run and stop mode. Refer to reference



72 * manual for the proper procedure and supported power mode that can be configured



73 * and switch between each other. Refer to smc_power_mode_config_t for required



74 * parameters to configure the power mode and the supported options. Other options



75 * may need to configure through the hal driver individaully. Refer to hal driver



76 * header for details.



77 *



78 *END**************************************************************************/



79status_t SMC_SetPowerMode(SMC_Type * const baseAddr,



80  const smc_power_mode_config_t * const powerModeConfig)



81{



82  status_t retCode;



83  smc_stop_mode_t stopMode;



84  power_manager_modes_t powerModeName = powerModeConfig->powerModeName;



85



86  /* Branch based on power mode name*/



87  switch (powerModeName)



88  {



89   case POWER_MANAGER_RUN:



90    /* Set to RUN mode. */



91    SMC_SetRunModeControl(baseAddr, SMC_RUN);



92



93    /* Wait for stat change */



94    if (!SMC_WaitForStatChange(baseAddr, STAT_RUN, SMC_TIMEOUT)) {



95      /* Timeout for power mode change expired. */



96      retCode = STATUS_MCU_TRANSITION_FAILED;



97      break;



98    }



99



100    retCode = STATUS_SUCCESS;



101    break;



102



103   case POWER_MANAGER_VLPR:



104    /* "Very-Low-Power Modes" is allowed? */



105    DEV_ASSERT(SMC_GetProtectionMode(baseAddr, ALLOW_VLP));



106



107    /* Set power mode to VLPR*/



108    SMC_SetRunModeControl(baseAddr, SMC_VLPR);



109



110    /* Wait for stat change */



111    if (!SMC_WaitForStatChange(baseAddr, STAT_VLPR, SMC_TIMEOUT)) {



112      /* Timeout for power mode change expired. */



113      retCode = STATUS_MCU_TRANSITION_FAILED;



114      break;



115    }



116



117    retCode = STATUS_SUCCESS;



118    break;



119



120#if FEATURE_SMC_HAS_HIGH_SPEED_RUN_MODE



121



122   case POWER_MANAGER_HSRUN:



123    /* "High Speed Mode" is allowed? */



124    DEV_ASSERT(SMC_GetProtectionMode(baseAddr, ALLOW_HSRUN));



125



126    /* Set power mode to HSRUN */



127    SMC_SetRunModeControl(baseAddr, SMC_HSRUN);



128



129    /* Wait for stat change */



130    if (!SMC_WaitForStatChange(baseAddr, STAT_HSRUN, SMC_TIMEOUT)) {



131      /* Timeout for power mode change expired. */



132      retCode = STATUS_MCU_TRANSITION_FAILED;



133      break;



134    }



135



136    retCode = STATUS_SUCCESS;



137    break;



138



139#endif                                 /* if FEATURE_SMC_HAS_HIGH_SPEED_RUN_MODE */



140



141#if FEATURE_SMC_HAS_WAIT_VLPW



142



143   case POWER_MANAGER_WAIT:



144    /* Fall-through */



145   case POWER_MANAGER_VLPW:



146    /* "Very-Low-Power Modes" is allowed? */



147    if (powerModeName == POWER_MODE_VLPW) {



148      DEV_ASSERT(SMC_GetProtectionMode(baseAddr, ALLOW_VLP));



149    }



150



151    /* Clear the SLEEPDEEP bit to disable deep sleep mode - WAIT */



152    S32_SCB->SCR &= ~S32_SCB_SCR_SLEEPDEEP_MASK;



153



154    /* Cpu is going into sleep state */



155    STANDBY();



156    retCode = STATUS_SUCCESS;



157    break;



158



159#endif                                 /* if FEATURE_SMC_HAS_WAIT_VLPW */



160



161   case POWER_MANAGER_STOP1:



162    /* Fall-through */



163   case POWER_MANAGER_STOP2:



164    /* Fall-through */



165   case POWER_MANAGER_VLPS:



166    if ((powerModeName == POWER_MANAGER_STOP1) || (powerModeName ==



167         POWER_MANAGER_STOP2)) {



168      stopMode = SMC_STOP;



169



170#if FEATURE_SMC_HAS_STOPO



171



172      SMC_SetStopOption(baseAddr, powerModeConfig->stopOptionValue);



173



174#endif



175



176#if FEATURE_SMC_HAS_PSTOPO



177



178      SMC_SetPStopOption(baseAddr, powerModeConfig->pStopOptionValue);



179



180#endif



181



182    } else {



183      stopMode = SMC_VLPS;



184



185      /* "Very-Low-Power Modes" is allowed? */



186      DEV_ASSERT(SMC_GetProtectionMode(baseAddr, ALLOW_VLP));



187    }



188



189    /* Set power mode to specified STOP mode*/



190    SMC_SetStopModeControl(baseAddr, stopMode);



191



192    /* Set the SLEEPDEEP bit to enable deep sleep mode (STOP)*/



193    S32_SCB->SCR |= S32_SCB_SCR_SLEEPDEEP_MASK;



194



195    /* Cpu is going into deep sleep state */



196    STANDBY();



197    retCode = STATUS_SUCCESS;



198    break;



199



200   default:



201    retCode = STATUS_UNSUPPORTED;



202    break;



203  }



204



205  return retCode;



206}

208/*FUNCTION**********************************************************************



209 *



210 * Function Name : SMC_SetProtectionMode



211 * Description   : Configure all power mode protection settings



212 * This function will configure the power mode protection settings for



213 * supported power mode on the specified chip family. The available power modes



214 * are defined in smc_power_mode_protection_config_t. Application should provide



215 * the protect settings for all supported power mode on the chip and also this



216 * should be done at early system level init stage. Refer to reference manual



217 * for details. This register can only write once after power reset. So either



218 * use this function or use the individual set function if you only have single



219 * option to set.



220 *



221 *END**************************************************************************/



222void SMC_SetProtectionMode(SMC_Type * const baseAddr,



223  const smc_power_mode_protection_config_t * const protectConfig)



224{



225  /* Initialize the setting */



226  uint32_t regValue = 0U;



227



228  /* Check configurations for each mode and combine the setting together */



229  if (protectConfig->vlpProt) {



230    regValue |= SMC_PMPROT_AVLP(1);



231  }



232



233#if FEATURE_SMC_HAS_HIGH_SPEED_RUN_MODE



234



235  if (protectConfig->hsrunProt) {



236    regValue |= SMC_PMPROT_AHSRUN(1);



237  }



238



239#endif



240



241  /* Write once into PMPROT register*/



242  baseAddr->PMPROT = regValue;



243}



244



245/*FUNCTION**********************************************************************



246 *



247 * Function Name : SMC_GetProtectionMode



248 * Description   : Get the current power mode protection setting



249 * This function will get the current power mode protection settings for



250 * a specified power mode.



251 *



252 *END**************************************************************************/



253bool SMC_GetProtectionMode(const SMC_Type * const baseAddr,



254  const power_modes_protect_t protect)



255{



256  bool retValue;



257  uint32_t regValue = (uint32_t)baseAddr->PMPROT;



258



259  /* Check the mode range */



260  DEV_ASSERT(protect < ALLOW_MAX);



261



262  /* Branch according to the mode and read the setting */



263  switch (protect)



264  {



265   case ALLOW_VLP:



266    regValue = (regValue & SMC_PMPROT_AVLP_MASK) >> SMC_PMPROT_AVLP_SHIFT;



267    break;



268



269#if FEATURE_SMC_HAS_HIGH_SPEED_RUN_MODE



270



271   case ALLOW_HSRUN:



272    regValue = (regValue & SMC_PMPROT_AHSRUN_MASK) >> SMC_PMPROT_AHSRUN_SHIFT;



273    break;



274



275#endif



276



277   default:



278    /* Invalid command */



279    regValue = 0UL;



280    break;



281  }



282



283  retValue = (regValue == 0UL) ? false : true;



284  return retValue;



285}



286



287/*FUNCTION**********************************************************************



288 * Function Name : SMC_WaitForStatChange



289 * Description   : Internal function used by SMC_SetPowerMode function



290 * to wait until the state is changed or timeout expires



291 *



292 * return power mode status change



293 *                - true: power mode has been changed successfully



294 *                - false: timeout expired, power mode has not been changed



295 *END**************************************************************************/



296static bool SMC_WaitForStatChange(const SMC_Type * const baseAddr,



297  const power_mode_stat_t mode,



298  const uint32_t timeout)



299{



300  uint32_t i;



301  bool retValue;



302  for (i = 0U; i < timeout; i++) {



303    if (mode == SMC_GetPowerModeStatus(baseAddr)) {



304      /* Power mode has been changed successfully */



305      break;



306    }



307  }



308



309  /* If i greater or equal to timeout, then timeout expired(the power mode has not been changed)*/



310  retValue = (i < timeout);



311  return retValue;



312}
