130/*FUNCTION**********************************************************************



131 *



132 * Function Name : EDMA_DRV_Init



133 * Description   : Initializes the eDMA module.



134 *



135 * Implements    : EDMA_DRV_Init_Activity



136 *END**************************************************************************/



137status_t EDMA_DRV_Init(edma_state_t *edmaState,



138  const edma_user_config_t *userConfig,



139  edma_chn_state_t * const chnStateArray[],



140  const edma_channel_config_t * const chnConfigArray[],



141  uint32_t chnCount)



142{



143  uint32_t index = 0U;



144  DMA_Type *edmaRegBase = NULL;



145  IRQn_Type irqNumber = NotAvail_IRQn;



146  status_t edmaStatus = STATUS_SUCCESS;



147  status_t chnInitStatus = STATUS_SUCCESS;



148



149#ifdef DEV_ERROR_DETECT



150



151  uint32_t freq = 0U;



152  status_t clockManagerStatus = STATUS_SUCCESS;



153



154#endif



155



156  /* Check the state and configuration structure pointers are valid */



157  DEV_ASSERT((edmaState != NULL) && (userConfig != NULL));



158



159  /* Check the module has not already been initialized */



160  DEV_ASSERT(s_virtEdmaState == NULL);



161



162#ifdef DEV_ERROR_DETECT



163



164  /* Check that eDMA and DMAMUX modules are clock gated on */



165  for (index = 0U; index < (uint32_t)DMA_INSTANCE_COUNT; index++) {



166    clockManagerStatus = CLOCK_SYS_GetFreq(s_edmaClockNames[index], &freq);



167    DEV_ASSERT(clockManagerStatus == STATUS_SUCCESS);



168  }



169



170  for (index = 0U; index < (uint32_t)DMAMUX_INSTANCE_COUNT; index++) {



171    clockManagerStatus = CLOCK_SYS_GetFreq(s_dmamuxClockNames[index], &freq);



172    DEV_ASSERT(clockManagerStatus == STATUS_SUCCESS);



173  }



174



175#endif



176



177  /* Save the runtime state structure for the driver */



178  s_virtEdmaState = edmaState;



179



180  /* Clear the state structure. */



181  EDMA_DRV_ClearStructure((uint8_t *)s_virtEdmaState, sizeof(edma_state_t));



182



183  /* Init all DMA instances */



184  for (index = 0U; index < (uint32_t)DMA_INSTANCE_COUNT; index++) {



185    edmaRegBase = s_edmaBase[index];



186



187    /* Init eDMA module on hardware level. */



188    EDMA_Init(edmaRegBase);



189



190#ifdef FEATURE_DMA_HWV3



191



192    /* Set arbitration mode */



193    EDMA_SetChannelArbitrationMode(edmaRegBase, userConfig->chnArbitration);



194



195#else



196



197    /* Set arbitration mode */



198    EDMA_SetChannelArbitrationMode(edmaRegBase, userConfig->chnArbitration);



199



200#if (FEATURE_DMA_CHANNEL_GROUP_COUNT > 0x1U)



201



202    EDMA_SetGroupArbitrationMode(edmaRegBase, userConfig->groupArbitration);



203    EDMA_SetGroupPriority(edmaRegBase, userConfig->groupPriority);



204



205#endif



206#endif



207



208    /* Set 'Halt on error' configuration */



209    EDMA_SetHaltOnErrorCmd(edmaRegBase, userConfig->haltOnError);



210  }



211



212#if defined FEATURE_DMA_HAS_ERROR_IRQ



213



214  /* Enable the error interrupts for eDMA module. */



215  for (index = 0U; index < (uint32_t)FEATURE_DMA_VIRTUAL_ERROR_INTERRUPT_LINES;



216       index++) {



217    /* Enable channel interrupt ID. */



218    irqNumber = s_edmaErrIrqId[index];



219    INT_SYS_EnableIRQ(irqNumber);



220  }



221



222#endif



223



224  /* Register all edma channel interrupt handlers into vector table. */



225  for (index = 0U; index < (uint32_t)



226       FEATURE_DMA_VIRTUAL_CHANNELS_INTERRUPT_LINES; index++) {



227    /* Enable channel interrupt ID. */



228    irqNumber = s_edmaIrqId[index];



229    INT_SYS_EnableIRQ(irqNumber);



230  }



231



232  /* Initialize all DMAMUX instances */



233  for (index = 0U; index < (uint32_t)DMAMUX_INSTANCE_COUNT; index++) {



234    DMAMUX_Init(s_dmaMuxBase[index]);



235  }



236



237  /* Initialize the channels based on configuration list */



238  if ((chnStateArray != NULL) && (chnConfigArray != NULL)) {



239    for (index = 0U; index < chnCount; index++) {



240      chnInitStatus = EDMA_DRV_ChannelInit(chnStateArray[index],



241        chnConfigArray[index]);



242      if (chnInitStatus != STATUS_SUCCESS) {



243        edmaStatus = chnInitStatus;



244      }



245    }



246  }



247



248  return edmaStatus;



249}

298/*FUNCTION**********************************************************************



299 *



300 * Function Name : EDMA_DRV_ChannelInit



301 * Description   : Initialize EDMA channel.



302 *



303 * Implements    : EDMA_DRV_ChannelInit_Activity



304 *END**************************************************************************/



305status_t EDMA_DRV_ChannelInit(edma_chn_state_t *edmaChannelState,



306  const edma_channel_config_t *edmaChannelConfig)



307{



308  /* Check the state and configuration structure pointers are valid */



309  DEV_ASSERT((edmaChannelState != NULL) && (edmaChannelConfig != NULL));



310



311  /* Check if the module is initialized */



312  DEV_ASSERT(s_virtEdmaState != NULL);



313



314  /* Check if the channel defined by user in the channel configuration structure is valid */



315  DEV_ASSERT(edmaChannelConfig->virtChnConfig < FEATURE_DMA_VIRTUAL_CHANNELS);



316



317  /* Get DMA instance from virtual channel */



318  uint8_t dmaInstance = (uint8_t)FEATURE_DMA_VCH_TO_INSTANCE



319    (edmaChannelConfig->virtChnConfig);



320



321  /* Get DMA channel from virtual channel */



322  uint8_t dmaChannel = (uint8_t)FEATURE_DMA_VCH_TO_CH



323    (edmaChannelConfig->virtChnConfig);



324



325  /* Request the channel */



326  status_t retStatus = EDMA_DRV_RequestChannel(edmaChannelConfig->virtChnConfig,



327    edmaChannelConfig->source, edmaChannelState);



328  if (retStatus == STATUS_SUCCESS) {



329    /* Load corresponding DMA instance */



330    DMA_Type *edmaRegBase = s_edmaBase[dmaInstance];



331



332    /* Set virtual channel status to normal */



333    s_virtEdmaState->virtChnState[edmaChannelConfig->virtChnConfig]->status =



334      EDMA_CHN_NORMAL;



335



336#ifdef FEATURE_DMA_ENGINE_STALL



337



338    /* Configure the DMA Engine to stall for a number of cycles after each R/W */



339    EDMA_TCDSetEngineStall(edmaRegBase, dmaChannel, EDMA_ENGINE_STALL_4_CYCLES);



340



341#endif



342



343#ifdef FEATURE_DMA_HWV3



344



345    /* Put the channel in a priority group, as defined in configuration */



346    EDMA_SetChannelPriorityGroup(edmaRegBase, dmaChannel,



347      edmaChannelConfig->groupPriority);



348



349#endif



350



351    /* Set the channel priority, as defined in the configuration, only if fixed arbitration mode is selected */



352    if ((EDMA_GetChannelArbitrationMode(edmaRegBase) ==



353         EDMA_ARBITRATION_FIXED_PRIORITY) &&



354        (edmaChannelConfig->channelPriority != EDMA_CHN_DEFAULT_PRIORITY)) {



355      EDMA_SetChannelPriority(edmaRegBase, dmaChannel,



356        edmaChannelConfig->channelPriority);



357    }



358



359    /* Install the user callback */



360    retStatus = EDMA_DRV_InstallCallback(edmaChannelConfig->virtChnConfig,



361      edmaChannelConfig->callback, edmaChannelConfig->callbackParam);



362  }



363



364  return retStatus;



365}

388/*FUNCTION**********************************************************************



389 *



390 * Function Name : EDMA_DRV_RequestChannel



391 * Description   : Request an eDMA channel.



392 *



393 *END**************************************************************************/



394static status_t EDMA_DRV_RequestChannel(uint8_t virtualChannel,



395  dma_request_source_t reqSrc,



396  edma_chn_state_t *reqChn)



397{



398  /* Get DMA instance from virtual channel */



399  uint8_t dmaInstance = (uint8_t)FEATURE_DMA_VCH_TO_INSTANCE(virtualChannel);



400



401  /* Get DMA channel from virtual channel */



402  uint8_t dmaChannel = (uint8_t)FEATURE_DMA_VCH_TO_CH(virtualChannel);



403



404  /* Get DMAMUX instance serving this channel */



405  uint8_t dmaMuxInstance = (uint8_t)FEATURE_DMAMUX_REQ_SRC_TO_INSTANCE(reqSrc);



406



407  /* Get request source index for the corresponding DMAMUX instance */



408  uint8_t sourceDmamuxChannel = (uint8_t)FEATURE_DMAMUX_REQ_SRC_TO_CH((uint32_t)



409    reqSrc);



410



411  /* Get DMAMUX channel for the selected request source */



412  uint8_t dmaMuxChannel = (uint8_t)FEATURE_DMAMUX_DMA_CH_TO_CH(dmaChannel);



413



414  /* Base addresses for eDMA and DMAMUX */



415  DMA_Type *edmaRegBase = s_edmaBase[dmaInstance];



416  DMAMUX_Type *dmaMuxRegBase = s_dmaMuxBase[dmaMuxInstance];



417



418  /* Check the channel has not already been allocated */



419  DEV_ASSERT(s_virtEdmaState->virtChnState[virtualChannel] == NULL);



420



421  /* Channel allocation */



422  s_virtEdmaState->virtChnState[virtualChannel] = reqChn;



423



424  /* Reset the channel state structure to default value. */



425  EDMA_DRV_ClearStructure((uint8_t *)reqChn, sizeof(edma_chn_state_t));



426



427  /* Init the channel state structure to the allocated channel number. */



428  reqChn->virtChn = virtualChannel;



429



430  /* Enable error interrupt for this channel */



431  EDMA_SetErrorIntCmd(edmaRegBase, dmaChannel, true);



432



433  /* Configure the DMAMUX for edma channel */



434  DMAMUX_SetChannelCmd(dmaMuxRegBase, dmaMuxChannel, false);



435  DMAMUX_SetChannelSource(dmaMuxRegBase, dmaMuxChannel, sourceDmamuxChannel);



436  DMAMUX_SetChannelCmd(dmaMuxRegBase, dmaMuxChannel, true);



437



438  /* Clear the TCD registers for this channel */



439  EDMA_TCDClearReg(edmaRegBase, dmaChannel);



440  return STATUS_SUCCESS;



441}

553/*FUNCTION**********************************************************************



554 *



555 * Function Name : EDMA_DRV_ConfigSingleBlockTransfer



556 * Description   : Configures a DMA single block transfer.



557 *



558 * Implements    : EDMA_DRV_ConfigSingleBlockTransfer_Activity



559 *END**************************************************************************/



560status_t EDMA_DRV_ConfigSingleBlockTransfer(uint8_t virtualChannel,



561  edma_transfer_type_t type,



562  uint32_t srcAddr,



563  uint32_t destAddr,



564  edma_transfer_size_t transferSize,



565  uint32_t dataBufferSize)



566{



567  /* Check that virtual channel number is valid */



568  DEV_ASSERT(virtualChannel < FEATURE_DMA_VIRTUAL_CHANNELS);



569



570  /* Check that eDMA module is initialized */



571  DEV_ASSERT(s_virtEdmaState != NULL);



572



573  /* Check that virtual channel is initialized */



574  DEV_ASSERT(s_virtEdmaState->virtChnState[virtualChannel] != NULL);



575



576#ifdef DEV_ERROR_DETECT



577



578  /* Check if the value passed for 'transferSize' is valid */



579  DEV_ASSERT(EDMA_DRV_ValidTransferSize(transferSize));



580



581#endif



582



583  /* Get DMA instance from virtual channel */



584  uint8_t dmaInstance = (uint8_t)FEATURE_DMA_VCH_TO_INSTANCE(virtualChannel);



585



586  /* Get DMA channel from virtual channel*/



587  uint8_t dmaChannel = (uint8_t)FEATURE_DMA_VCH_TO_CH(virtualChannel);



588  DMA_Type *edmaRegBase = s_edmaBase[dmaInstance];



589  uint8_t transferOffset;



590  status_t retStatus = STATUS_SUCCESS;



591



592  /* Compute the transfer offset, based on transfer size.



593   * The number of bytes transferred in each source read/destination write



594   * is obtained with the following formula:



595   *    source_read_size = 2^SSIZE



596   *    destination_write_size = 2^DSIZE



597   */



598  transferOffset = (uint8_t) (1U << ((uint8_t)transferSize));



599



600  /* The number of bytes to be transferred (buffer size) must



601   * be a multiple of the source read/destination write size



602   */



603  if ((dataBufferSize % transferOffset) != 0U) {



604    retStatus = STATUS_ERROR;



605  }



606



607  if (retStatus == STATUS_SUCCESS) {



608    /* Clear transfer control descriptor for the current channel */



609    EDMA_TCDClearReg(edmaRegBase, dmaChannel);



610



611#ifdef FEATURE_DMA_ENGINE_STALL



612



613    /* Configure the DMA Engine to stall for a number of cycles after each R/W */



614    EDMA_TCDSetEngineStall(edmaRegBase, dmaChannel, EDMA_ENGINE_STALL_4_CYCLES);



615



616#endif



617



618    /* Configure source and destination addresses */



619    EDMA_TCDSetSrcAddr(edmaRegBase, dmaChannel, srcAddr);



620    EDMA_TCDSetDestAddr(edmaRegBase, dmaChannel, destAddr);



621



622    /* Set transfer size (1B/2B/4B/16B/32B) */



623    EDMA_TCDSetAttribute(edmaRegBase, dmaChannel, EDMA_MODULO_OFF,



624                         EDMA_MODULO_OFF, transferSize, transferSize);



625



626    /* Configure source/destination offset. */



627    switch (type)



628    {



629     case EDMA_TRANSFER_PERIPH2MEM:



630      EDMA_TCDSetSrcOffset(edmaRegBase, dmaChannel, 0);



631      EDMA_TCDSetDestOffset(edmaRegBase, dmaChannel, (int8_t) transferOffset);



632      break;



633



634     case EDMA_TRANSFER_MEM2PERIPH:



635      EDMA_TCDSetSrcOffset(edmaRegBase, dmaChannel, (int8_t) transferOffset);



636      EDMA_TCDSetDestOffset(edmaRegBase, dmaChannel, 0);



637      break;



638



639     case EDMA_TRANSFER_MEM2MEM:

780/*FUNCTION**********************************************************************



781 *



782 * Function Name : EDMA_DRV_ConfigScatterGatherTransfer



783 * Description   : Configure eDMA for scatter/gather operation



784 *



785 * Implements    : EDMA_DRV_ConfigScatterGatherTransfer_Activity



786 *END**************************************************************************/



787status_t EDMA_DRV_ConfigScatterGatherTransfer(uint8_t virtualChannel,



788  edma_software_tcd_t *stcd,



789  edma_transfer_size_t transferSize,



790  uint32_t bytesOnEachRequest,



791  const edma_scatter_gather_list_t *srcList,



792  const edma_scatter_gather_list_t *destList,



793  uint8_t tcdCount)



794{



795  /* Check that virtual channel number is valid */



796  DEV_ASSERT(virtualChannel < FEATURE_DMA_VIRTUAL_CHANNELS);



797



798  /* Check that eDMA module is initialized */



799  DEV_ASSERT(s_virtEdmaState != NULL);



800



801  /* Check that virtual channel is initialized */



802  DEV_ASSERT(s_virtEdmaState->virtChnState[virtualChannel] != NULL);



803



804  /* Check the input arrays for scatter/gather operation are valid */



805  DEV_ASSERT((stcd != NULL) && (srcList != NULL) && (destList != NULL));



806



807#ifdef DEV_ERROR_DETECT



808



809  /* Check if the value passed for 'transferSize' is valid */



810  DEV_ASSERT(EDMA_DRV_ValidTransferSize(transferSize));



811



812#endif



813



814  uint8_t i = 0U;



815  uint16_t transferOffset = 0U;



816  uint32_t stcdAlignedAddr = STCD_ADDR(stcd);



817  edma_software_tcd_t *edmaSwTcdAddr = (edma_software_tcd_t *)stcdAlignedAddr;



818  edma_loop_transfer_config_t edmaLoopConfig;



819  edma_transfer_config_t edmaTransferConfig;



820  status_t retStatus = STATUS_SUCCESS;



821



822  /* Set virtual channel status to normal */



823  s_virtEdmaState->virtChnState[virtualChannel]->status = EDMA_CHN_NORMAL;



824



825  /* Compute the transfer offset, based on transfer size.



826   * The number of bytes transferred in each source read/destination write



827   * is obtained with the following formula:



828   *    source_read_size = 2^SSIZE



829   *    destination_write_size = 2^DSIZE



830   */



831  transferOffset = (uint16_t) (1UL << ((uint16_t)transferSize));



832



833  /* The number of bytes to be transferred on each request must



834   * be a multiple of the source read/destination write size



835   */



836  if ((bytesOnEachRequest % transferOffset) != 0U) {



837    retStatus = STATUS_ERROR;



838  }



839



840  /* Clear the configuration structures before initializing them. */



841  EDMA_DRV_ClearStructure((uint8_t *)(&edmaTransferConfig), sizeof



842    (edma_transfer_config_t));



843  EDMA_DRV_ClearStructure((uint8_t *)(&edmaLoopConfig), sizeof



844    (edma_loop_transfer_config_t));



845



846  /* Configure the transfer for scatter/gather mode. */



847  edmaTransferConfig.srcLastAddrAdjust = 0;



848  edmaTransferConfig.destLastAddrAdjust = 0;



849  edmaTransferConfig.srcModulo = EDMA_MODULO_OFF;



850  edmaTransferConfig.destModulo = EDMA_MODULO_OFF;



851  edmaTransferConfig.srcTransferSize = transferSize;



852  edmaTransferConfig.destTransferSize = transferSize;



853  edmaTransferConfig.minorByteTransferCount = bytesOnEachRequest;



854  edmaTransferConfig.interruptEnable = true;



855  edmaTransferConfig.scatterGatherEnable = true;



856  edmaTransferConfig.loopTransferConfig = &edmaLoopConfig;



857  edmaTransferConfig.loopTransferConfig->srcOffsetEnable = false;



858  edmaTransferConfig.loopTransferConfig->dstOffsetEnable = false;



859  edmaTransferConfig.loopTransferConfig->minorLoopChnLinkEnable = false;



860  edmaTransferConfig.loopTransferConfig->majorLoopChnLinkEnable = false;



861



862  /* Copy scatter/gather lists to transfer configuration*/



863  for (i = 0U; (i < tcdCount) && (retStatus == STATUS_SUCCESS); i++) {



864    edmaTransferConfig.srcAddr = srcList[i].address;



865    edmaTransferConfig.destAddr = destList[i].address;



866    if ((srcList[i].length != destList[i].length) || (srcList[i].type !=



867         destList[i].type)) {



868      retStatus = STATUS_ERROR;



869      break;



870    }



871



872    edmaTransferConfig.loopTransferConfig->majorLoopIterationCount = srcList[i].



873      length/bytesOnEachRequest;



874    switch (srcList[i].type)



875    {



876     case EDMA_TRANSFER_PERIPH2MEM:



877      /* Configure Source Read. */



878      edmaTransferConfig.srcOffset = 0;



879



880      /* Configure Dest Write. */



881      edmaTransferConfig.destOffset = (int16_t) transferOffset;



882      break;



883



884     case EDMA_TRANSFER_MEM2PERIPH:



885      /* Configure Source Read. */



886      edmaTransferConfig.srcOffset = (int16_t) transferOffset;



887



888      /* Configure Dest Write. */



889      edmaTransferConfig.destOffset = 0;



890      break;



891



892     case EDMA_TRANSFER_MEM2MEM:



893      /* Configure Source Read. */



894      edmaTransferConfig.srcOffset = (int16_t) transferOffset;



895



896      /* Configure Dest Write. */



897      edmaTransferConfig.destOffset = (int16_t) transferOffset;



898      break;



899



900     case EDMA_TRANSFER_PERIPH2PERIPH:



901      /* Configure Source Read. */



902      edmaTransferConfig.srcOffset = 0;



903



904      /* Configure Dest Write. */



905      edmaTransferConfig.destOffset = 0;



906      break;



907



908     default:



909      /* This should never be reached - all the possible values have been handled. */



910      break;



911    }



912



913    /* Configure the pointer to next software TCD structure; for the last one, this address should be 0 */



914    if (i == ((uint8_t)(tcdCount - 1U))) {



915      edmaTransferConfig.scatterGatherNextDescAddr = 0U;



916    } else {



917      edma_software_tcd_t * ptNextAddr = &edmaSwTcdAddr[i];



918      edmaTransferConfig.scatterGatherNextDescAddr = ((uint32_t) ptNextAddr);



919    }



920



921    if (i == 0U) {



922      /* Push the configuration for the first descriptor to registers */



923      EDMA_DRV_PushConfigToReg(virtualChannel, &edmaTransferConfig);



924    } else {



925      /* Copy configuration to software TCD structure */



926      EDMA_DRV_PushConfigToSTCD(&edmaTransferConfig, &edmaSwTcdAddr[i - 1U]);



927    }



928  }



929



930  return retStatus;



931}
