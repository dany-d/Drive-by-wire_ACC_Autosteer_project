109/** \brief  Reverse byte order in a word.



110 */



111#if defined (__GNUC__) || defined (__ICCARM__) || defined (__ghs__) || defined (__ARMCC_VERSION)



112#define REV_BYTES_32(a, b) __asm volatile ("rev %0, %1" : "=r" (b) : "r" (a))



113#else



114#define REV_BYTES_32(a, b) (b = ((a & 0xFF000000U) >> 24U) | ((a & 0xFF0000U) >> 8U) \



115                                | ((a & 0xFF00U) << 8U) | ((a & 0xFFU) << 24U))



116#endif
