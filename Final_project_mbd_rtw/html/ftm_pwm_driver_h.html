40/*! @brief Maximum value for PWM duty cycle */



41#define FTM_MAX_DUTY_CYCLE             (0x8000U)



42



43/*! @brief Shift value which converts duty to ticks */



44#define FTM_DUTY_TO_TICKS_SHIFT        (15U)



45



46/*!



47 * @brief FlexTimer Configure type of PWM update in the duty cycle or in ticks



48 *



49 * Implements : ftm_pwm_update_option_t_Class



50 */



51typedef enum {



52  FTM_PWM_UPDATE_IN_DUTY_CYCLE = 0x00U,/*!< The type of PWM update in the duty cycle/pulse or also use in frequency update */



53  FTM_PWM_UPDATE_IN_TICKS = 0x01U      /*!< The type of PWM update in ticks */



54} ftm_pwm_update_option_t;



55



56/*!



57 * @brief FlexTimer PWM output pulse mode, high-true or low-true on match up



58 *



59 * Implements : ftm_polarity_t_Class



60 */



61typedef enum {



62  FTM_POLARITY_LOW = 0x00U,            /*!< When counter > CnV output signal is LOW */



63  FTM_POLARITY_HIGH = 0x01U            /*!< When counter > CnV output signal is HIGH */



64} ftm_polarity_t;



65



66/*!



67 * @brief FlexTimer PWM channel (n+1) polarity for combine mode



68 *



69 * Implements : ftm_second_channel_polarity_t_Class



70 */



71typedef enum {



72  FTM_MAIN_INVERTED = 0x01U,           /*!< The channel (n+1) output is the inverse of the



73                                        *   channel (n) output  */



74  FTM_MAIN_DUPLICATED = 0x00U          /*!< The channel (n+1) output is the same as the



75                                        *   channel (n) output */



76} ftm_second_channel_polarity_t;



77



78/*!



79 * @brief FlexTimer fault control



80 *



81 * Implements : ftm_fault_mode_t_Class



82 */



83typedef enum {



84  FTM_FAULT_CONTROL_DISABLED = 0x00U,  /*!< Fault control is disabled for all channels */



85  FTM_FAULT_CONTROL_MAN_EVEN = 0x01U,  /*!< Fault control is enabled for even channels



86                                        *   only (channels 0, 2, 4, and 6), and the selected



87                                        *   mode is the manual fault clearing */



88  FTM_FAULT_CONTROL_MAN_ALL = 0x02U,   /*!< Fault control is enabled for all channels,



89                                        *   and the selected mode is the manual fault clearing */



90  FTM_FAULT_CONTROL_AUTO_ALL = 0x03U   /*!< Fault control is enabled for all channels, and



91                                        *   the selected mode is the automatic fault clearing */



92} ftm_fault_mode_t;



93



94/*!



95 * @brief Select level of the channel (n) output at the beginning



96 *



97 * Implements : ftm_safe_state_polarity_t_Class



98 */



99typedef enum {



100  FTM_HIGH_TRUE_PULSE = 0x02U,         /*!< Clear the channel (n) output on match,



101                                        *   (Clear the channel (n+1) output con match in combine mode and modified combine mode) */



102  FTM_LOW_TRUE_PULSE = 0x03U           /*!< Set the channel (n) output on match,



103                                        *   (set the channel (n+1) output on match in combine mode and modified combine mode) */



104} ftm_safe_state_polarity_t;

130/*!



131 * @brief FlexTimer driver independent PWM parameter



132 *



133 * Implements : ftm_independent_ch_param_t_Class



134 */



135typedef struct {



136  uint8_t hwChannelId;                 /*!< Physical hardware channel ID */



137  ftm_polarity_t polarity;             /*!< PWM output polarity */



138  uint16_t uDutyCyclePercent;          /*!< PWM pulse width, value should be between



139                                        *   0 (0%) to FTM_MAX_DUTY_CYCLE (100%) */



140  bool enableExternalTrigger;          /*!< true: enable the generation of a trigger is used for on-chip modules



141                                        *   false: disable the generation of a trigger */



142  ftm_safe_state_polarity_t levelSelect;/*!< The selection of the channel (n) mode */



143  bool enableSecondChannelOutput;      /*!< Enable complementary mode on next channel */



144  ftm_second_channel_polarity_t secondChannelPolarity;/*!< Polarity of the channel n+1 relative to channel n */



145  bool deadTime;                       /*!< Enable/disable dead time for channel */



146} ftm_independent_ch_param_t;



147



148/*!



149 * @brief FlexTimer driver combined PWM parameter



150 *



151 * Implements : ftm_combined_ch_param_t_Class



152 */



153typedef struct {



154  uint8_t hwChannelId;                 /*!< Physical hardware channel ID for channel (n) */



155  uint16_t firstEdge;                  /*!< First edge time. This time is relative to signal period. The value for this parameter is



156                                        *   between 0 and FTM_MAX_DUTY_CYCLE(0 = 0% from period and FTM_MAX_DUTY_CYCLE = 100% from period) */



157  uint16_t secondEdge;                 /*!< Second edge time. This time is relative to signal period. The value for this parameter is



158                                        *   between 0 and FTM_MAX_DUTY_CYCLE(0 = 0% from period and FTM_MAX_DUTY_CYCLE = 100% from period) */



159  bool deadTime;                       /*!< Enable/disable dead time for channel */



160  bool enableModifiedCombine;          /*!< Enable/disable the modified combine mode for channels (n) and (n+1) */



161  ftm_polarity_t mainChannelPolarity;  /*!< Main channel polarity. For FTM_POLARITY_HIGH first output value is 0 and for



162                                        *   FTM_POLAIRTY first output value is 1 */



163  bool enableSecondChannelOutput;      /*!< Select if channel (n+1)  output is enabled/disabled for the complementary mode */



164  ftm_second_channel_polarity_t secondChannelPolarity;/*!< Select channel (n+1) polarity relative to channel (n) */



165  bool enableExternalTrigger;          /*!< The generation of the channel (n) trigger



166                                        *   true: enable the generation of a trigger on the channel (n)



167                                        *   false: disable the generation of a trigger on the channel (n) */



168  bool enableExternalTriggerOnNextChn; /*!< The generation of the channel (n+1) trigger



169                                        *   true: enable the generation of a trigger on the channel (n+1)



170                                        *   false: disable the generation of a trigger on the channel (n+1) */



171  ftm_safe_state_polarity_t levelSelect;/*!< The selection of the channel (n)  mode */



172  ftm_safe_state_polarity_t levelSelectOnNextChn;/*!< The selection of the channel (n+1) mode */



173} ftm_combined_ch_param_t;

222  /*!



223   * @brief This function updates the waveform output in PWM mode (duty cycle and phase).



224   *



225   * @param [in] instance The FTM peripheral instance number.



226   * @param [in] channel The channel number. In combined mode, the code finds the channel.



227   * @param [in] typeOfUpdate The type of PWM update in the duty cycle/pulse or in ticks.



228   * @param [in] firstEdge  Duty cycle or first edge time for PWM mode. Can take value between



229   *                       0 - FTM_MAX_DUTY_CYCLE(0 = 0% from period  and FTM_MAX_DUTY_CYCLE = 100% from period)



230   *                       Or value in ticks for the first of the PWM mode in which can have value between 0



231   *                       and ftmPeriod is stored in the state structure.



232   * @param [in] secondEdge Second edge time - only for combined mode. Can take value



233   *                       between 0 - FTM_MAX_DUTY_CYCLE(0 = 0% from period  and FTM_MAX_DUTY_CYCLE = 100% from period).



234   *                       Or value in ticks for the second of the PWM mode in which can have value between 0



235   *                       and ftmPeriod is stored in the state structure.



236   * @param [in] softwareTrigger If true a software trigger is generate to update PWM parameters.



237   * @return success



238   *        - STATUS_SUCCESS : Completed successfully.



239   *        - STATUS_ERROR : Error occurred.



240   */



241  status_t FTM_DRV_UpdatePwmChannel(uint32_t instance,



242    uint8_t channel,



243    ftm_pwm_update_option_t typeOfUpdate,



244    uint16_t firstEdge,



245    uint16_t secondEdge,



246    bool softwareTrigger);



247



248  /*!



249   * @brief This function will update the duty cycle of PWM output for multiple channels.



250   *



251   * @param [in] instance The FTM peripheral instance number.



252   * @param [in] numberOfChannels The number of channels which should be updated.



253   * @param [in] channels The list of channels which should be updated.



254   * @param [in] duty The list of duty cycles for selected channels.



255   * @param [in] softwareTrigger If true a software trigger is generate to update PWM parameters.



256   * @return success



257   *        - STATUS_SUCCESS : Completed successfully.



258   *        - STATUS_ERROR : Error occurred.



259   */



260  status_t FTM_DRV_FastUpdatePwmChannels(uint32_t instance,



261    uint8_t numberOfChannels,



262    const uint8_t * channels,



263    const uint16_t * duty,



264    bool softwareTrigger);



265



266  /*!



267   * @brief This function will update the new period in the frequency or



268   * in the counter value into mode register which modify the period of PWM signal



269   * on the channel output



270   *



271   * @param [in] instance The FTM peripheral instance number.



272   * @param [in] typeOfUpdate The type of PWM update is a period in Hz or in ticks.



273   *                          - For FTM_PWM_UPDATE_IN_DUTY_CYCLE which reuse in FTM_DRV_UpdatePwmChannel function will update in Hz.



274   *                          - For FTM_PWM_UPDATE_IN_TICKS will update in ticks.



275   * @param [in] newValue The frequency or the counter value which will select with modified value for PWM signal.



276   *                      If the type of update in the duty cycle, the newValue parameter must be value



277   *                      between 1U and maximum is the frequency of the FTM counter.



278   *                      If the type of update in ticks, the newValue parameter must be value between 1U and 0xFFFFU.



279   * @param [in] softwareTrigger If true a software trigger is generate to update PWM parameters.



280   * @return operation status



281   *        - STATUS_SUCCESS : Completed successfully.



282   *        - STATUS_ERROR : Error occurred.



283   */



284  status_t FTM_DRV_UpdatePwmPeriod(uint32_t instance,



285    ftm_pwm_update_option_t typeOfUpdate,



286    uint32_t newValue,



287    bool softwareTrigger);
