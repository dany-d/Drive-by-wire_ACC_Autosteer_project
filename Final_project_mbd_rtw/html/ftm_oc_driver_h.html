37/*!



38 * @brief FlexTimer Mode configuration for output compare mode



39 *



40 * Implements : ftm_output_compare_mode_t_Class



41 */



42typedef enum {



43  FTM_DISABLE_OUTPUT = 0x00U,          /*!< No action on output pin */



44  FTM_TOGGLE_ON_MATCH = 0x01U,         /*!< Toggle on match */



45  FTM_CLEAR_ON_MATCH = 0x02U,          /*!< Clear on match */



46  FTM_SET_ON_MATCH = 0x03U             /*!< Set on match */



47} ftm_output_compare_mode_t;



48



49/*!



50 * @brief FlexTimer input capture type of the next output compare value



51 *



52 * Implements : ftm_output_compare_update_t_Class



53 */



54typedef enum {



55  FTM_RELATIVE_VALUE = 0x00U,          /*!< Next compared value is relative to current value */



56  FTM_ABSOLUTE_VALUE = 0x01U           /*!< Next compared value is absolute */



57} ftm_output_compare_update_t;



58



59/*!



60 * @brief FlexTimer driver PWM parameters each channel in the output compare mode



61 *



62 * Implements : ftm_output_cmp_ch_param_t_Class



63 */



64typedef struct {



65  uint8_t hwChannelId;                 /*!< Physical hardware channel ID*/



66  ftm_output_compare_mode_t chMode;    /*!< Channel output mode*/



67  uint16_t comparedValue;              /*!< The compared value */



68  bool enableExternalTrigger;          /*!< true: enable the generation of a trigger is used for on-chip modules



69                                        *   false: disable the generation of a trigger */



70} ftm_output_cmp_ch_param_t;



71



72/*!



73 * @brief FlexTimer driver PWM parameters which is configured for the list of channels



74 *



75 * Implements : ftm_output_cmp_param_t_Class



76 */



77typedef struct {



78  uint8_t nNumOutputChannels;          /*!< Number of output compare channels */



79  ftm_config_mode_t mode;              /*!< FlexTimer PWM operation mode */



80  uint16_t maxCountValue;              /*!< Maximum count value in ticks */



81  ftm_output_cmp_ch_param_t * outputChannelConfig;/*!< Output compare channels configuration */



82} ftm_output_cmp_param_t;

89extern "C" {

92



93  /*!



94   * @brief Configures the FTM to generate timed pulses (Output compare mode).



95   *



96   * When the FTM counter matches the value of CnV, the channel output is changed based on what is



97   * specified in the mode argument. The signal period can be modified using



98   * param->maxCountValue. After this function when the max counter value and CnV are equal.



99   * FTM_DRV_UpdateOutputCompareChannel function can be used to change CnV value.



100   *



101   * @param [in] instance The FTM peripheral instance number.



102   * @param [in] param configuration of the output compare channels



103   * @return success



104   *        - STATUS_SUCCESS : Completed successfully.



105   *        - STATUS_ERROR : Error occurred.



106   */



107  status_t FTM_DRV_InitOutputCompare(uint32_t instance,



108    const ftm_output_cmp_param_t * param);



109



110  /*!



111   * @brief  Disables compare match output control and clears FTM timer configuration



112   *



113   * @param [in] instance The FTM peripheral instance number.



114   * @param [in] param Configuration of the output compare channel



115   * @return success



116   *        - STATUS_SUCCESS : Completed successfully.



117   *        - STATUS_ERROR : Error occurred.



118   */



119  status_t FTM_DRV_DeinitOutputCompare(uint32_t instance,



120    const ftm_output_cmp_param_t * param);



121



122  /*!



123   * @brief Sets the next compare match value based on the current counter value



124   *



125   * @param [in] instance The FTM peripheral instance number.



126   * @param [in] channel Configuration of the output compare channel



127   * @param [in] nextComparematchValue Timer value in ticks until the next compare match event should appear



128   * @param [in] update



129   *        - FTM_RELATIVE_VALUE : nextComparemantchValue will be added to current counter value



130   *        - FTM_ABSOLUTE_VALUE : nextComparemantchValue will be written in counter register as it is



131   * @param [in] softwareTrigger This parameter will be true if software trigger sync is enabled and



132   * the user want to generate a software trigger (the value from buffer will be moved to register immediate or



133   * at next loading point depending on the sync configuration). Otherwise this parameter must be false



134   * and the next compared value will be stored in buffer until a trigger signal will be received.



135   * @return success



136   *        - STATUS_SUCCESS : Completed successfully.



137   *        - STATUS_ERROR : Error occurred.



138   */



139  status_t FTM_DRV_UpdateOutputCompareChannel(uint32_t instance,



140    uint8_t channel,



141    uint16_t nextComparematchValue,



142    ftm_output_compare_update_t update,



143    bool softwareTrigger);



144



145#if defined(__cplusplus)



146



147}



148#endif
